149 GameObject* gameObj = GameObjectFactory::getInstance().create(type, engine, name);

154
155 auto components = obj["components"];
156 if (!components.is_array()) {
157 log("Object '%s' has no components to load", name.c_str());
158 } else {

160 std::string compType = comp.value("type", "");
161 if (compType.empty()) {
162 log("Error: Component missing type for object '%s'", name.c_str());
163 continue;
164 }
165 ComponentRegistry::getInstance().loadComponent(*gameObj, compType, comp);
166 }
167 }
168
169 std::string parent = obj.value("parent", "");
170 if (!parent.empty()) {
171 for(const auto& m_obj : m_objects) {
172 if (m_obj->GetComponent<NameComponent>().name == parent) {
173 gameObj->ParentTo(m_obj->GetEntity());
174 //gameObj->GetComponent<TransformComponent>().parent = m_obj->GetEntity();
175 log("Parented object '%s' to '%s'", gameObj->GetComponent<NameComponent>().name.c_str(), m_obj->GetComponent<NameComponent>().name.c_str());
176 break;
177 }
178 }
179 }
180
181 if (gameObj) {
182 m_objects.emplace_back(gameObj);
183 }
