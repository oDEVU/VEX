28 glm::mat4 matrix(const entt::registry& registry) const {
29 glm::mat4 local(1.0f);
30 local = glm::translate(local, position);
31 local = glm::rotate(local, glm::radians(rotation.x), {1, 0, 0});
32 local = glm::rotate(local, glm::radians(rotation.y), {0, 1, 0});
33 local = glm::rotate(local, glm::radians(rotation.z), {0, 0, 1});

35 if (parent != entt::null && registry.valid(parent) && registry.all_of<TransformComponent>(parent)) {
36 return registry.get<TransformComponent>(parent).matrix(registry) * local;
37 }

73 glm::vec3 getForwardVector(const entt::registry& registry) const {
74 float pitch = glm::radians(getWorldRotation(registry).x);
75 float yaw = glm::radians(getWorldRotation(registry).y);

77 return glm::normalize(glm::vec3(
