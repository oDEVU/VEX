18 std::vector<char> VulkanPipeline::readFile(const std::string& filename) {
19 log("Reading shader...");
20
21 std::string realPath = GetAssetPath(filename);
22
23 std::ifstream file(realPath, std::ios::ate | std::ios::binary);
24 if (!file.is_open()) {
25 throw_error("Failed to open shader file: " + realPath);
26 }
27 size_t fileSize = file.tellg();
28 std::vector<char> buffer(fileSize);
29 file.seekg(0);
30 file.read(buffer.data(), fileSize);

39 const std::string& vertShaderPath,
40 const std::string& fragShaderPath,
41 const VkVertexInputBindingDescription& bindingDescription,
42 const std::vector<VkVertexInputAttributeDescription>& attributeDescriptions
43 ) {
44 auto vertShaderCode = readFile(vertShaderPath);
45 auto fragShaderCode = readFile(fragShaderPath);
46
47 log("Creating shader modules...");
48 auto createShaderModule = [&](const std::vector<char>& code) {
49 VkShaderModuleCreateInfo createInfo{};
50 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
51 createInfo.codeSize = code.size();
52 createInfo.pCode = reinterpret_cast<const uint32_t*>(code.data());
53
54 VkShaderModule shaderModule;
55 if (vkCreateShaderModule(m_r_context.device, &createInfo, nullptr, &shaderModule) != VK_SUCCESS) {
56 throw_error("Failed to create shader module");
57 }
58 return shaderModule;
59 };
60
61 VkShaderModule vertShaderModule = createShaderModule(vertShaderCode);
62 VkShaderModule fragShaderModule = createShaderModule(fragShaderCode);
63
64 log("Defining shader stages...");
65 VkPipelineShaderStageCreateInfo vertShaderStageInfo{};
66 vertShaderStageInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
67 vertShaderStageInfo.stage = VK_SHADER_STAGE_VERTEX_BIT;
68 vertShaderStageInfo.module = vertShaderModule;
69 vertShaderStageInfo.pName = "main";
70
71 VkPipelineShaderStageCreateInfo fragShaderStageInfo{};
72 fragShaderStageInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
73 fragShaderStageInfo.stage = VK_SHADER_STAGE_FRAGMENT_BIT;
74 fragShaderStageInfo.module = fragShaderModule;
75 fragShaderStageInfo.pName = "main";
76
77 VkPipelineShaderStageCreateInfo shaderStages[] = {vertShaderStageInfo, fragShaderStageInfo};
78
79 log("Creating vertex input state...");
80 VkPipelineVertexInputStateCreateInfo vertexInputInfo{};
81 vertexInputInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_VERTEX_INPUT_STATE_CREATE_INFO;
82 vertexInputInfo.vertexBindingDescriptionCount = 1;
83 vertexInputInfo.pVertexBindingDescriptions = &bindingDescription;
84 vertexInputInfo.vertexAttributeDescriptionCount = static_cast<uint32_t>(attributeDescriptions.size());
85 vertexInputInfo.pVertexAttributeDescriptions = attributeDescriptions.data();
86
87 log("Creating pipeline state...");
88 VkPipelineInputAssemblyStateCreateInfo inputAssembly{};
89 inputAssembly.sType = VK_STRUCTURE_TYPE_PIPELINE_INPUT_ASSEMBLY_STATE_CREATE_INFO;
90 inputAssembly.topology = VK_PRIMITIVE_TOPOLOGY_TRIANGLE_LIST;
91 inputAssembly.primitiveRestartEnable = VK_FALSE;
92
93 VkViewport viewport{};
94 viewport.x = 0.0f;
95 viewport.y = 0.0f;
96 viewport.width = static_cast<float>(m_r_context.currentRenderResolution.x);
97 viewport.height = static_cast<float>(m_r_context.currentRenderResolution.y);
98 viewport.minDepth = 0.0f;
99 viewport.maxDepth = 1.0f;
100
101 VkRect2D scissor{};
102 scissor.offset = {0, 0};
103 scissor.extent = {m_currentRenderResolution.x, m_currentRenderResolution.y};
104
105 log("Creating viewport state...");
106 VkPipelineViewportStateCreateInfo viewportState{};
107 viewportState.sType = VK_STRUCTURE_TYPE_PIPELINE_VIEWPORT_STATE_CREATE_INFO;
108 viewportState.viewportCount = 1;
109 viewportState.pViewports = &viewport;
110 viewportState.scissorCount = 1;
111 viewportState.pScissors = &scissor;
112
113 log("Creating resterazer...");
114 VkPipelineRasterizationStateCreateInfo rasterizer{};
115 rasterizer.sType = VK_STRUCTURE_TYPE_PIPELINE_RASTERIZATION_STATE_CREATE_INFO;
116 rasterizer.depthClampEnable = VK_FALSE;
117 rasterizer.rasterizerDiscardEnable = VK_FALSE;
118 rasterizer.polygonMode = VK_POLYGON_MODE_FILL;
119 rasterizer.lineWidth = 1.0f;
120 rasterizer.cullMode = VK_CULL_MODE_BACK_BIT;
121 rasterizer.frontFace = VK_FRONT_FACE_COUNTER_CLOCKWISE;
122 rasterizer.depthBiasEnable = VK_FALSE;
123
124 log("Creating multisampling...");
125 VkPipelineMultisampleStateCreateInfo multisampling{};
126 multisampling.sType = VK_STRUCTURE_TYPE_PIPELINE_MULTISAMPLE_STATE_CREATE_INFO;
127 multisampling.sampleShadingEnable = VK_FALSE;
128 multisampling.rasterizationSamples = VK_SAMPLE_COUNT_1_BIT;
129
130 VkPipelineColorBlendAttachmentState colorBlendAttachment{};
131 colorBlendAttachment.colorWriteMask =
132 VK_COLOR_COMPONENT_R_BIT |
133 VK_COLOR_COMPONENT_G_BIT |
134 VK_COLOR_COMPONENT_B_BIT |
135 VK_COLOR_COMPONENT_A_BIT;
136 colorBlendAttachment.blendEnable = VK_TRUE;
137 colorBlendAttachment.srcColorBlendFactor = VK_BLEND_FACTOR_SRC_ALPHA;
138 colorBlendAttachment.dstColorBlendFactor = VK_BLEND_FACTOR_ONE_MINUS_SRC_ALPHA;
139 colorBlendAttachment.colorBlendOp = VK_BLEND_OP_ADD;
140 colorBlendAttachment.srcAlphaBlendFactor = VK_BLEND_FACTOR_ONE;
141 colorBlendAttachment.dstAlphaBlendFactor = VK_BLEND_FACTOR_ZERO;
142 colorBlendAttachment.alphaBlendOp = VK_BLEND_OP_ADD;

145 colorBlending.sType = VK_STRUCTURE_TYPE_PIPELINE_COLOR_BLEND_STATE_CREATE_INFO;
146 colorBlending.logicOpEnable = VK_FALSE;
147 colorBlending.attachmentCount = 1;
148 colorBlending.pAttachments = &colorBlendAttachment;

150 log("Creating Pipeline layout...");
151 VkPushConstantRange pushConstantRange{};
152 pushConstantRange.stageFlags = VK_SHADER_STAGE_VERTEX_BIT | VK_SHADER_STAGE_FRAGMENT_BIT;
153 pushConstantRange.offset = 0;

178 depthStencil.depthBoundsTestEnable = VK_FALSE;
179 depthStencil.minDepthBounds = 0.0f;
180 depthStencil.maxDepthBounds = 1.0f;
181 depthStencil.stencilTestEnable = VK_FALSE;
182

186 };
187

189 dynamicState.sType = VK_STRUCTURE_TYPE_PIPELINE_DYNAMIC_STATE_CREATE_INFO;
190 dynamicState.dynamicStateCount = static_cast<uint32_t>(dynamicStates.size());
191 dynamicState.pDynamicStates = dynamicStates.data();
192
193 VkPipelineRenderingCreateInfo renderingCreateInfo{};
194 renderingCreateInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_RENDERING_CREATE_INFO;
195 renderingCreateInfo.colorAttachmentCount = 1;
196 renderingCreateInfo.pColorAttachmentFormats = &m_r_context.swapchainImageFormat;
197 renderingCreateInfo.depthAttachmentFormat = m_r_context.depthFormat;
198
199 log("Creating Graphics pipeline...");
200 VkGraphicsPipelineCreateInfo pipelineInfo{};
201 pipelineInfo.sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
202 pipelineInfo.stageCount = 2;
203 pipelineInfo.pStages = shaderStages;
204 pipelineInfo.pVertexInputState = &vertexInputInfo;
205 pipelineInfo.pInputAssemblyState = &inputAssembly;
206 pipelineInfo.pViewportState = &viewportState;
207 pipelineInfo.pRasterizationState = &rasterizer;
208 pipelineInfo.pMultisampleState = &multisampling;
209 pipelineInfo.pColorBlendState = &colorBlending;
210 pipelineInfo.pDepthStencilState = &depthStencil;
211 pipelineInfo.pDynamicState = &dynamicState;
212 pipelineInfo.layout = m_layout;
213 pipelineInfo.pNext = &renderingCreateInfo;
214
215 if (vkCreateGraphicsPipelines(m_r_context.device, VK_NULL_HANDLE, 1, &pipelineInfo, nullptr, &m_pipeline) != VK_SUCCESS) {
216 throw_error("Failed to create graphics pipeline");
217 }
218
219 log("Cleanup shader modules...");
220 vkDestroyShaderModule(m_r_context.device, vertShaderModule, nullptr);
221 vkDestroyShaderModule(m_r_context.device, fragShaderModule, nullptr);
222 }
