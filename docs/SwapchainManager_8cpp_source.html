17 log("Creating Swapchain");
18 VkSurfaceCapabilitiesKHR capabilities;
19 if (vkGetPhysicalDeviceSurfaceCapabilitiesKHR(m_r_context.physicalDevice, m_r_context.surface, &capabilities) != VK_SUCCESS) {
20 throw_error("Failed to get surface capabilities");
21 }
22
23 uint32_t formatCount;
24 vkGetPhysicalDeviceSurfaceFormatsKHR(m_r_context.physicalDevice, m_r_context.surface, &formatCount, nullptr);
25 if (formatCount == 0) {
26 throw_error("No surface formats supported");
27 }
28 std::vector<VkSurfaceFormatKHR> formats(formatCount);
29 vkGetPhysicalDeviceSurfaceFormatsKHR(m_r_context.physicalDevice, m_r_context.surface, &formatCount, formats.data());
30
31 uint32_t presentModeCount;
32 vkGetPhysicalDeviceSurfacePresentModesKHR(m_r_context.physicalDevice, m_r_context.surface, &presentModeCount, nullptr);
33 if (presentModeCount == 0) {
34 throw_error("No present modes supported");
35 }
36
37 std::vector<VkPresentModeKHR> presentModes(presentModeCount);
38 vkGetPhysicalDeviceSurfacePresentModesKHR(m_r_context.physicalDevice, m_r_context.surface, &presentModeCount, presentModes.data());
39
40 VkSurfaceFormatKHR surfaceFormat = chooseSwapSurfaceFormat(formats);
41 VkPresentModeKHR presentMode = chooseSwapPresentMode(presentModes);
42 VkExtent2D extent = chooseSwapExtent(capabilities);
43
44 VkFormatProperties formatProps;
45 vkGetPhysicalDeviceFormatProperties(m_r_context.physicalDevice, surfaceFormat.format, &formatProps);
46
47 if (!(formatProps.optimalTilingFeatures & VK_FORMAT_FEATURE_BLIT_SRC_BIT) ||
48 !(formatProps.optimalTilingFeatures & VK_FORMAT_FEATURE_BLIT_DST_BIT)) {
49 throw_error("Swapchain format doesn't support blitting!");
50 }
51
52 uint32_t imageCount = capabilities.minImageCount + 1;
53 if (capabilities.maxImageCount > 0 && imageCount > capabilities.maxImageCount) {
54 imageCount = capabilities.maxImageCount;
55 }
56
57 VkSwapchainCreateInfoKHR createInfo = {};
58 createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
59 createInfo.surface = m_r_context.surface;
60 createInfo.minImageCount = imageCount;
61 createInfo.imageFormat = surfaceFormat.format;
62 createInfo.imageColorSpace = surfaceFormat.colorSpace;
63 createInfo.imageExtent = extent;
64 createInfo.imageArrayLayers = 1;
65 createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT;
66
67 uint32_t queueFamilyIndices[] = {m_r_context.graphicsQueueFamily, m_r_context.presentQueueFamily};
68 if (m_r_context.graphicsQueueFamily != m_r_context.presentQueueFamily) {
69 createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
70 createInfo.queueFamilyIndexCount = 2;
71 createInfo.pQueueFamilyIndices = queueFamilyIndices;
72 } else {
73 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
74 createInfo.queueFamilyIndexCount = 0;
75 createInfo.pQueueFamilyIndices = nullptr;
76 }
77
78 createInfo.preTransform = capabilities.currentTransform;
79 createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
80 createInfo.presentMode = presentMode;
81 createInfo.clipped = VK_TRUE;
82 createInfo.oldSwapchain = VK_NULL_HANDLE;
83
84 if (vkCreateSwapchainKHR(m_r_context.device, &createInfo, nullptr, &m_r_context.swapchain) != VK_SUCCESS) {
85 throw_error("Failed to create swapchain");
86 }
87
88 vkGetSwapchainImagesKHR(m_r_context.device, m_r_context.swapchain, &imageCount, nullptr);
89 m_r_context.swapchainImages.resize(imageCount);
90 vkGetSwapchainImagesKHR(m_r_context.device, m_r_context.swapchain, &imageCount, m_r_context.swapchainImages.data());
91
92 m_r_context.swapchainImageFormat = surfaceFormat.format;
93 m_r_context.swapchainExtent = extent;
94
95 createImageViews();
96
97 VkCommandBuffer cmd = m_r_context.beginSingleTimeCommands();
98
99 for (auto& image : m_r_context.swapchainImages) {
100 VkImageMemoryBarrier barrier{};
101 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
102 barrier.oldLayout = VK_IMAGE_LAYOUT_UNDEFINED;
103 barrier.newLayout = VK_IMAGE_LAYOUT_PRESENT_SRC_KHR;
104 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
105 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
106 barrier.image = image;
107 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
108 barrier.subresourceRange.levelCount = 1;
109 barrier.subresourceRange.layerCount = 1;
110
111 vkCmdPipelineBarrier(
112 cmd,
113 VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT,
114 VK_PIPELINE_STAGE_BOTTOM_OF_PIPE_BIT,
115 0,
116 0, nullptr,
117 0, nullptr,
118 1, &barrier
119 );
120 }
121
122 m_r_context.endSingleTimeCommands(cmd);
123 }

235 void VulkanSwapchainManager::createSyncObjects() {
236 log("Creating synchronization objects...");
237
238 m_r_context.imageAvailableSemaphores.resize(m_r_context.MAX_FRAMES_IN_FLIGHT);
239 m_r_context.renderFinishedSemaphores.resize(m_r_context.MAX_FRAMES_IN_FLIGHT);
240 m_r_context.inFlightFences.resize(m_r_context.MAX_FRAMES_IN_FLIGHT);
241
242 VkSemaphoreCreateInfo semaphoreInfo = {};
243 semaphoreInfo.sType = VK_STRUCTURE_TYPE_SEMAPHORE_CREATE_INFO;
244
245 VkFenceCreateInfo fenceInfo = {};
246 fenceInfo.sType = VK_STRUCTURE_TYPE_FENCE_CREATE_INFO;
247 fenceInfo.flags = VK_FENCE_CREATE_SIGNALED_BIT;
248
249 VkResult result;
250
251 for (size_t i = 0; i < m_r_context.MAX_FRAMES_IN_FLIGHT; i++) {
252 result = vkCreateSemaphore(m_r_context.device, &semaphoreInfo, nullptr, &m_r_context.imageAvailableSemaphores[i]);
253 if (result != VK_SUCCESS) {
254 throw_error("Failed to create image available semaphore");
255 }
256
257 result = vkCreateSemaphore(m_r_context.device, &semaphoreInfo, nullptr, &m_r_context.renderFinishedSemaphores[i]);
258 if (result != VK_SUCCESS) {
259 throw_error("Failed to create render finished semaphore");
260 }
261
262 result = vkCreateFence(m_r_context.device, &fenceInfo, nullptr, &m_r_context.inFlightFences[i]);
263 if (result != VK_SUCCESS) {
264 throw_error("Failed to create fence");
265 }
266
267 assert(m_r_context.inFlightFences[i] != VK_NULL_HANDLE);
268
269 log("Created sync objects: fence=%p", (void*)m_r_context.inFlightFences[i]);
270 }
271
272 if (m_r_context.depthImageView) {
273 createLowResResources();
274 } else {
275 SDL_LogWarn(SDL_LOG_CATEGORY_RENDER,
276 "Deferring low-res resource creation - depth image not ready");

412 VkExtent2D VulkanSwapchainManager::chooseSwapExtent(const VkSurfaceCapabilitiesKHR& capabilities) {
413 log("choosing swap extent");
414 if (capabilities.currentExtent.width != std::numeric_limits<uint32_t>::max()) {
415 return capabilities.currentExtent;
416 } else {
417 int width, height;
418 SDL_GetWindowSizeInPixels(m_p_window, &width, &height);

426 actualExtent.width,

438 VkFormat VulkanSwapchainManager::findDepthFormat() {
439 std::vector<VkFormat> candidates = {
440 VK_FORMAT_D32_SFLOAT,
441 VK_FORMAT_D32_SFLOAT_S8_UINT,
442 VK_FORMAT_D24_UNORM_S8_UINT
443 };
444
445 for (VkFormat format : candidates) {
446 VkFormatProperties props;
447 vkGetPhysicalDeviceFormatProperties(m_r_context.physicalDevice, format, &props);
448 if (props.optimalTilingFeatures & VK_FORMAT_FEATURE_DEPTH_STENCIL_ATTACHMENT_BIT) {
449 return format;
450 }
451 }
452 throw_error("Failed to find supported depth format");
