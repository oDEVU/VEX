469 VkFormat VulkanSwapchainManager::findDepthFormat() {
470 std::vector<VkFormat> candidates = {
471 VK_FORMAT_D32_SFLOAT,
472 VK_FORMAT_D32_SFLOAT_S8_UINT,
473 VK_FORMAT_D24_UNORM_S8_UINT
474 };
475
476 for (VkFormat format : candidates) {
477 VkFormatProperties props;
478 vkGetPhysicalDeviceFormatProperties(m_r_context.physicalDevice, format, &props);
479 if (props.optimalTilingFeatures & VK_FORMAT_FEATURE_DEPTH_STENCIL_ATTACHMENT_BIT) {
480 return format;
481 }
482 }
483 throw_error("Failed to find supported depth format");
484 return VK_FORMAT_UNDEFINED;
