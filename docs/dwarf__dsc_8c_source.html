49/* When called with ary and *arraycount 0
50 this just counts the elements found.
51 Otherwise it records the values in ary and
52 recounts. The arraycount pointer must be
53 passed-in non-null always. */
54static int
55get_dsc_leb_entries(Dwarf_Debug dbg,
56 Dwarf_Small * blockpointer,
57 Dwarf_Unsigned blocklen,
58 int dounsigned,
59 struct Dwarf_Dsc_Entry_s *ary,
60 size_t * arraycount,
61 Dwarf_Error * error)
62{
63 Dwarf_Small *p = blockpointer;
64 Dwarf_Small *endp = blockpointer + blocklen;
65 size_t larraycount = 0;
66 size_t iarraycount = *arraycount;
67
68 if (!ary) {
69 if (iarraycount) {
70 /* Internal botch calling this static function. */
71 _dwarf_error(dbg, error, DW_DLE_DISCR_ARRAY_ERROR);
72 return DW_DLV_ERROR;
73 } else {}
74 } else {
75 if (!iarraycount) {
76 /* Internal botch calling this static function. */
77 _dwarf_error(dbg, error, DW_DLE_DISCR_ARRAY_ERROR);
78 return DW_DLV_ERROR;
79 } else {}
80 }
81 if (dounsigned) {
82 while (p < endp) {
83 Dwarf_Unsigned dsc = 0;
84 Dwarf_Unsigned low = 0;
85 Dwarf_Unsigned high = 0;
86
87 if (ary && (larraycount >= iarraycount)) {
88 _dwarf_error(dbg, error, DW_DLE_DISCR_ARRAY_ERROR);
89 return DW_DLV_ERROR;
90 }
91 DECODE_LEB128_UWORD_CK(p,dsc,
92 dbg,error,endp);
93 if (!dsc) {
94 DECODE_LEB128_UWORD_CK(p,low,
95 dbg,error,endp);
96 } else {
97 DECODE_LEB128_UWORD_CK(p,low,
98 dbg,error,endp);
99 DECODE_LEB128_UWORD_CK(p,high,
100 dbg,error,endp);
101 }
102 if (ary) {
103 struct Dwarf_Dsc_Entry_s *arye =
104 ary+larraycount;
105
106 /* type reads the same as uleb and leb because
107 it is only zero or one. */
108 arye->dsc_type = (Dwarf_Half)dsc;
109 arye->dsc_low_u = low;
110 arye->dsc_high_u = high;
111 }
112 larraycount++;
113 }
114 } else {
115 while (p < endp) {
116 Dwarf_Signed dsc = 0;
117 Dwarf_Signed low = 0;
118 Dwarf_Signed high = 0;
119 Dwarf_Unsigned leblen = 0;
120
121 (void)leblen;
122 if (ary && (larraycount >= iarraycount)) {
123 _dwarf_error(dbg, error, DW_DLE_DISCR_ARRAY_ERROR);
124 return DW_DLV_ERROR;
125 }
126 DECODE_LEB128_SWORD_LEN_CK(p,dsc,
127 leblen,dbg,error,endp);
128 if (!dsc) {
129 DECODE_LEB128_SWORD_LEN_CK(p,low,
130 leblen,dbg,error,endp);
131 } else {
132 DECODE_LEB128_SWORD_LEN_CK(p,low,
133 leblen,dbg,error,endp);
134 DECODE_LEB128_SWORD_LEN_CK(p,high,
135 leblen,dbg,error,endp);
136 }
137 if (ary) {
138 struct Dwarf_Dsc_Entry_s *arye =
139 ary+larraycount;
140
141 /* type reads the same as uleb and leb because
142 it is only zero or one. */
143 arye->dsc_type = (Dwarf_Half)dsc;
144 arye->dsc_low_s = low;
145 arye->dsc_high_s = high;
146 }
147 larraycount++;
148 }
149 }
150 if (ary) {
151 /* Just verify this recount matches original */
152 if (iarraycount != larraycount) {
153 _dwarf_error(dbg, error, DW_DLE_DISCR_ARRAY_ERROR);
154 return DW_DLV_ERROR;
155 }
156 } else {
157 /* This matters for first call with
158 ary 0 and iarraycount 0 as we are generating the
159 count. */
160 *arraycount = larraycount;
161 }
162 return DW_DLV_OK;
163}

165int dwarf_discr_list(Dwarf_Debug dbg,
166 Dwarf_Small * blockpointer,
167 Dwarf_Unsigned blocklen,
168 Dwarf_Dsc_Head * dsc_head_out,
169 Dwarf_Unsigned * dsc_array_length_out,
170 Dwarf_Error * error)
171{
172 Dwarf_Dsc_Head h = 0;
173 int res = 0;
174 size_t arraycount = 0;
175 struct Dwarf_Dsc_Entry_s *ary = 0;
176 Dwarf_Small * dscblockp = 0;
177 Dwarf_Unsigned dscblocklen = 0;
178
179 CHECK_DBG(dbg,error,"dwarf_discr_list()");
180 if (blocklen == 0) {
181 return DW_DLV_NO_ENTRY;
182 }
183 dscblockp = (Dwarf_Small *)calloc(blocklen,sizeof(Dwarf_Small));
184 if (!dscblockp) {
185 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
186 return DW_DLV_ERROR;
187 }
188 dscblocklen = blocklen;
189 memcpy(dscblockp,blockpointer,blocklen);
190
191 res = get_dsc_leb_entries(dbg,dscblockp,dscblocklen,
192 /* TRUE or FALSE here is not important, the arraycount
193 returned to us will be identical either way. */
194 FALSE, 0, &arraycount,error);
195 if (res != DW_DLV_OK) {
196 free(dscblockp);
197 return res;
198 }
199
200 h = (Dwarf_Dsc_Head)_dwarf_get_alloc(dbg,DW_DLA_DSC_HEAD,1);
201 if (!h) {
202 free(dscblockp);
203 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
204 return DW_DLV_ERROR;
205 }
206
207 h->dsh_block = dscblockp;
208 h->dsh_block_len = dscblocklen;
209 h->dsh_debug = dbg;
210 /* Now the destructor for h will deal with block malloc space. */
211
212 ary = (struct Dwarf_Dsc_Entry_s *)calloc(arraycount,
213 sizeof(struct Dwarf_Dsc_Entry_s));
214 if (!ary) {
215 /* Coverity scan CID 531838. Mem leak.
216 Free the block h points to here too. */
217 free(h->dsh_block);
218 h->dsh_block = 0;
219 h->dsh_block_len = 0;
220 dscblockp = 0;
221 dwarf_dealloc(dbg,h,DW_DLA_DSC_HEAD);
222 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
223 return DW_DLV_ERROR;
224 }
225 h->dsh_count = arraycount;
226 h->dsh_array = ary;
227 h->dsh_set_unsigned = 0;
228 h->dsh_set_signed = 0;
229 *dsc_head_out = h;
230 *dsc_array_length_out = arraycount;
231 return DW_DLV_OK;
232}

234/* NEW September 2016. Allows easy access to DW_AT_discr_list
235 entry. Callers must know which is the appropriate
236 one of the following two interfaces, though both
237 will work. */
238int
239dwarf_discr_entry_u(Dwarf_Dsc_Head dsh ,
240 Dwarf_Unsigned entrynum,
241 Dwarf_Half * out_type,
242 Dwarf_Unsigned * out_discr_low,
243 Dwarf_Unsigned * out_discr_high,
244 Dwarf_Error * error)
245{
246 struct Dwarf_Dsc_Entry_s *dse = 0;
247
248 (void)error;
249 if (entrynum >= dsh->dsh_count) {
250 return DW_DLV_NO_ENTRY;
251 }
252 if (!dsh->dsh_set_unsigned) {
253 int res =0;
254 int dounsigned = 1;
255 size_t count = dsh->dsh_count;
256
257 res = get_dsc_leb_entries(dsh->dsh_debug,
258 dsh->dsh_block,
259 dsh->dsh_block_len,
260 dounsigned,
261 dsh->dsh_array,
262 &count,
263 error);
264 if (res != DW_DLV_OK) {
265 return res;
266 }
267 dsh->dsh_set_unsigned = TRUE;
268 }
269 if (!dsh->dsh_array) {
270 _dwarf_error(dsh->dsh_debug, error, DW_DLE_DISCR_ARRAY_ERROR);
271 return DW_DLV_ERROR;
272 }
273 dse = dsh->dsh_array + entrynum;
274 *out_type = dse->dsc_type;
275 *out_discr_low = dse->dsc_low_u;
276 *out_discr_high = dse->dsc_high_u;
277 return DW_DLV_OK;
278}
