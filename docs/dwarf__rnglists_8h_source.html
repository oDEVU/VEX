40/* Rangelists header for a CU. The
41 type is never visible to libdwarf callers */
42struct Dwarf_Rnglists_Context_s {
43 Dwarf_Debug rc_dbg;
44 Dwarf_Unsigned rc_index; /* An index assigned by
45 libdwarf to each rnglists context. Starting
46 with zero at the zero offset in .debug_rnglists. */
47
48 /* Offset of the .debug_rnglists header involved. */
49 Dwarf_Unsigned rc_header_offset;
50 Dwarf_Unsigned rc_length;
51
52 /* Many places in in libdwarf this is called length_size. */
53 Dwarf_Small rc_offset_size;
54 unsigned long rc_magic;
55
56 /* rc_extension_size is zero unless this is standard
57 DWARF3 and later 64bit dwarf using the extension mechanism.
58 64bit DWARF3 and later: rc_extension_size is 4.
59 64bit DWARF2 MIPS/IRIX: rc_extension_size is zero.
60 32bit DWARF: rc_extension_size is zero. */
61 Dwarf_Small rc_extension_size;
62
63 unsigned rc_version; /* 5 */
64 Dwarf_Small rc_address_size;
65 Dwarf_Small rc_segment_selector_size;
66
67 /* The next three are the table of offsets.
68 used by DW_FORM_rnglistx.
69 The count may be zero if no DW_FORM_rnglistx.
70 If they exist the
71 index values in the array are offsets into area starting
72 with rc_first_rnglist_offset */
73 Dwarf_Unsigned rc_offset_entry_count;
74 /* rc_offset_entry_count values. Each local offset to
75 a locdesc set. We need this as a way to know
76 which lle entry offsets are relevant from a loclistx.
77 as nothing else reveals these special LLE entries. */
78 Dwarf_Unsigned *rc_offset_value_array;
79
80 /* offset in the section of the offset entries */
81 Dwarf_Unsigned rc_offsets_off_in_sect;
82
83 /* Do not free. Points into section memory */
84 Dwarf_Small * rc_offsets_array;
85 /* End table of offsets */
86
87 /* Now to the rnglists. If DW_AT_ranges FORM is
88 DW_FORM_sec_offset those offsets are inside
89 here.
90 Offset in the .debug_rnglists section of the
91 first rangelist in the set of rangelists for the
92 CU. */
93 Dwarf_Unsigned rc_first_rnglist_offset;
94 Dwarf_Unsigned rc_past_last_rnglist_offset;
95
96 /* pointer to 1st byte of rangelist header*/
97 Dwarf_Small * rc_rnglists_header;
98 /* pointer to first byte of the rnglist data
99 for rnglist involved. Do not free. */
100 Dwarf_Small *rc_startaddr;
101 /* pointer one past end of the rnglist data. */
102 Dwarf_Small *rc_endaddr;
103};

105typedef struct Dwarf_Rnglists_Entry_s *Dwarf_Rnglists_Entry;
106struct Dwarf_Rnglists_Entry_s {
107 unsigned rle_entrylen;
108 unsigned rle_code;
109 /* Failed means .debug_addr section needed but missing.
110 (possibly tied file needed) */
111 Dwarf_Bool rle_index_failed;
112 Dwarf_Unsigned rle_raw1;
113 Dwarf_Unsigned rle_raw2;
114 /* Cooked means the raw values from the .debug_rnglists
115 section translated to CU-specific addresses. */
116 Dwarf_Unsigned rle_cooked1;
117 Dwarf_Unsigned rle_cooked2;
118 Dwarf_Rnglists_Entry rle_next;
119};
