37/* In a given CU, one of these is (eventually) set up
38 for every abbreviation we need to find (and for all
39 those earlier in the abbreviations for that CU).
40 So we don't want elements needlessly big.
41*/
42struct Dwarf_Abbrev_s {
43 /* No TAG should exceed DW_TAG_hi_user, 0xffff, but
44 we do allow a larger value here. */
45 Dwarf_Unsigned dab_tag;
46 /* Abbreviations are numbered (normally sequentially from
47 1 and so 16 bits is not enough! */
48 Dwarf_Unsigned dab_code;
49 Dwarf_Small dab_has_child;
50 /* dab_abbrev_ptr points to the abbreviations
51 themselves in memory, the list of attr/form
52 integers terminated by 0,0. */
53 Dwarf_Byte_Ptr dab_abbrev_ptr;
54 Dwarf_Debug dab_dbg;
55
56 /* Section global offset of the abbrev. */
57 Dwarf_Off dab_goffset;
58 /* dab_count is the number of attr/form uleb pairs */
59 Dwarf_Off dab_count;
60 /* The number of DW_FORM_implicit_const in the uleb pairs*/
61 Dwarf_Unsigned dab_implicit_count;
62
63 /* When the caller cycles through attr/form pairs
64 by index from zero this lets the code read just one
65 pair to work. */
66 Dwarf_Byte_Ptr dab_next_ptr;
67 Dwarf_Unsigned dab_next_index;
68};
