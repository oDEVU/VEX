23 AudioClip(const std::string& path, vex::VirtualFileSystem* vfs) {
24 if (!vfs) return;
25
26 auto fileData = vfs->load_file(path);
27 if (!fileData) {
28 vex::log(vex::LogLevel::ERROR, "AudioClip: VFS failed to load path: %s", path.c_str());
29 return;
30 }
31
32 SDL_IOStream* io = SDL_IOFromConstMem(fileData->data.data(), fileData->size);
33 if (SDL_LoadWAV_IO(io, true, &spec, &buffer, &length)) {
34 valid = true;
35 } else {
36 vex::log(vex::LogLevel::ERROR, "AudioClip: SDL_LoadWAV failed for %s. SDL Error: %s", path.c_str(), SDL_GetError());
37 }
38 }
