16void ResolutionManager::update() {
17 int width, height;
18 SDL_GetWindowSizeInPixels(m_p_window, &width, &height);
19 m_windowResolution = {width, height};
20
21 switch(m_currentMode) {
22 case ResolutionMode::NATIVE:
23 m_renderResolution = m_windowResolution;
24 m_upscaleRatio = 1.0f;
25 break;
26
27 case ResolutionMode::RES_240P: {
28 float aspect = width / static_cast<float>(height);
29 m_renderResolution.y = 240;
30 m_renderResolution.x = static_cast<uint32_t>(240 * aspect);
31 m_upscaleRatio = height / 240.0f;
32 break;
33 }
34
35 case ResolutionMode::RES_480P: {
36 float aspect = width / static_cast<float>(height);
37 m_renderResolution.y = 480;
38 m_renderResolution.x = static_cast<uint32_t>(480 * aspect);
39 m_upscaleRatio = height / 480.0f;
40 break;
41 }
42
43 case ResolutionMode::PS1_SHARP:
44 calculatePS1SharpResolution();
45 break;
46 }
47
48 log("Resolution mode updated: %dx%d (render) -> %dx%d (window), scale: %.2f",
49 m_renderResolution.x, m_renderResolution.y,
50 m_windowResolution.x, m_windowResolution.y,
51 m_upscaleRatio);
52}
53
54void ResolutionManager::calculatePS1SharpResolution() {
55 int yscale = floor(m_windowResolution.y / 240);
56 int maxScale = std::max(1, yscale);
57
58 maxScale = std::clamp(maxScale, 1, 4);
59
60 m_renderResolution = m_windowResolution / static_cast<unsigned int>(maxScale);
61 m_upscaleRatio = maxScale;
62
63 if (m_renderResolution.y < 240) {
64 float aspect = m_windowResolution.x / static_cast<float>(m_windowResolution.y);
65 m_renderResolution.y = 240;
66 m_renderResolution.x = static_cast<uint32_t>(240 * aspect);
67 m_upscaleRatio = m_windowResolution.y / 240.0f;
68 }
69}
