193/* Unifies the basic duplicate/empty testing and section
194 data setting to one place. */
195static int
196get_basic_section_data(Dwarf_Debug dbg,
197 struct Dwarf_Section_s *secdata,
198 struct Dwarf_Obj_Access_Section_a_s *doas,
199 Dwarf_Unsigned section_index,
200 unsigned group_number,
201 Dwarf_Error* error,
202 int duperr, int emptyerr )
203{
204 /* There is an elf convention that section index 0 is reserved,
205 and that section is always empty.
206 Non-elf object formats must honor that by ensuring that
207 (when they assign numbers to 'sections' or
208 'section-like-things')
209 they never assign a real section section-number
210 0 to dss_index. */
211 if (secdata->dss_index != 0) {
212 DWARF_DBG_ERROR(dbg, duperr, DW_DLV_ERROR);
213 }
214 if (doas->as_size == 0) {
215 /* As of 2018 it seems impossible to detect
216 (via dwarfdump) whether emptyerr has any
217 practical effect, whether TRUE or FALSE. */
218 if (emptyerr == 0 ) {
219 /* Allow empty section. */
220 return DW_DLV_OK;
221 }
222 /* Know no reason to allow section */
223 DWARF_DBG_ERROR(dbg, emptyerr, DW_DLV_ERROR);
224 }
225 secdata->dss_index = section_index;
226 secdata->dss_size = doas->as_size;
227 secdata->dss_group_number = group_number;
228 secdata->dss_addr = doas->as_addr;
229 secdata->dss_link = doas->as_link;
230 secdata->dss_flags = doas->as_flags;
231 if (secdata->dss_flags & SHF_COMPRESSED) {
232 secdata->dss_shf_compressed = TRUE;
233 }
234 secdata->dss_entrysize = doas->as_entrysize;
235 secdata->dss_addralign = doas->as_addralign;
236 return DW_DLV_OK;
237}

297/* Given an Elf ptr, set up dbg with pointers
298 to all the Dwarf data sections.
299 Return NULL on error.
300
301 This function is also responsible for determining
302 whether the given object contains Dwarf information
303 or not. The test currently used is that it contains
304 either a .debug_info or a .debug_frame section. If
305 not, it returns DW_DLV_NO_ENTRY causing dwarf_init() also to
306 return DW_DLV_NO_ENTRY. Earlier, we had thought of using only
307 the presence/absence of .debug_info to test, but we
308 added .debug_frame since there could be stripped objects
309 that have only a .debug_frame section for exception
310 processing.
311 DW_DLV_NO_ENTRY or DW_DLV_OK or DW_DLV_ERROR
312
313 This does not allow for section-groups in object files,
314 for which many .debug_info (and other DWARF) sections may exist.
315
316 We process. .rela (SHT_RELA) and .rel (SHT_REL)
317 sections because with .rela the referencing section
318 offset value is zero whereas with .rel the
319 referencing section value is already correct for
320 the object itself. In other words, we do it because
321 of the definition of .rela relocations in Elf.
322
323 However! In some cases clang emits a .rel section (at least
324 for .rel.debug_info) where symtab entries have an st_value
325 that must be treated like an addend: the compiler did not
326 bother to backpatch the DWARF information for these.
327*/

329/* For an object file with an incorrect rela section name,
330 readelf prints correct debug information,
331 as the tool takes the section type instead
332 of the section name. So check the
333 section name but test section type. */
334static int
335is_a_relx_section(const char *scn_name,int type,int *is_rela)
336{
337 if (type == SHT_RELA) {
338 *is_rela = TRUE;
339 return TRUE;
340 }
341 if (_dwarf_startswith(scn_name,".rela.")) {
342 *is_rela = TRUE;
343 return TRUE;
344 }
345 if (_dwarf_startswith(scn_name,".rel.")) {
346 *is_rela = FALSE;
347 return TRUE;
348 }
349 if (type == SHT_REL) {
350 *is_rela = FALSE;
351 return TRUE;
352 }
353 *is_rela = FALSE;
354 return FALSE;
355}

370static int
371this_section_dwarf_relevant(const char *scn_name,
372 int type,
373 int *is_rela)
374{
375 /* A small helper function for _dwarf_setup(). */
376 if (_dwarf_startswith(scn_name, ".zdebug_") ||
377 _dwarf_startswith(scn_name, ".debug_")) {
378 /* standard debug */
379 return TRUE;
380 }
381 if (_dwarf_ignorethissection(scn_name)) {
382 return FALSE;
383 }
384 /* Now check if a special section could be
385 in a section_group, but though seems unlikely. */
386 if (!strcmp(scn_name, ".eh_frame")) {
387 /* This is not really a group related file, but
388 it is harmless to consider it such. */
389 return TRUE;
390 }
391 if (!strcmp(scn_name, ".gnu_debuglink")) {
392 /* This is not a group or DWARF related file, but
393 it is useful for split dwarf. */
394 return TRUE;
395 }
396 if (!strcmp(scn_name, ".note.gnu.build-id")) {
397 /* This is not a group or DWARF related file, but
398 it is useful for split dwarf. */
399 return TRUE;
400 }
401 if (!strcmp(scn_name, ".gdb_index")) {
402 return TRUE;
403 }
404 if (is_a_special_section_semi_dwarf(scn_name)) {
405 return TRUE;
406 }
407 if (is_a_relx_section(scn_name,type,is_rela)) {
408 return TRUE;
409 }
410 /* All sorts of sections are of no interest: .text
411 .rel. and many others. */
412 return FALSE;
413}

415/* This assumes any non-Elf object files have no SHT_GROUP
416 sections. So this code will not be invoked on non-Elf objects.
417 One supposes this is unlikely to match any non-Elf
418 version of COMDAT. */
419static int
420insert_sht_list_in_group_map(Dwarf_Debug dbg,
421 struct Dwarf_Obj_Access_Section_a_s *doas,
422 unsigned comdat_group_number,
423 unsigned section_number,
424 Dwarf_Unsigned section_count,
425 struct Dwarf_Obj_Access_Interface_a_s * obj,
426 unsigned *did_add_map,
427 Dwarf_Error *error)
428{
429 struct Dwarf_Section_s secdata;
430 Dwarf_Small * data = 0;
431 int res = 0;
432 Dwarf_Small* secend = 0;
433
434 memset(&secdata,0,sizeof(secdata));
435 secdata.dss_size = doas->as_size;
436 secdata.dss_entrysize = doas->as_entrysize;
437 secdata.dss_group_number = 1; /* arbitrary. */
438 secdata.dss_index = section_number;
439 secdata.dss_name = ".group";
440 secdata.dss_standard_name = ".group";
441 secdata.dss_number = section_number;
442 secdata.dss_ignore_reloc_group_sec = TRUE;
443 res = _dwarf_load_section(dbg,&secdata,error);
444 if (res != DW_DLV_OK) {
445 _dwarf_malloc_section_free(&secdata);
446 return res;
447 }
448 if (!secdata.dss_data) {
449 _dwarf_error(dbg,error,DW_DLE_GROUP_INTERNAL_ERROR);
450 return DW_DLV_ERROR;
451 }
452 if (doas->as_entrysize != 4) {
453 _dwarf_malloc_section_free(&secdata);
454 _dwarf_error(dbg,error,DW_DLE_GROUP_INTERNAL_ERROR);
455 return DW_DLV_ERROR;
456 }
457 /* So now pick up the data in dss_data.
458 It is an array of 32 bit fields.
459 Entry zero is just a constant 1.
460 Each additional is a section number. */
461 data = secdata.dss_data;
462 secend = data + secdata.dss_size;
463 {
464 Dwarf_Unsigned i = 1;
465 Dwarf_Unsigned count = doas->as_size/doas->as_entrysize;
466 Dwarf_Unsigned fval = 0;
467
468 /* The fields treatments with regard
469 to endianness is unclear. In any case a single
470 bit should be on, as 0x01000000
471 without any endiannes swapping.
472 Or so it seems given limited evidence.
473 We read with length checking and allow the
474 reader to byte swap and then fix things.
475 At least one test case has big-endian
476 data but little-endian SHT_GROUP data. */
477 if ((data+DWARF_32BIT_SIZE) > secend) {
478 /* Duplicates the check in READ_UNALIGNED_CK
479 so we can free allocated memory bere. */
480 _dwarf_malloc_section_free(&secdata);
481 _dwarf_error(dbg,error,DW_DLE_GROUP_INTERNAL_ERROR);
482 return DW_DLV_ERROR;
483 }
484 READ_UNALIGNED_CK(dbg,fval,Dwarf_Unsigned,
485 data,
486 DWARF_32BIT_SIZE,
487 error,
488 secend);
489 if (fval != 1 && fval != 0x1000000) {
490 /* Could be corrupted elf object. */
491 _dwarf_malloc_section_free(&secdata);
492 _dwarf_error(dbg,error,DW_DLE_GROUP_INTERNAL_ERROR);
493 return DW_DLV_ERROR;
494 }
495
496 data = data + doas->as_entrysize;
497 for (i = 1 ; i < count ; ++i) {
498 Dwarf_Unsigned val = 0;
499
500 if ((data+DWARF_32BIT_SIZE) > secend) {
501 /* Duplicates the check in READ_UNALIGNED_CK
502 so we can free allocated memory bere. */
503 _dwarf_malloc_section_free(&secdata);
504 _dwarf_error(dbg,error,DW_DLE_GROUP_INTERNAL_ERROR);
505 return DW_DLV_ERROR;
506 }
507 READ_UNALIGNED_CK(dbg,val,Dwarf_Unsigned,
508 data,
509 DWARF_32BIT_SIZE,
510 error,
511 secend);
512 if (val > section_count) {
513 /* Might be confused endianness by
514 the compiler generating the SHT_GROUP.
515 This is pretty horrible. */
516 Dwarf_Unsigned valr = 0;
517 _dwarf_memcpy_swap_bytes(&valr,&val,
518 DWARF_32BIT_SIZE);
519 if (valr > section_count) {
520 _dwarf_malloc_section_free(&secdata);
521 _dwarf_error(dbg,error,
522 DW_DLE_GROUP_INTERNAL_ERROR);
523 return DW_DLV_ERROR;
524 }
525 /* Ok. Yes, ugly. */
526 val = valr;
527 }
528 {
529 /* Ensure this group entry DWARF relevant before
530 adding to group map */
531 struct Dwarf_Obj_Access_Section_a_s doasx;
532 int resx = DW_DLV_ERROR;
533 int err = 0;
534 int is_rela = FALSE;
535
536 memset(&doasx,0,sizeof(doasx));
537 resx = obj->ai_methods->
538 om_get_section_info(obj->ai_object,
539 val,
540 &doasx, &err);
541 if (resx == DW_DLV_NO_ENTRY){
542 /* Should we really ignore this? */
543 continue;
544 }
545 if (resx == DW_DLV_ERROR){
546 _dwarf_malloc_section_free(&secdata);
547 _dwarf_error(dbg,error,err);
548 return resx;
549 }
550 if (!this_section_dwarf_relevant(doasx.as_name,
551 (int)doasx.as_type,&is_rela) ) {
552 continue;
553 }
554 data += DWARF_32BIT_SIZE;
555 *did_add_map = TRUE;
556 res = _dwarf_insert_in_group_map(dbg,
557 (unsigned)comdat_group_number,
558 (unsigned)val,
559 doasx.as_name,
560 error);
561 if (res != DW_DLV_OK) {
562 _dwarf_malloc_section_free(&secdata);
563 return res;
564 }
565 }
566 }
567 }
568 _dwarf_malloc_section_free(&secdata);
569 return DW_DLV_OK;
570}

572/* Split dwarf CUs can be in an object with non-split
573 or split may be in a separate object.
574 If all in one object the default is to deal with group_number
575 and ignore DW_GROUPNUMBER_DWO.
576 If only .dwo the default is DW_GROUPNUMBER_DWO(2).
577 Otherwise use DW_GROUP_NUMBER_BASE(1).
578
579 If there are COMDAT SHT_GROUP sections, these
580 are assigned group numbers 3-N as needed.
581
582 At present this makes the assumption that COMDAT group
583 (ie, SHT_GROUP) sections
584 have lower section numbers than the sections COMDAT refers to.
585 It is not clear whether this is guaranteed, COMDAT is not
586 an official Elf thing and documentation is scarce.
587 In the 1990's SGI folks and others formed a committee
588 and attempted to get COMDAT and a feature allowing section
589 numbers greater than 16 bits into Elf, but there was no
590 group that was able to approve such things.
591
592 This is called once at dbg init time.
593*/
594
595static int
596determine_target_group(Dwarf_Unsigned section_count,
597 struct Dwarf_Obj_Access_Interface_a_s * obj,
598 unsigned *group_number_out,
599 Dwarf_Debug dbg,
600 Dwarf_Error *error)
601{
602 unsigned obj_section_index = 0;
603 int found_group_one = 0;
604 int found_group_two = 0;
605 struct Dwarf_Group_Data_s *grp = 0;
606 unsigned comdat_group_next = 3;
607 unsigned lowest_comdat_groupnum = 0;
608
609 grp = &dbg->de_groupnumbers;
610 grp->gd_number_of_groups = 0;
611 grp->gd_number_of_sections = (unsigned int)section_count;
612 if (grp->gd_map) {
613 _dwarf_error(dbg,error,DW_DLE_GROUP_INTERNAL_ERROR);
614 return DW_DLV_OK;
615 }
616 for (obj_section_index = 0; obj_section_index < section_count;
617 ++obj_section_index) {
618
619 struct Dwarf_Obj_Access_Section_a_s doas;
620 int res = DW_DLV_ERROR;
621 int err = 0;
622 const char *scn_name = 0;
623 unsigned groupnumber = 0;
624 unsigned mapgroupnumber = 0;
625 int is_rela = FALSE;
626
627 memset(&doas,0,sizeof(doas));
628 res = obj->ai_methods->om_get_section_info(obj->ai_object,
629 obj_section_index,
630 &doas, &err);
631 if (res == DW_DLV_NO_ENTRY){
632 return res;
633 }
634 if (res == DW_DLV_ERROR){
635 _dwarf_error(dbg, error,err);
636 return res;
637 }
638
639 if (doas.as_type == SHT_GROUP) {
640 /* See assumptions in function comment above. */
641 unsigned did_add_map = 0;
642 /* Add to our map. Here we
643 are assuming SHT_GROUP records come first.
644 Till proven wrong. */
645 res = insert_sht_list_in_group_map(dbg,&doas,
646 comdat_group_next,
647 obj_section_index,
648 section_count,
649 obj,
650 &did_add_map,error);
651 if (res != DW_DLV_OK) {
652 return res;
653 }
654 if (!lowest_comdat_groupnum) {
655 lowest_comdat_groupnum = comdat_group_next;
656 }
657 if (did_add_map) {
658 ++grp->gd_number_of_groups;
659 ++comdat_group_next;
660 }
661 continue;
662 }
663 scn_name = doas.as_name;
664 if (!this_section_dwarf_relevant(scn_name,
665 (int)doas.as_type,
666 &is_rela) ) {
667 continue;
668 }
669
670 /* Now at a 'normal' section, though we do not
671 quite know what group it is. */
672
673 res = _dwarf_section_get_target_group_from_map(dbg,
674 obj_section_index,&groupnumber,error);
675 if (res == DW_DLV_OK ) {
676 /* groupnumber is set. Fall through.
677 All COMDAT group should get here. */
678 mapgroupnumber = groupnumber;
679 } else if (res == DW_DLV_ERROR) {
680 return res;
681 } else { /* DW_DLV_NO_ENTRY */
682 /* Normal non-COMDAT. groupnumber is zero. */
683 }
684
685 /* BUILDING_MAP. See also BUILDING_SECTIONS, SETUP_SECTION */
686 if (!groupnumber) {
687 res =_dwarf_dwo_groupnumber_given_name(scn_name,
688 &groupnumber);
689 /* DW_DLV_ERROR impossible here. */
690 if (res == DW_DLV_OK) {
691 /* groupnumber set 2 */
692 } else {
693 /* This is what it has to be.
694 .rela in here too. */
695 groupnumber = DW_GROUPNUMBER_BASE;
696 }
697 }
698 if (is_a_relx_section(scn_name,(int)doas.as_type,
699 &is_rela)) {
700 continue;
701 }
702
703 /* ASSERT: groupnumber non-zero now */
704 if (!is_a_special_section_semi_dwarf(scn_name)) {
705 if (mapgroupnumber) {
706 /* Already in group map */
707 continue;
708 }
709 /* !mapgroupnumber */
710 res = _dwarf_insert_in_group_map(dbg,
711 (unsigned)groupnumber,
712 (unsigned)obj_section_index,
713 scn_name,
714 error);
715 if (res != DW_DLV_OK) {
716 return res;
717 }
718 if (groupnumber == 1) {
719 found_group_one++;
720 } else if (groupnumber == 2) {
721 found_group_two++;
722 } else { /* fall through to continue */ }
723 continue;
724 }
725 }
726 if (found_group_two) {
727 ++grp->gd_number_of_groups;
728 }
729 if (found_group_one) {
730 *group_number_out = DW_GROUPNUMBER_BASE;
731 ++grp->gd_number_of_groups;
732 } else {
733 if (found_group_two) {
734 *group_number_out = DW_GROUPNUMBER_DWO;
735 } else {
736 if (lowest_comdat_groupnum) {
737 *group_number_out = lowest_comdat_groupnum;
738 } else {
739 *group_number_out = DW_GROUPNUMBER_BASE;
740 }
741 }
742 }
743 return DW_DLV_OK;
744}

746static int
747_dwarf_setup(Dwarf_Debug dbg, Dwarf_Error * error)
748{
749 const char *scn_name = 0;
750 struct Dwarf_Obj_Access_Interface_a_s * obj = 0;
751 int resn = 0;
752 struct Dwarf_Section_s **sections = 0;
753 Dwarf_Small endianness = 0;
754 Dwarf_Unsigned section_count = 0;
755 unsigned default_group_number = 0;
756 unsigned foundDwarf = FALSE;
757 Dwarf_Unsigned obj_section_index = 0;
758
759 dbg->de_assume_string_in_bounds =
760 _dwarf_assume_string_in_bounds;
761 /* First make an arbitrary assumption. */
762 dbg->de_copy_word = _dwarf_memcpy_noswap_bytes;
763 obj = dbg->de_obj_file;
764 endianness = obj->ai_methods->om_get_byte_order(obj->ai_object);
765 /* Then adjust any changes we need. */
766#ifdef WORDS_BIGENDIAN
767 dbg->de_big_endian_object = 1;
768 if (endianness == DW_END_little) {
769 dbg->de_big_endian_object = 0;
770 dbg->de_copy_word = _dwarf_memcpy_swap_bytes;
771 }
772#else /* little endian */
773 dbg->de_big_endian_object = 0;
774 if (endianness == DW_END_big ) {
775 dbg->de_big_endian_object = 1;
776 dbg->de_copy_word = _dwarf_memcpy_swap_bytes;
777 }
778#endif /* !WORDS_BIGENDIAN */
779
780 /* The following de_length_size is Not Too Significant.
781 Only used one calculation, and an approximate one
782 at that. */
783 dbg->de_length_size = obj->ai_methods->
784 om_get_length_size(obj->ai_object);
785 dbg->de_pointer_size =
786 obj->ai_methods->om_get_pointer_size(obj->ai_object);
787 section_count = obj->ai_methods->
788 om_get_section_count(obj->ai_object);
789 resn = determine_target_group(section_count,obj,
790 &default_group_number,dbg,error);
791 if (resn == DW_DLV_ERROR) {
792 return DW_DLV_ERROR;
793 }
794 if (dbg->de_groupnumber == DW_GROUPNUMBER_ANY) {
795 dbg->de_groupnumber = default_group_number;
796 }
797 /* Allocate space to record references to debug sections
798 that can be referenced by RELA sections in
799 the 'sh_info' field. */
800 sections = (struct Dwarf_Section_s **)calloc(section_count + 1,
801 sizeof(struct Dwarf_Section_s *));
802 if (!sections) {
803 /* Impossible case, we hope. Give up. */
804 _dwarf_error(dbg, error, DW_DLE_SECTION_ERROR);
805 return DW_DLV_ERROR;
806 }
807
808 /* We can skip index 0 when considering ELF files, but not other
809 object types. Indeed regardless of the object type we should
810 skip section 0 here.
811 This is a convention. We depend on it.
812 Non-elf object access code should
813 (in itself) understand we will index beginning at 1 and adjust
814 itself to deal with this Elf convention. Without this
815 convention various parts of the code in this file won't
816 work correctly.
817 A dss_index of 0 must not be used, even though we start at 0
818 here. So the get_section_info() must adapt to the situation
819 (the elf version does automatically as a result of Elf having
820 a section zero with zero length and an empty name). */
821
822 /* ASSERT: all group map entries set up. */
823
824 for (obj_section_index = 0; obj_section_index < section_count;
825 ++obj_section_index) {
826
827 struct Dwarf_Obj_Access_Section_a_s doas;
828 int res = DW_DLV_ERROR;
829 int err = 0;
830 unsigned groupnumber = 0;
831 unsigned mapgroupnumber = 0;
832 int is_rela = FALSE;
833
834 res = _dwarf_section_get_target_group_from_map(dbg,
835 (unsigned int)obj_section_index, &groupnumber,error);
836 if (res == DW_DLV_OK ) {
837 /* groupnumber is set. Fall through */
838 mapgroupnumber = groupnumber;
839 } else if (res == DW_DLV_ERROR) {
840 free(sections);
841 return res;
842 } else { /* DW_DLV_NO_ENTRY */
843 /* fall through, a BASE or DWO group, possibly */
844 }
845 memset(&doas,0,sizeof(doas));
846
847 res = obj->ai_methods->om_get_section_info(obj->ai_object,
848 obj_section_index,
849 &doas, &err);
850 if (res == DW_DLV_NO_ENTRY){
851 free(sections);
852 return res;
853 }
854 if (res == DW_DLV_ERROR){
855 free(sections);
856 DWARF_DBG_ERROR(dbg, err, DW_DLV_ERROR);
857 }
858 scn_name = doas.as_name;
859 if (!groupnumber) {
860 /* This finds dwo sections, group 2 */
861 res = _dwarf_dwo_groupnumber_given_name(scn_name,
862 &groupnumber);
863 if (res == DW_DLV_NO_ENTRY) {
864 /* No, must be group 1 */
865 groupnumber = DW_GROUPNUMBER_BASE;
866 }
867 }
868 if (!this_section_dwarf_relevant(scn_name,
869 (int)doas.as_type,
870 &is_rela) ) {
871 continue;
872 }
873 if (!is_a_relx_section(scn_name,(int)doas.as_type,
874 &is_rela)
875 && !is_a_special_section_semi_dwarf(scn_name)) {
876 /* We do these actions only for group-related
877 sections. Do for .debug_info etc,
878 never for .strtab or .rela.*
879 We already tested for relevance, so that part
880 is not news. */
881 if (mapgroupnumber == dbg->de_groupnumber) {
882 /* OK. Mapped. Part of the group.. This will
883 catch the cases where there are versions of
884 a section in multiple COMDATs and in BASE
885 an DWO to get the right one */
886 } else {
887 /* This section not mapped into this group. */
888 if (groupnumber == 1 && dbg->de_groupnumber > 2 &&
889 !_dwarf_section_in_group_by_name(dbg,scn_name,
890 dbg->de_groupnumber)) {
891 /* Load the section (but as group 1) */
892 } else {
893 continue;
894 }
895 }
896 }
897 /* BUILDING_SECTIONS. See also BUILDING_MAP, SETUP_SECTION */
898 {
899 /* Build up the sections table and the
900 de_debug* etc pointers in Dwarf_Debug. */
901 struct Dwarf_dbg_sect_s *section = 0;
902 int found_match = FALSE;
903
904 res = is_section_name_known_already(dbg,scn_name);
905 if (res == DW_DLV_OK) {
906#if 0 /* Removed check for section duplication */
907 /* DUPLICATE */
908 DWARF_DBG_ERROR(dbg, DW_DLE_SECTION_DUPLICATION,
909 DW_DLV_ERROR);
910 /* Metrowerks does this nonsense */
911#endif
912 continue;
913 }
914 if (res == DW_DLV_ERROR) {
915 free(sections);
916 DWARF_DBG_ERROR(dbg, err, DW_DLV_ERROR);
917 }
918 /* No entry: new-to-us section, the normal case. */
919 res = _dwarf_enter_section_in_de_debug_sections_array(dbg,
920 scn_name, obj_section_index, groupnumber,&err);
921 if (res == DW_DLV_OK) {
922 section = &dbg->de_debug_sections[
923 dbg->de_debug_sections_total_entries-1];
924 res = get_basic_section_data(dbg,
925 section->ds_secdata, &doas,
926 obj_section_index,
927 groupnumber,
928 error,
929 section->ds_duperr,
930 section->ds_emptyerr);
931 if (res != DW_DLV_OK) {
932 free(sections);
933 return res;
934 }
935 sections[obj_section_index] = section->ds_secdata;
936 foundDwarf += section->ds_have_dwarf;
937 found_match = TRUE;
938 /* Normal section set up.
939 Fall through. */
940 } else if (res == DW_DLV_NO_ENTRY) {
941 /* We get here for relocation sections.
942 Fall through. */
943 } else {
944 free(sections);
945 DWARF_DBG_ERROR(dbg, err, DW_DLV_ERROR);
946 }
947
948 if (!found_match) {
949 /* For an object file with incorrect rel[a]
950 section name, the 'readelf' tool,
951 prints correct debug information,
952 as the tool takes the section type instead
953 of the section name. If the current section
954 is a RELA one and the 'sh_info'
955 refers to a debug section, add the
956 relocation data. */
957 if (is_a_relx_section(scn_name,
958 (int)doas.as_type, &is_rela)) {
959 if ( doas.as_info < section_count) {
960 if (sections[doas.as_info]) {
961 add_relx_data_to_secdata(
962 sections[doas.as_info],
963 &doas,
964 obj_section_index,is_rela);
965 }
966 } else {
967 /* Something is wrong with the ELF file. */
968 free(sections);
969 DWARF_DBG_ERROR(dbg, DW_DLE_ELF_SECT_ERR,
970 DW_DLV_ERROR);
971 }
972 }
973 }
974 /* Fetch next section */
975 }
976 }
977
978 /* Free table with section information. */
979 free(sections);
980 if (foundDwarf) {
981 return DW_DLV_OK;
982 }
983 return DW_DLV_NO_ENTRY;
984}

986/* There is one table per CU and one per TU, and each
987 table refers to the associated other DWARF data
988 for that CU or TU.
989 See DW_SECT_*
990
991 In DWARF4 the type units are in .debug_types
992 In DWARF5 the type units are in .debug_info.

995static int
996load_debugfission_tables(Dwarf_Debug dbg,Dwarf_Error *error)
997{
998 int i = 0;
999 if (dbg->de_debug_cu_index.dss_size ==0 &&
1000 dbg->de_debug_tu_index.dss_size ==0) {
1001 /* This is the normal case.
1002 No debug fission. Not a .dwp object. */
1003 return DW_DLV_NO_ENTRY;
1004 }
1005
1006 for (i = 0; i < 2; ++i) {
1007 Dwarf_Xu_Index_Header xuptr = 0;
1008 struct Dwarf_Section_s* dwsect = 0;
1009 Dwarf_Unsigned version = 0;
1010 Dwarf_Unsigned number_of_cols /* L */ = 0;
1011 Dwarf_Unsigned number_of_CUs /* N */ = 0;
1012 Dwarf_Unsigned number_of_slots /* M */ = 0;
1013 const char *secname = 0;
1014 int res = 0;
1015 const char *type = 0;
1016
1017 if (i == 0) {
1018 dwsect = &dbg->de_debug_cu_index;
1019 type = "cu";
1020 } else {
1021 dwsect = &dbg->de_debug_tu_index;
1022 type = "tu";
1023 }

1025 continue;
1026 }

1029 &number_of_CUs,&number_of_slots,

1031 if (res == DW_DLV_NO_ENTRY) {
1032 continue;
1033 }
1034 if (res != DW_DLV_OK) {
1035 return res;
1036 }
1037 if (i == 0) {
1038 dbg->de_cu_hashindex_data = xuptr;
1039 } else {
1040 dbg->de_tu_hashindex_data = xuptr;
1041 }
1042 }
1043 return DW_DLV_OK;

1046/*
1047 Use a Dwarf_Obj_Access_Interface to kick things off.
1048 All other init routines eventually use this one.
1049 The returned Dwarf_Debug contains a copy of *obj
1050 the callers copy of *obj may be freed whenever the caller
1051 wishes.
1052
1053 New March 2017. Enables dealing with DWARF5 split
1054 dwarf more fully. */
1055int
1056dwarf_object_init_b(Dwarf_Obj_Access_Interface_a* obj,
1057 Dwarf_Handler errhand,
1058 Dwarf_Ptr errarg,
1059 unsigned groupnumber,
1060 Dwarf_Debug* ret_dbg,
1061 Dwarf_Error* error)
1062{
1063 Dwarf_Debug dbg = 0;
1064 int setup_result = DW_DLV_OK;
1065 Dwarf_Unsigned filesize = 0;
1066
1067 if (!ret_dbg) {
1068 DWARF_DBG_ERROR(NULL,DW_DLE_DWARF_INIT_DBG_NULL,
1069 DW_DLV_ERROR);
1070 }
1071 /* Non-null *ret_dbg will cause problems dealing with
1072 DW_DLV_ERROR */
1073 *ret_dbg = 0;
1074 filesize = obj->ai_methods->om_get_filesize(obj->ai_object);
1075 /* Initializes Dwarf_Debug struct and returns
1076 a pointer to that empty record.
1077 Filesize is to set up a sensible default hash tree
1078 size. */
1079 dbg = _dwarf_get_debug(filesize);
1080 if (IS_INVALID_DBG(dbg)) {
1081 dwarf_finish(dbg);
1082 dbg = 0;
1083 DWARF_DBG_ERROR(dbg, DW_DLE_DBG_ALLOC, DW_DLV_ERROR);
1084 }
1085 dbg->de_errhand = errhand;
1086 dbg->de_errarg = errarg;
1087 dbg->de_frame_rule_initial_value = DW_FRAME_REG_INITIAL_VALUE;
1088 dbg->de_frame_reg_rules_entry_count = DW_FRAME_LAST_REG_NUM;
1089 dbg->de_frame_cfa_col_number = DW_FRAME_CFA_COL3;
1090 dbg->de_frame_same_value_number = DW_FRAME_SAME_VAL;
1091 dbg->de_frame_undefined_value_number = DW_FRAME_UNDEFINED_VAL;
1092 dbg->de_dbg = dbg;
1093 /* See dwarf_set_tied_dbg() dwarf_get_tied_dbg()
1094 and comments in dwarf_opaque.h*/
1095 dbg->de_primary_dbg = dbg;
1096 dbg->de_secondary_dbg = 0;
1097 dbg->de_errors_dbg = dbg;
1098
1099 dbg->de_obj_file = obj;
1100 dbg->de_filesize = filesize;
1101 dbg->de_groupnumber = groupnumber;
1102 setup_result = _dwarf_setup(dbg, error);
1103 if (setup_result == DW_DLV_OK) {
1104 int fission_result = load_debugfission_tables(dbg,error);
1105 /* In most cases we get
1106 setup_result == DW_DLV_NO_ENTRY here
1107 as having debugfission (.dwp objects)
1108 is fairly rare. */
1109 if (fission_result == DW_DLV_ERROR) {
1110 /* Something is very wrong. */
1111 setup_result = fission_result;
1112 }
1113 if (setup_result == DW_DLV_OK) {
1114 _dwarf_harmless_init(&dbg->de_harmless_errors,
1115 DW_HARMLESS_ERROR_CIRCULAR_LIST_DEFAULT_SIZE);
1116 *ret_dbg = dbg;
1117 /* This is the normal return. */
1118 return setup_result;
1119 }
1120 }
1121 if (setup_result == DW_DLV_NO_ENTRY) {
1122 _dwarf_free_all_of_one_debug(dbg);
1123 dbg = 0;
1124 /* ASSERT: _dwarf_free_all_of_one_debug() never returns
1125 DW_DLV_ERROR */
1126 return setup_result;
1127 }
1128 /* An error of some sort. Report it as well as
1129 possible.
1130 ASSERT: setup_result == DW_DLV_ERROR
1131 here */
1132 {
1133 Dwarf_Unsigned myerr = 0;
1134 dwarfstring msg;
1135
1136 dwarfstring_constructor(&msg);
1137 /* We cannot use any _dwarf_setup()
1138 error here as
1139 we are freeing dbg, making that error (setup
1140 as part of dbg) stale.
1141 Hence we have to make a new error without a dbg.
1142 But error might be NULL and the init call
1143 error-handler function might be set.
1144 */
1145 if (error && *error) {
1146 /* Preserve our _dwarf_setup error number, but
1147 this does not apply if error NULL. */
1148 /* *error safe */
1149 myerr = dwarf_errno(*error);
1150 /* *error safe */
1151 dwarfstring_append(&msg,dwarf_errmsg(*error));
1152 /* deallocate the soon-stale error pointer. */
1153 dwarf_dealloc_error(dbg,*error);
1154 /* *error safe */
1155 *error = 0;
1156 }
1157 /* The status we want to return here is of _dwarf_setup,
1158 not of the _dwarf_free_all_of_one_debug(dbg) call.
1159 So use a local status variable for the free. */
1160 _dwarf_free_all_of_one_debug(dbg);
1161 dbg = 0;
1162 if (myerr) {
1163 /* Use the _dwarf_setup error number.
1164 If error is NULL the following will issue
1165 a message on stderr, as without
1166 dbg there is no error-handler function.
1167 */
1168 _dwarf_error_string(dbg,error,myerr,
1169 dwarfstring_string(&msg));
1170 dwarfstring_destructor(&msg);
1171 } /* else return quietly, a serious error
1172 was already reported. */
1173 }
1174 return setup_result;
1175}

1306do_decompress(Dwarf_Debug dbg,
1307 struct Dwarf_Section_s *section,
1308 Dwarf_Error * error)
1309{

1327 section->dss_compressed_length = srclen;
1328 if (!strncmp("ZLIB",(const char *)src,4)) {
1329 unsigned i = 0;
1330 unsigned l = 8;
1331 unsigned char *c = src+4;
1332 for ( ; i < l; ++i,c++) {
1333 uncompressed_len <<= 8;
1334 uncompressed_len += *c;
1335 }
1336 src = src + 12;
1337 srclen -= 12;
1338 section->dss_uncompressed_length = uncompressed_len;
1339 section->dss_ZLIB_compressed = TRUE;
1340 } else if (flags & SHF_COMPRESSED) {
1341 /* The prefix is a struct:
1342 unsigned int type; followed by pad if following are 64bit!
1343 size-of-target-address size
1344 size-of-target-address
1345 */
1346 Dwarf_Small *ptr = (Dwarf_Small *)src;
1347 Dwarf_Unsigned type = 0;
1348 Dwarf_Unsigned size = 0;
1349 /* Dwarf_Unsigned addralign = 0; */
1350 unsigned fldsize = dbg->de_pointer_size;
1351 unsigned structsize = 3* fldsize;
1352 READ_UNALIGNED_CK(dbg,type,Dwarf_Unsigned,ptr,
1353 DWARF_32BIT_SIZE,
