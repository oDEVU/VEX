31 __attribute__((target("avx2")))
32 bool testSphereAVX(const glm::vec3& center, float radius) const {
33 __m256 cx = _mm256_set1_ps(center.x);
34 __m256 cy = _mm256_set1_ps(center.y);
35 __m256 cz = _mm256_set1_ps(center.z);
36 __m256 r = _mm256_set1_ps(-radius);
37
38 __m256 dot = _mm256_fmadd_ps(nx, cx, dist);
39 dot = _mm256_fmadd_ps(ny, cy, dot);
40 dot = _mm256_fmadd_ps(nz, cz, dot);
41 __m256 mask = _mm256_cmp_ps(dot, r, _CMP_LT_OQ);
42
43 int res = _mm256_movemask_ps(mask);
44
45 return (res & 0x3F) == 0;
46 }

193 bool Renderer::beginFrame(glm::uvec2 renderResolution, SceneRenderData& outData) {
194 try {
195 if (renderResolution.x == 0 || renderResolution.y == 0 ||
196 renderResolution.x > 32768 || renderResolution.y > 32768) {
197 return false;
198 }
199
200 if (renderResolution != m_r_context.currentRenderResolution || m_r_context.requestSwapchainRecreation) {
201 log(LogLevel::INFO, "Recreating Swapchain", m_r_context.currentRenderResolution.x, m_r_context.currentRenderResolution.y, renderResolution.x, renderResolution.y);
202 m_r_context.currentRenderResolution = renderResolution;
203 m_p_pipeline->updateViewport(renderResolution);
204 m_p_swapchainManager->recreateSwapchain();
205 m_lastUsedView = VK_NULL_HANDLE;
206 if (m_cachedImGuiDescriptor != VK_NULL_HANDLE) {
207 m_garbageDescriptors[m_r_context.currentFrame].push_back(m_cachedImGuiDescriptor);
208 }
209 m_cachedImGuiDescriptor = VK_NULL_HANDLE;
210 m_lastUsedView = m_r_context.lowResColorView;
211 updateScreenDescriptor(m_r_context.lowResColorView);
212 }
213
214 vkWaitForFences(m_r_context.device, 1, &m_r_context.inFlightFences[m_r_context.currentFrame], VK_TRUE, UINT64_MAX);
215
216 VkResult result = vkAcquireNextImageKHR(
217 m_r_context.device,
218 m_r_context.swapchain,
219 UINT64_MAX,
220 m_r_context.imageAvailableSemaphores[m_r_context.currentFrame],
221 VK_NULL_HANDLE,
222 &m_r_context.currentImageIndex
223 );
224
225 if (result == VK_ERROR_OUT_OF_DATE_KHR) {
226 m_p_swapchainManager->recreateSwapchain();
227 outData.isSwapchainValid = false;
228 log(LogLevel::WARNING, "Swapchain out of date");
229 return false;
230 } else if (result != VK_SUCCESS && result != VK_SUBOPTIMAL_KHR) {
231 throw_error("Failed to acquire swap chain image!");
232 }
233
234 vkResetFences(m_r_context.device, 1, &m_r_context.inFlightFences[m_r_context.currentFrame]);
235 vkResetCommandPool(m_r_context.device, m_r_context.commandPools[m_r_context.currentFrame], 0);
236
237 outData.commandBuffer = m_r_context.commandBuffers[m_r_context.currentFrame];
238 outData.frameIndex = m_r_context.currentFrame;
239 outData.imageIndex = m_r_context.currentImageIndex;
240 outData.isSwapchainValid = true;
241
242 VkCommandBufferBeginInfo beginInfo{};
243 beginInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
244 vkBeginCommandBuffer(outData.commandBuffer, &beginInfo);
245
246 return true;
247 } catch (const std::exception& e) {
248 log(LogLevel::ERROR, "beginFrame failed");
249 handle_critical_exception(e);
250 return false;
251 }

843 void Renderer::endFrame(SceneRenderData& data) {
844 //std::cout << "currentRes x:" << m_r_context.currentRenderResolution.x << ", y:" << m_r_context.currentRenderResolution.y << std::endl;
845 //std::cout << "swapchainExtent x:" << m_r_context.swapchainExtent.width << ", y:" << m_r_context.swapchainExtent.height << std::endl;
846 //std::cout << "IsValid: " << data.isSwapchainValid << std::endl;
847 if (!data.isSwapchainValid) return;
848
849 vkEndCommandBuffer(data.commandBuffer);
850
851 VkSubmitInfo submitInfo{};
852 submitInfo.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
853
854 VkSemaphore waitSemaphores[] = {m_r_context.imageAvailableSemaphores[data.frameIndex]};
855 VkPipelineStageFlags waitStages[] = {VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT};
856 submitInfo.waitSemaphoreCount = 1;
857 submitInfo.pWaitSemaphores = waitSemaphores;
858 submitInfo.pWaitDstStageMask = waitStages;
859 submitInfo.commandBufferCount = 1;
860 submitInfo.pCommandBuffers = &data.commandBuffer;
861
862 VkSemaphore signalSemaphores[] = {m_r_context.renderFinishedSemaphores[data.imageIndex]};
863 submitInfo.signalSemaphoreCount = 1;
864 submitInfo.pSignalSemaphores = signalSemaphores;
865
866 try {
867 if (vkQueueSubmit(m_r_context.graphicsQueue, 1, &submitInfo, m_r_context.inFlightFences[m_r_context.currentFrame]) != VK_SUCCESS) {
868 throw_error("Failed to submit draw command buffer!");
869 }
870
871 VkPresentInfoKHR presentInfo{};
872 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
873 presentInfo.waitSemaphoreCount = 1;
874 presentInfo.pWaitSemaphores = signalSemaphores;
875
876 VkSwapchainKHR swapchains[] = {m_r_context.swapchain};
877 presentInfo.swapchainCount = 1;
878 presentInfo.pSwapchains = swapchains;
879 presentInfo.pImageIndices = &m_r_context.currentImageIndex;
880
881 VkResult result = vkQueuePresentKHR(m_r_context.presentQueue, &presentInfo);
882
883 if (result == VK_ERROR_OUT_OF_DATE_KHR || result == VK_SUBOPTIMAL_KHR) {
884 m_p_swapchainManager->recreateSwapchain();
885 } else if (result != VK_SUCCESS) {
886 throw_error("Failed to present swap chain image!");
887 }
888 } catch (const std::exception& e) {
889 log(LogLevel::ERROR, "Queue Submit/Present failed");
890 handle_critical_exception(e);
891 }
892
893 m_r_context.currentFrame = (m_r_context.currentFrame + 1) % m_r_context.MAX_FRAMES_IN_FLIGHT;
894 }
