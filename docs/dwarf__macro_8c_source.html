56/* Given the dwarf macro string, return a pointer to
57 the value. Returns pointer to 0 byte at end of string
58 if no value found (meaning the value is the empty string).
59
60 Only understands well-formed .debug_macinfo
61 and .debug_macro strings.
62
63*/
64char *
65dwarf_find_macro_value_start(char *str)
66{
67 char *lcp;
68 int funclike = 0;
69
70 for (lcp = str; *lcp; ++lcp) {
71 switch (*lcp) {
72 case LEFTPAREN:
73 ++funclike;
74 break;
75 case RIGHTPAREN:
76 --funclike;
77 break;
78 case SPACE:
79 /* We allow extraneous spaces inside macro parameter **
80 list, just in case... This is not really needed. */
81 if (!funclike) {
82 return lcp + 1;
83 }
84 break;
85 default:
86 break;
87 }
88 }
89 /* Never found value: returns pointer to the 0 byte at end of
90 string.
91 Or maybe the parentheses are unbalanced! Compiler error?
92 */
93 return lcp;
94}

96/*
97 Try to keep fileindex correct in every Macro_Details
98 record by tracking file starts and ends.
99 Uses high water mark: space reused, not freed.
100 Presumption is that this makes sense for most uses.
101 STARTERMAX is set so that the array need not be expanded for
102 most files: it is the initial include file depth.
103*/
104struct macro_stack_s {
105 Dwarf_Signed *st_base;
106 long st_max;
107 long st_next_to_use;
108 int st_was_fault;
109};
110
111static void _dwarf_reset_index_macro_stack(struct macro_stack_s *ms);
112static void
113free_macro_stack(Dwarf_Debug dbg, struct macro_stack_s *ms)
114{
115 dwarf_dealloc(dbg,ms->st_base,DW_DLA_STRING);
116 _dwarf_reset_index_macro_stack(ms);
117}
118
119#define STARTERMAX 10
120static void
121_dwarf_reset_index_macro_stack(struct macro_stack_s *ms)
122{
123 ms->st_base = 0;
124 ms->st_max = 0;
125 ms->st_next_to_use = 0;
126 ms->st_was_fault = 0;
127}
128static int
129_dwarf_macro_stack_push_index(Dwarf_Debug dbg, Dwarf_Signed indx,
130 struct macro_stack_s *ms)
131{
132
133 if (!ms->st_max || ms->st_next_to_use >= ms->st_max) {
134 long new_size = ms->st_max;
135 Dwarf_Signed *newbase = 0;
136
137 if (!new_size) {
138 new_size = STARTERMAX;
139 }
140 new_size = new_size * 2;
141 newbase =
142 (Dwarf_Signed *)_dwarf_get_alloc(dbg, DW_DLA_STRING,
143 new_size * sizeof(Dwarf_Signed));
144 if (!newbase) {
145 /* just leave the old array in place */
146 ms->st_was_fault = 1;
147 return DW_DLV_ERROR;
148 }
149 if (ms->st_base) {
150 memcpy(newbase, ms->st_base,
151 ms->st_next_to_use * sizeof(*newbase));
152 dwarf_dealloc(dbg, ms->st_base, DW_DLA_STRING);
153 }
154 ms->st_base = newbase;
155 ms->st_max = new_size;
156 }
157 ms->st_base[ms->st_next_to_use] = indx;
158 ++ms->st_next_to_use;
159 return DW_DLV_OK;
160}
161
162static Dwarf_Signed
163_dwarf_macro_stack_pop_index(struct macro_stack_s *ms)
164{
165 if (ms->st_was_fault) {
166 return -1;
167 }
168 if (ms->st_next_to_use > 0) {
169 ms->st_next_to_use--;
170 return (ms->st_base[ms->st_next_to_use]);
171 } else {
172 ms->st_was_fault = 1;
173 }
174 return -1;
175}

177/* Starting at macro_offset in .debug_macinfo,
178 if maximum_count is 0, treat as if it is infinite.
179 get macro data up thru
180 maximum_count entries or the end of a compilation
181 unit's entries (whichever comes first).
182
183 .debug_macinfo never appears in a .dwp Package File.
184 So offset adjustment for such is not needed.
185*/
186
187int
188dwarf_get_macro_details(Dwarf_Debug dbg,
189 Dwarf_Off macro_offset,
190 Dwarf_Unsigned maximum_count,
191 Dwarf_Signed * entry_count,
192 Dwarf_Macro_Details ** details,
193 Dwarf_Error * error)
194{
195 Dwarf_Small *macro_base = 0;
196 Dwarf_Small *macro_end = 0;
197 Dwarf_Small *pnext = 0;
198 Dwarf_Unsigned endloc = 0;
199 unsigned char uc = 0;
200 unsigned long depth = 0;
201 /* By section 6.3.2 Dwarf3 draft 8/9,
202 the base file should appear as
203 DW_MACINFO_start_file. See
204 http://gcc.gnu.org/ml/gcc-bugs/2005-02/msg03442.html
205 on "[Bug debug/20253] New: [3.4/4.0 regression]:
206 Macro debug info broken due to lexer change" for how
207 gcc is broken in some versions. We no longer use
208 depth as a stopping point, it's not needed as a
209 stopping point anyway. */
210 int res = 0;
211 /* count space used by strings */
212 unsigned long str_space = 0;
213 int done = 0;
214 unsigned long space_needed = 0;
215 unsigned long space_used = 0;
216 unsigned long string_offset = 0;
217 Dwarf_Small *return_data = 0;
218 Dwarf_Small *pdata = 0;
219 unsigned long final_count = 0;
220 Dwarf_Signed fileindex = -1;
221 Dwarf_Small *latest_str_loc = 0;
222 struct macro_stack_s msdata;
223
224 unsigned long count = 0;
225 unsigned long max_count = (unsigned long) maximum_count;
226 _dwarf_reset_index_macro_stack(&msdata);
227 if (IS_INVALID_DBG(dbg)) {
228 _dwarf_error_string(NULL, error, DW_DLE_DBG_NULL,
229 "DW_DLE_DBG_NULL: Either null or it contains"
230 "a stale Dwarf_Debug pointer");
231 free_macro_stack(dbg,&msdata);
232 return DW_DLV_ERROR;
233 }
234
235 res = _dwarf_load_section(dbg, &dbg->de_debug_macinfo,error);
236 if (res != DW_DLV_OK) {
237 free_macro_stack(dbg,&msdata);
238 return res;
239 }
240 if (!dbg->de_debug_abbrev.dss_size) {
241 free_macro_stack(dbg,&msdata);
242 return DW_DLV_NO_ENTRY;
243 }
244
245 macro_base = dbg->de_debug_macinfo.dss_data;
246 if (macro_base == NULL) {
247 free_macro_stack(dbg,&msdata);
248 return DW_DLV_NO_ENTRY;
249 }
250 if (macro_offset >= dbg->de_debug_macinfo.dss_size) {
251 free_macro_stack(dbg,&msdata);
252 return DW_DLV_NO_ENTRY;
253 }
254 macro_end = macro_base + dbg->de_debug_macinfo.dss_size;
255 /* FIXME debugfission is NOT handled here. */
256
257 pnext = macro_base + macro_offset;
258 if (maximum_count == 0) {
259 max_count = ULONG_MAX;
260 }
261
262 /* how many entries and how much space will they take? */
263 endloc = (pnext - macro_base);
264 if (endloc >= dbg->de_debug_macinfo.dss_size) {
265 if (endloc == dbg->de_debug_macinfo.dss_size) {
266 /* normal: found last entry */
267 free_macro_stack(dbg,&msdata);
268 return DW_DLV_NO_ENTRY;
269 }
270 _dwarf_error(dbg, error, DW_DLE_DEBUG_MACRO_LENGTH_BAD);
271 free_macro_stack(dbg,&msdata);
272 return DW_DLV_ERROR;
273 }
274 for (count = 0; !done && count < max_count; ++count) {
275 unsigned long slen = 0;
276
277 uc = *pnext;
278 ++pnext; /* get past the type code */
279 switch (uc) {
280 case DW_MACINFO_define:
281 case DW_MACINFO_undef:
282 /* line, string */
283 case DW_MACINFO_vendor_ext:
284 /* number, string */
285 SKIP_LEB128_CK(pnext,dbg,error,
286 macro_end);
287 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
288 dbg->de_debug_macinfo.dss_size) {
289 free_macro_stack(dbg,&msdata);
290 _dwarf_error(dbg, error,
291 DW_DLE_DEBUG_MACRO_INCONSISTENT);
292 return DW_DLV_ERROR;
293 }
294 res = _dwarf_check_string_valid(dbg,
295 macro_base,pnext,macro_end,
296 DW_DLE_MACINFO_STRING_BAD,error);
297 if (res != DW_DLV_OK) {
298 return res;
299 }
300 slen = (unsigned long)strlen((char *) pnext) + 1;
301 pnext += slen;
302 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
303 dbg->de_debug_macinfo.dss_size) {
304 free_macro_stack(dbg,&msdata);
305 _dwarf_error(dbg, error,
306 DW_DLE_DEBUG_MACRO_INCONSISTENT);
307 return DW_DLV_ERROR;
308 }
309 str_space += slen;
310 break;
311 case DW_MACINFO_start_file:
312 /* line, file index */
313 SKIP_LEB128_CK(pnext,dbg,error,
314 macro_end);
315 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
316 dbg->de_debug_macinfo.dss_size) {
317 free_macro_stack(dbg,&msdata);
318 _dwarf_error(dbg, error,
319 DW_DLE_DEBUG_MACRO_INCONSISTENT);
320 return DW_DLV_ERROR;
321 }
322 SKIP_LEB128_CK(pnext,dbg,error,
323 macro_end);
324 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
325 dbg->de_debug_macinfo.dss_size) {
326 free_macro_stack(dbg,&msdata);
327 _dwarf_error(dbg, error,
328 DW_DLE_DEBUG_MACRO_INCONSISTENT);
329 return DW_DLV_ERROR;
330 }
331 ++depth;
332 break;
333
334 case DW_MACINFO_end_file:
335 if (depth) {
336 --depth;
337 }
338 if (!depth) {
339 /* done = 1; no, do not stop here,
340 at least one gcc had
341 the wrong depth settings in the
342 gcc 3.4 timeframe. */
343 }
344 /* no string or number here */
345 break;
346 case 0:
347 /* end of cu's entries */
348 done = 1;
349 break;
350 default:
351 free_macro_stack(dbg,&msdata);
352 _dwarf_error(dbg, error, DW_DLE_DEBUG_MACRO_INCONSISTENT);
353 return DW_DLV_ERROR;
354 /* bogus macinfo! */
355 }
356
357 endloc = (pnext - macro_base);
358 if (endloc == dbg->de_debug_macinfo.dss_size) {
359 done = 1;
360 } else if (endloc > dbg->de_debug_macinfo.dss_size) {
361 _dwarf_error(dbg, error, DW_DLE_DEBUG_MACRO_LENGTH_BAD);
362 free_macro_stack(dbg,&msdata);
363 return DW_DLV_ERROR;
364 }
365 }
366 /* ASSERT: The above loop will never let us get here
367 with count < 1. No need to test for a zero count.
368
369 We have 'count' array entries to allocate and
370 str_space bytes of string space to provide for. */
371
372 string_offset = count * sizeof(Dwarf_Macro_Details);
373
374 /* extra 2 not really needed */
375 space_needed = string_offset + str_space + 2;
376 space_used = 0;
377 return_data = pdata = (Dwarf_Small *)_dwarf_get_alloc(
378 dbg, DW_DLA_STRING, space_needed);
379 latest_str_loc = pdata + string_offset;
380 if (pdata == 0) {
381 free_macro_stack(dbg,&msdata);
382 _dwarf_error(dbg, error, DW_DLE_DEBUG_MACRO_MALLOC_SPACE);
383 return DW_DLV_ERROR;
384 }
385 pnext = macro_base + macro_offset;
386
387 done = 0;
388
389 /* A series ends with a type code of 0. */
390
391 for (final_count = 0; !done && final_count < count;
392 ++final_count) {
393 unsigned long slen = 0;
394 Dwarf_Unsigned v1 = 0;
395 Dwarf_Macro_Details *pdmd = (Dwarf_Macro_Details *) (pdata +
396 (final_count * sizeof (Dwarf_Macro_Details)));
397
398 endloc = (pnext - macro_base);
399 if (endloc > dbg->de_debug_macinfo.dss_size) {
400 dwarf_dealloc(dbg,return_data,DW_DLA_STRING);
401 free_macro_stack(dbg,&msdata);
402 _dwarf_error(dbg, error, DW_DLE_DEBUG_MACRO_LENGTH_BAD);
403 return DW_DLV_ERROR;
404 }
405 uc = *pnext;
406 pdmd->dmd_offset = (pnext - macro_base);
407 pdmd->dmd_type = uc;
408 pdmd->dmd_fileindex = fileindex;
409 pdmd->dmd_lineno = 0;
410 pdmd->dmd_macro = 0;
411 ++pnext; /* get past the type code */
412 switch (uc) {
413 case DW_MACINFO_define:
414 case DW_MACINFO_undef:
415 /* line, string */
416 case DW_MACINFO_vendor_ext:
417 /* number, string */
418 res = _dwarf_leb128_uword_wrapper(dbg,&pnext,
419 macro_end,&v1,error);
420 if (res != DW_DLV_OK) {
421 free_macro_stack(dbg,&msdata);
422 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
423 return res;
424 }
425 pdmd->dmd_lineno = v1;
426
427 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
428 dbg->de_debug_macinfo.dss_size) {
429 free_macro_stack(dbg,&msdata);
430 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
431 _dwarf_error(dbg, error,
432 DW_DLE_DEBUG_MACRO_INCONSISTENT);
433 return DW_DLV_ERROR;
434 }
435 res = _dwarf_check_string_valid(dbg,
436 macro_base,pnext,macro_end,
437 DW_DLE_MACINFO_STRING_BAD,error);
438 if (res != DW_DLV_OK) {
439 dwarf_dealloc(dbg,return_data,DW_DLA_STRING);
440 return res;
441 }
442 slen = (unsigned long)strlen((char *) pnext) + 1;
443 _dwarf_safe_strcpy((char *)latest_str_loc,
444 space_needed - space_used,
445 (const char *)pnext,slen-1);
446 pdmd->dmd_macro = (char *) latest_str_loc;
447 latest_str_loc += slen;
448 space_used +=slen;
449 pnext += slen;
450 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
451 dbg->de_debug_macinfo.dss_size) {
452 free_macro_stack(dbg,&msdata);
453 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
454 _dwarf_error(dbg, error,
455 DW_DLE_DEBUG_MACRO_INCONSISTENT);
456 return DW_DLV_ERROR;
457 }
458 break;
459 case DW_MACINFO_start_file:
460 /* Line, file index */
461 res = _dwarf_leb128_uword_wrapper(dbg,&pnext,
462 macro_end,&v1,error);
463 if (res != DW_DLV_OK) {
464 free_macro_stack(dbg,&msdata);
465 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
466 return res;
467 }
468 pdmd->dmd_lineno = v1;
469 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
470 dbg->de_debug_macinfo.dss_size) {
471 free_macro_stack(dbg,&msdata);
472 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
473 _dwarf_error(dbg, error,
474 DW_DLE_DEBUG_MACRO_INCONSISTENT);
475 return DW_DLV_ERROR;
476 }
477 res = _dwarf_leb128_uword_wrapper(dbg,&pnext,
478 macro_end,&v1,error);
479 if (res != DW_DLV_OK) {
480 free_macro_stack(dbg,&msdata);
481 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
482 return res;
483 }
484 pdmd->dmd_fileindex = v1;
485 (void) _dwarf_macro_stack_push_index(dbg, fileindex,
486 &msdata);
487 /* We ignore the error, we just let
488 fileindex ** be -1 when
489 we pop this one. */
490 fileindex = v1;
491 if (((Dwarf_Unsigned)(pnext - macro_base)) >=
492 dbg->de_debug_macinfo.dss_size) {
493 free_macro_stack(dbg,&msdata);
494 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
495 _dwarf_error(dbg, error,
496 DW_DLE_DEBUG_MACRO_INCONSISTENT);
497 return DW_DLV_ERROR;
498 }
499 break;
500
501 case DW_MACINFO_end_file:
502 fileindex = _dwarf_macro_stack_pop_index(&msdata);
503 break; /* no string or number here */
504 case 0:
505 /* Type code of 0 means the end of cu's entries. */
506 done = 1;
507 break;
508 default:
509 /* Bogus macinfo! */
510 dwarf_dealloc(dbg, return_data, DW_DLA_STRING);
511 free_macro_stack(dbg,&msdata);
512 _dwarf_error(dbg, error, DW_DLE_DEBUG_MACRO_INCONSISTENT);
513 return DW_DLV_ERROR;
514 }
515 }
516 *entry_count = count;
517 *details = (Dwarf_Macro_Details *) return_data;
518 free_macro_stack(dbg,&msdata);
519 return DW_DLV_OK;
520}
