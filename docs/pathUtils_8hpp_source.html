25inline std::filesystem::path GetExecutableDir() {
26 std::filesystem::path exePath;
27
28#ifdef _WIN32
29 // Windows: Use GetModuleFileNameW
30 wchar_t wPath[MAX_PATH];
31 DWORD len = GetModuleFileNameW(NULL, wPath, MAX_PATH);
32 if (len == 0 || len == MAX_PATH) {
33 throw_error("Failed to get executable path on Windows");
34 }
35 // Convert UTF-16 to UTF-8
36 int utf8Size = WideCharToMultiByte(CP_UTF8, 0, wPath, -1, nullptr, 0, nullptr, nullptr);
37 if (utf8Size == 0) {
38 throw_error("Failed to convert executable path to UTF-8");
39 }
40 std::string utf8Path(utf8Size, '\0');
41 WideCharToMultiByte(CP_UTF8, 0, wPath, -1, utf8Path.data(), utf8Size, nullptr, nullptr);
42 exePath = std::filesystem::path(utf8Path);
43#elif defined(__APPLE__)
44 // macOS: Use _NSGetExecutablePath
45 char pathBuf[MAXPATHLEN];
46 uint32_t size = sizeof(pathBuf);
47 if (_NSGetExecutablePath(pathBuf, &size) != 0) {
48 throw_error("Failed to get executable path on macOS");
49 }
50 exePath = std::filesystem::canonical(pathBuf); // Resolve symlinks/aliases
51#else
52 // Linux: Use readlink
53 char pathBuf[1024];
54 ssize_t len = ::readlink("/proc/self/exe", pathBuf, sizeof(pathBuf) - 1);
55 if (len == -1) {
56 throw_error("Failed to read /proc/self/exe on Linux");
57 }
58 pathBuf[len] = '\0';
59 exePath = std::filesystem::path(pathBuf);
60#endif
61
62 return exePath.parent_path();
63}
