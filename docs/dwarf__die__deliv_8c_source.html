198/*
199 For a given Dwarf_Debug dbg, this function checks
200 if a CU that includes the given offset has been read
201 or not. If yes, it returns the Dwarf_CU_Context
202 for the CU. Otherwise it returns NULL. Being an
203 internal routine, it is assumed that a valid dbg
204 is passed.
205
206 **This is a sequential search. May be too slow.
207
208 If debug_info and debug_abbrev not loaded, this will
209 wind up returning NULL. So no need to load before calling
210 this.
211*/
212static Dwarf_CU_Context
213_dwarf_find_CU_Context(Dwarf_Debug dbg,
214 Dwarf_Off offset,
215 Dwarf_Bool is_info)
216{
217 Dwarf_CU_Context cu_context = 0;
218 Dwarf_Debug_InfoTypes dis = is_info? &dbg->de_info_reading:
219 &dbg->de_types_reading;
220
221 if (offset >= dis->de_last_offset){
222 return NULL;
223 }
224 if (dis->de_cu_context != NULL &&
225 dis->de_cu_context->cc_next != NULL &&
226 dis->de_cu_context->cc_next->cc_debug_offset == offset) {
227 return dis->de_cu_context->cc_next;
228 }
229 if (dis->de_cu_context != NULL &&
230 dis->de_cu_context->cc_debug_offset <= offset) {
231 for (cu_context = dis->de_cu_context;
232 cu_context != NULL;
233 cu_context = cu_context->cc_next) {
234 if (offset >= cu_context->cc_debug_offset &&
235 offset < cu_context->cc_debug_offset +
236 cu_context->cc_length + cu_context->cc_length_size
237 + cu_context->cc_extension_size) {
238 return cu_context;
239 }
240 }
241 }
242 for (cu_context = dis->de_cu_context_list;
243 cu_context != NULL;
244 cu_context = cu_context->cc_next) {
245 if (offset >= cu_context->cc_debug_offset &&
246 offset < cu_context->cc_debug_offset +
247 cu_context->cc_length + cu_context->cc_length_size
248 + cu_context->cc_extension_size) {
249 return cu_context;
250 }
251 }
252 return NULL;
253}

255int
256dwarf_get_debugfission_for_die(Dwarf_Die die,
257 struct Dwarf_Debug_Fission_Per_CU_s *fission_out,
258 Dwarf_Error *error)
259{
260 Dwarf_CU_Context context = 0;
261 Dwarf_Debug dbg = 0;
262 struct Dwarf_Debug_Fission_Per_CU_s * percu = 0;
263
264 CHECK_DIE(die, DW_DLV_ERROR);
265 context = die->di_cu_context;
266 dbg = context->cc_dbg;
267 if (!_dwarf_file_has_debug_fission_index(dbg)) {
268 return DW_DLV_NO_ENTRY;
269 }
270
271 /* Logic should work for DW4 and DW5. */
272 if (context->cc_unit_type == DW_UT_type||
273 context->cc_unit_type == DW_UT_split_type ) {
274 if (!_dwarf_file_has_debug_fission_tu_index(dbg)) {
275 return DW_DLV_NO_ENTRY;
276 }
277 } else if (context->cc_unit_type == DW_UT_split_compile) {
278 if (!_dwarf_file_has_debug_fission_cu_index(dbg)) {
279 return DW_DLV_NO_ENTRY;
280 }
281 } else { /* Fall through*/ }
282 percu = &context->cc_dwp_offsets;
283 if (!percu->pcu_type) {
284 return DW_DLV_NO_ENTRY;
285 }
286 *fission_out = *percu;
287 return DW_DLV_OK;
288}

393static int
394_dwarf_read_cu_version_and_abbrev_offset(Dwarf_Debug dbg,
395 Dwarf_Small *data,
396 Dwarf_Bool is_info,
397 unsigned offset_size, /* 4 or 8 */
398 Dwarf_CU_Context cu_context,
399 /* end_data used for sanity checking */
400 Dwarf_Small * end_data,
401 Dwarf_Unsigned * bytes_read_out,
402 Dwarf_Error * error)
403{
404 Dwarf_Small * data_start = data;
405 Dwarf_Small * dataptr = data;
406 Dwarf_Ubyte unit_type = 0;
407 Dwarf_Ubyte addrsize = 0;
408 Dwarf_Unsigned abbrev_offset = 0;
409 Dwarf_Half version = 0;
410
411 READ_UNALIGNED_CK(dbg, version, Dwarf_Half,
412 dataptr,DWARF_HALF_SIZE,error,end_data);
413 dataptr += DWARF_HALF_SIZE;
414 if (version == DW_CU_VERSION5) {
415 Dwarf_Ubyte unit_typeb = 0;
416 Dwarf_Unsigned herelen = sizeof(unit_typeb) +
417 sizeof(addrsize) + offset_size;
418
419 if ((dataptr+herelen) > end_data) {
420 _dwarf_error_string(dbg, error,
421 DW_DLE_CU_UT_TYPE_ERROR,
422 "DW_DLE_UT_TYPE_ERROR: "
423 " Reading the unit type, address size, "
424 "and abbrev_offset of the DWARF5 header"
425 " will run off the end of the section. "
426 "Corrupt DWARF");
427 }
428 READ_UNALIGNED_CK(dbg, unit_typeb, Dwarf_Ubyte,
429 dataptr, sizeof(unit_typeb),error,end_data);
430 dataptr += sizeof(unit_typeb);
431
432 unit_type = unit_typeb;
433 /* We do not need is_info flag in DWARF5 */
434 if (is_unknown_UT_value(unit_type)) {
435 /* DWARF5 object file is corrupt. Invalid value */
436 dwarfstring m;
437 dwarfstring_constructor(&m);
438 dwarfstring_append_printf_u(&m,
439 "DW_DLE_CU_UT_TYPE_ERROR: we do not know "
440 " the CU header unit_type 0x%x",unit_type);
441 dwarfstring_append_printf_u(&m," (%u) so cannot"
442 "process this compilation_unit. A valid type ",
443 unit_type);
444 dwarfstring_append(&m,"would be DW_UT_compile"
445 ", for example");
446 _dwarf_error_string(dbg, error,
447 DW_DLE_CU_UT_TYPE_ERROR,
448 dwarfstring_string(&m));
449 dwarfstring_destructor(&m);
450 return DW_DLV_ERROR;
451 }
452 READ_UNALIGNED_CK(dbg, addrsize, unsigned char,
453 dataptr, sizeof(addrsize),error,end_data);
454 dataptr += sizeof(char);
455
456 READ_UNALIGNED_CK(dbg, abbrev_offset, Dwarf_Unsigned,
457 dataptr, offset_size,error,end_data);
458 dataptr += offset_size;
459
460 } else if (version == DW_CU_VERSION2 ||
461 version == DW_CU_VERSION3 ||
462 version == DW_CU_VERSION4) {
463 Dwarf_Unsigned herelen = sizeof(addrsize) + offset_size;
464
465 if ((dataptr+herelen) > end_data) {
466 _dwarf_error_string(dbg, error,
467 DW_DLE_CU_UT_TYPE_ERROR,
468 "DW_DLE_UT_TYPE_ERROR: "
469 " Reading the address size, "
470 "and abbrev_offset of the DWARF header"
471 " will run off the end of the section. "
472 "Corrupt DWARF");
473 }
474 /* DWARF2,3,4 */
475 READ_UNALIGNED_CK(dbg, abbrev_offset, Dwarf_Unsigned,
476 dataptr, offset_size,error,end_data);
477 dataptr += offset_size;
478
479 READ_UNALIGNED_CK(dbg, addrsize, Dwarf_Ubyte,
480 dataptr, sizeof(addrsize),error,end_data);
481 dataptr += sizeof(addrsize);
482
483 /* This is an initial approximation of unit_type.
484 For DW4 we will refine this after we
485 have built the CU header (by reading
486 CU_die)
487 */
488 unit_type = is_info?DW_UT_compile:DW_UT_type;
489 } else {
490 _dwarf_error(dbg, error, DW_DLE_VERSION_STAMP_ERROR);
491 return DW_DLV_ERROR;
492 }
493 cu_context->cc_version_stamp = version;
494 cu_context->cc_unit_type = unit_type;
495 cu_context->cc_address_size = addrsize;
496 cu_context->cc_abbrev_offset = abbrev_offset;
497 if (!addrsize) {
498 _dwarf_error(dbg,error,DW_DLE_ADDRESS_SIZE_ZERO);
499 return DW_DLV_ERROR;
500 }
501 if (addrsize < MINIMUM_ADDRESS_SIZE ||
502 addrsize > MAXIMUM_ADDRESS_SIZE ) {
503 _dwarf_create_address_size_dwarf_error(dbg,error,addrsize,
504 DW_DLE_ADDRESS_SIZE_ERROR,
505 "DW_DLE_ADDRESS_SIZE_ERROR::");
506 return DW_DLV_ERROR;
507 }
508 if (addrsize > sizeof(Dwarf_Addr)) {
509 _dwarf_create_address_size_dwarf_error(dbg,error,addrsize,
510 DW_DLE_ADDRESS_SIZE_ERROR,
511 "DW_DLE_ADDRESS_SIZE_ERROR: not representable"
512 " in Dwarf_Addr field.");
513 return DW_DLV_ERROR;
514 }
515
516 /* We are ignoring this. Can get it from DWARF5. */
517 cu_context->cc_segment_selector_size = 0;
518 *bytes_read_out = (dataptr - data_start);
519 return DW_DLV_OK;
520}

524static int
525read_info_area_length_and_check(Dwarf_Debug dbg,
526 Dwarf_CU_Context cu_context,
527 Dwarf_Unsigned offset,
528 Dwarf_Byte_Ptr *cu_ptr_io,
529 Dwarf_Unsigned section_size,
530 Dwarf_Byte_Ptr section_end_ptr,
531 Dwarf_Unsigned *max_cu_global_offset_out,
532 Dwarf_Error *error)
533{
534 Dwarf_Byte_Ptr cu_ptr = 0;
535 /* The following two will be either 0,4, or 8. */
536 Dwarf_Unsigned local_length_size = 0;
537 Dwarf_Unsigned local_extension_size = 0;
538
539 Dwarf_Unsigned max_cu_global_offset = 0;
540 Dwarf_Unsigned length = 0;
541
542 cu_ptr = *cu_ptr_io;
543 /* READ_AREA_LENGTH updates cu_ptr for consumed bytes */
544 READ_AREA_LENGTH_CK(dbg, length, Dwarf_Unsigned,

546 error,section_size,section_end_ptr);
547 if (!length) {
548 return DW_DLV_NO_ENTRY;
549 }
550
551 /* ASSERT: The following is either 4 or 8. */
552 cu_context->cc_length_size = (Dwarf_Small)local_length_size;
553 /* ASSERT: The following is either 0 or 4. */
554 cu_context->cc_extension_size = (Dwarf_Small)local_extension_size;
555 cu_context->cc_length = length;
556
557 /* This is a bare minimum, not the real max offset.
558 A preliminary sanity check. */
559 max_cu_global_offset = offset + length +
560 local_extension_size + local_length_size;
561 if (length > section_size ||
562 (length+local_length_size + local_extension_size)>
563 section_size) {
564 _dwarf_error(dbg, error, DW_DLE_CU_LENGTH_ERROR);
565 return DW_DLV_ERROR;
566 }
567 if (max_cu_global_offset > section_size) {
568 _dwarf_error(dbg, error, DW_DLE_CU_LENGTH_ERROR);
569 return DW_DLV_ERROR;
570 }
571 *cu_ptr_io = cu_ptr;
572 *max_cu_global_offset_out = max_cu_global_offset;
573 return DW_DLV_OK;
574}

576/* In DWARF4 GNU dwp there is a problem.
577 We cannot read the CU die and it's
578 DW_AT_GNU_dwo_id until we know the
579 section offsets from the index files.
580 Hence we do not know how to search the
581 index files by key. So search by offset.

583 There is no such problem in DWARF5.
584
585 We have not yet corrected the unit_type so, for DWARF4,
586 we check for simpler unit types.

589static int
590fill_in_dwp_offsets_if_present(Dwarf_Debug dbg,
591 Dwarf_CU_Context cu_context,
592 Dwarf_Sig8 * signaturedata,
593 Dwarf_Off offset,
594 Dwarf_Error *error)
595{
596 Dwarf_Half unit_type = cu_context->cc_unit_type;
597 const char * typename = 0;
598 Dwarf_Half ver = cu_context->cc_version_stamp;
599
600 if (unit_type == DW_UT_split_type ||
601 (ver == DW_CU_VERSION4 && unit_type == DW_UT_type)){
602 typename = "tu";
603 if (!_dwarf_file_has_debug_fission_tu_index(dbg) ){
604 /* nothing to do. */
605 return DW_DLV_OK;
606 }
607 } else if (unit_type == DW_UT_split_compile ||
608 (ver == DW_CU_VERSION4 &&
609 unit_type == DW_UT_compile)){
610 typename = "cu";
611 if (!_dwarf_file_has_debug_fission_cu_index(dbg) ){
612 /* nothing to do. */
613 return DW_DLV_OK;
614 }
615 } else {
616 /* nothing to do. */
617 return DW_DLV_OK;
618 }
619
620 if (cu_context->cc_signature_present) {
621 int resdf = 0;
622
623 resdf = dwarf_get_debugfission_for_key(dbg,
624 signaturedata,
625 typename,
626 &cu_context->cc_dwp_offsets,
627 error);
628 if (resdf == DW_DLV_ERROR) {
629 return resdf;
630 }
631 if (resdf == DW_DLV_NO_ENTRY) {
632 _dwarf_error_string(dbg, error,
633 DW_DLE_MISSING_REQUIRED_CU_OFFSET_HASH,
634 "DW_DLE_MISSING_REQUIRED_CU_OFFSET_HASH: "
635 " dwarf_get_debugfission_for_key returned"
636 " DW_DLV_NO_ENTRY, something is wrong");
637 return DW_DLV_ERROR;
638 }
639 } else {
640 int resdf = 0;
641
642 resdf = _dwarf_get_debugfission_for_offset(dbg,
643 offset,
644 typename,
645 &cu_context->cc_dwp_offsets,
646 error);
647 if (resdf == DW_DLV_ERROR) {
648 return resdf;
649 }
650 if (resdf == DW_DLV_NO_ENTRY) {
651 _dwarf_error_string(dbg, error,
652 DW_DLE_MISSING_REQUIRED_CU_OFFSET_HASH,
653 "DW_DLE_MISSING_REQUIRED_CU_OFFSET_HASH: "
654 " dwarf_get_debugfission_for_offset returned"
655 " DW_DLV_NO_ENTRY, something is wrong");
656 return DW_DLV_ERROR;
657 }
658 cu_context->cc_signature =
659 cu_context->cc_dwp_offsets.pcu_hash;
660 cu_context->cc_signature_present = TRUE;
661 }
662 return DW_DLV_OK;
663}

679 DW5: hash in skeleton CU die
680 Also pick up cc_str_offset_base and
681 any other base values. */
682 Dwarf_Die cudie = 0;
683 int resdwo = 0;
684
685 /* Must call the internal siblingof so
686 we do not depend on the dbg...de_cu_context
687 used by and for dwarf_cu_header_* calls.
688 Safe because we know the correct cu_context. */
689 resdwo = _dwarf_siblingof_internal(dbg,NULL,
690 cu_context,
691 cu_context->cc_is_info,
692 &cudie, error);
693 if (resdwo == DW_DLV_OK) {
694 Dwarf_Half cutag = 0;
695 Dwarf_Bool bad_pc_form = FALSE;
696 int resdwob = 0;
697 resdwob = find_cu_die_base_fields(dbg,
698 cu_context,
699 cudie, &bad_pc_form,
700 error);
701 if (resdwob == DW_DLV_NO_ENTRY) {
702 /* The CU die has no children or has
703 some other issue like DW_FORM_ref_addr
704 on a low or high pc attribute. (Metrowerks) */
705 if (local_cudie_return) {
706 *local_cudie_return = cudie;
707 } else {
708 dwarf_dealloc_die(cudie);
709 }
710 if (!bad_pc_form) {
711 cu_context->cc_cu_die_has_children = FALSE;
712 }
713 return DW_DLV_OK;
714 }
715 if (resdwob == DW_DLV_ERROR) {
716 /* Not applicable or an error */
717 dwarf_dealloc_die(cudie);
718 cudie = 0;
719 return resdwob;
720 }
721 resdwob = dwarf_tag(cudie,&cutag,error);
722 if (resdwob == DW_DLV_OK) {
723 cu_context->cc_cu_die_tag = cutag;
724 }
725 if (local_cudie_return) {
726 *local_cudie_return = cudie;
727 } else {
728 dwarf_dealloc_die(cudie);
729 }
730 return resdwob;
731 }
732 if (resdwo == DW_DLV_NO_ENTRY) {
733 /* no cudie. Empty CU. */
734 return DW_DLV_OK;
735 }
736 /* no cudie. DW_DLV_ERROR.*/
737 return resdwo;
738}

783/* This function is used to create a CU Context for
784 a compilation-unit that begins at offset in
785 .debug_info. The CU Context is attached to the
786 list of CU Contexts for this dbg. It is assumed
787 that the CU at offset has not been read before,
788 and so do not call this routine before making
789 sure of this with _dwarf_find_CU_Context().
790 Returns NULL on error. As always, being an
791 internal routine, assumes a good dbg.
792
793 The offset argument is global offset, the offset
794 in the section, irrespective of CUs.
795 The offset has the DWP Package File offset built in
796 as it comes from the actual section.
797
798 max_cu_local_offset is a local offset in this CU.
799 So zero of this field is immediately following the length
800 field of the CU header. so max_cu_local_offset is
801 identical to the CU length field.
802 max_cu_global_offset is the offset one-past the end
803 of this entire CU. */
804static int
805_dwarf_make_CU_Context(Dwarf_Debug dbg,
806 Dwarf_Off offset,Dwarf_Bool is_info,
807 Dwarf_CU_Context * context_out,Dwarf_Error * error)
808{
809 Dwarf_CU_Context cu_context = 0;
810 Dwarf_Unsigned length = 0;
811 Dwarf_Unsigned typeoffset = 0;
812 Dwarf_Sig8 signaturedata;
813 Dwarf_Unsigned types_extra_len = 0;
814 Dwarf_Unsigned max_cu_local_offset = 0;
815 Dwarf_Unsigned max_cu_global_offset = 0;
816 Dwarf_Byte_Ptr cu_ptr = 0;
817 Dwarf_Byte_Ptr section_end_ptr = 0;
818 int local_length_size = 0;
819 Dwarf_Unsigned bytes_read = 0;
820 const char * secname = 0;
821 Dwarf_Debug_InfoTypes dis = 0;
822 struct Dwarf_Section_s * secdp = 0;
823 Dwarf_Unsigned section_size = 0;
824 Dwarf_Half unit_type = 0;
825 Dwarf_Unsigned version = 0;
826 Dwarf_Small * dataptr = 0;
827 int res = 0;
828 if (is_info) {
829 secname = dbg->de_debug_info.dss_name;
830 dis = &dbg->de_info_reading;
831 secdp = &dbg->de_debug_info;
832 } else {
833 secname = dbg->de_debug_types.dss_name;
834 dis = &dbg->de_types_reading;
835 secdp = &dbg->de_debug_types;
836 }
837 section_size = secdp->dss_size;
838
839 signaturedata = dwarfsig8zero;
840 cu_context =
841 (Dwarf_CU_Context)_dwarf_get_alloc(dbg, DW_DLA_CU_CONTEXT, 1);
842 if (!cu_context) {
843 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
844 return DW_DLV_ERROR;
845 }
846 cu_context->cc_dbg = dbg;
847 cu_context->cc_is_info = is_info;
848
849 dataptr = is_info? dbg->de_debug_info.dss_data:
850 dbg->de_debug_types.dss_data;
851 /* Preliminary sanity checking. */
852 if (!dataptr) {
853 local_dealloc_cu_context(dbg,cu_context);
854 _dwarf_error(dbg, error, DW_DLE_INFO_HEADER_ERROR);
855 return DW_DLV_ERROR;
856 }
857 if (offset >= section_size) {
858 local_dealloc_cu_context(dbg,cu_context);
859 _dwarf_error(dbg, error, DW_DLE_INFO_HEADER_ERROR);
860 return DW_DLV_ERROR;
861 }
862 if ((offset+4) > section_size) {
863 local_dealloc_cu_context(dbg,cu_context);
864 _dwarf_error(dbg, error, DW_DLE_INFO_HEADER_ERROR);
865 return DW_DLV_ERROR;
866 }
867 section_end_ptr = dataptr+section_size;
868 cu_ptr = (Dwarf_Byte_Ptr) (dataptr+offset);
869
870 if (section_name_ends_with_dwo(secname)) {
871 cu_context->cc_is_dwo = TRUE;
872 }
873 res = read_info_area_length_and_check(dbg,
874 cu_context,
875 offset,
876 &cu_ptr,
877 section_size,
878 section_end_ptr,
879 &max_cu_global_offset,
880 error);
881 if (res != DW_DLV_OK) {
882 local_dealloc_cu_context(dbg,cu_context);
883 return res;
884 }
885 local_length_size = cu_context->cc_length_size;
886 length = cu_context->cc_length;
887 max_cu_local_offset = length;
888 res = _dwarf_read_cu_version_and_abbrev_offset(dbg,
889 cu_ptr,
890 is_info,
891 local_length_size,
892 cu_context,
893 section_end_ptr,
894 &bytes_read,error);
895 if (res != DW_DLV_OK) {
896 local_dealloc_cu_context(dbg,cu_context);
897 return res;
898 }
899 version = cu_context->cc_version_stamp;
900 cu_ptr += bytes_read;
901 unit_type = cu_context->cc_unit_type;
902 if (cu_ptr > section_end_ptr) {
903 local_dealloc_cu_context(dbg,cu_context);
904 _dwarf_error(dbg, error, DW_DLE_INFO_HEADER_ERROR);
905 return DW_DLV_ERROR;
906 }
907
908 /* In a dwp context, the abbrev_offset is
909 still incomplete.
910 We need to add in the base from the .debug_cu_index
911 or .debug_tu_index . Done below */
912
913 /* At this point, for DW4, the unit_type is not fully
914 correct as we don't know if it is a skeleton or
915 a split_compile or split_type */
916 if (version == DW_CU_VERSION5 ||
917 version == DW_CU_VERSION4) {
918 /* DW4/DW5 header fields, depending on UT type.
919 See DW5 section 7.5.1.x, DW4
920 data is a GNU extension of DW4. */
921 switch(unit_type) {
922 case DW_UT_split_type:
923 case DW_UT_type: {
924 types_extra_len = sizeof(Dwarf_Sig8) /* 8 */ +
925 local_length_size /*type_offset size*/;
926 break;
927 }
928 case DW_UT_skeleton:
929 case DW_UT_split_compile: {
930 types_extra_len = sizeof(Dwarf_Sig8) /* 8 */;
931 break;
932 }
933 case DW_UT_compile: /* No additional fields */
934 case DW_UT_partial: /* No additional fields */
935 break;
936 default:
937 /* Data corruption in libdwarf? */
938 report_local_unit_type_error(dbg, unit_type,
939 "(DW4 or DW5)",error);
940 local_dealloc_cu_context(dbg,cu_context);
941 return DW_DLV_ERROR;
942 }
943 }
944
945 /* Compare the space following the length field
946 to the bytes in the CU header. */
947 if (length <
948 (CU_VERSION_STAMP_SIZE /* is 2 */ +
949 local_length_size /*for debug_abbrev offset */ +
950 CU_ADDRESS_SIZE_SIZE /* is 1 */ +
951 /* and finally size of the rest of the header: */
952 types_extra_len)) {
953
954 local_dealloc_cu_context(dbg,cu_context);
955 _dwarf_error_string(dbg, error, DW_DLE_CU_LENGTH_ERROR,
956 "DW_DLE_CU_LENGTH_ERROR: reading version "
957 "stamp and address size fields");
958 return DW_DLV_ERROR;
959 }
960 /* Now we can read the fields with some confidence,
961 we know the fields of the header are inside
962 the section. */
963
964 cu_context->cc_unit_type = unit_type;
965 switch(unit_type) {
966 case DW_UT_split_type:
967 case DW_UT_type: {
968 int tres = 0;
969 /* ASSERT: DW_CU_VERSION4 or DW_CU_VERSION5,
970 determined by logic above.
971 Now read the debug_types extra header fields of
972 the signature (8 bytes) and the typeoffset.
973 This can be in executable, ordinary object
974 (as in Type Unit),
975 there was no dwo in DWARF4
976 */
977 if ((cu_ptr + sizeof(signaturedata)) > section_end_ptr) {
978 _dwarf_error_string(dbg, error, DW_DLE_CU_LENGTH_ERROR,
979 "DW_DLE_CU_LENGTH_ERROR: reading "
980 "Dwarf_Sig8 signature field");
981 local_dealloc_cu_context(dbg,cu_context);
982 return DW_DLV_ERROR;
983 }
984 memcpy(&signaturedata,cu_ptr,sizeof(signaturedata));
985 cu_ptr += sizeof(signaturedata);
986 tres = _dwarf_read_unaligned_ck_wrapper(dbg,
987 &typeoffset,cu_ptr,local_length_size,
988 section_end_ptr,error);
989 if (tres != DW_DLV_OK ) {
990 local_dealloc_cu_context(dbg,cu_context);
991 return tres;
992 }
993 cu_context->cc_signature = signaturedata;
994 cu_context->cc_signature_present = TRUE;
995 cu_context->cc_signature_offset = typeoffset;
996 if (typeoffset >= max_cu_local_offset) {
997 local_dealloc_cu_context(dbg,cu_context);
998 _dwarf_error(dbg, error,
999 DW_DLE_DEBUG_TYPEOFFSET_BAD);
1000 return DW_DLV_ERROR;
1001 }
1002 }
1003 break;
1004 case DW_UT_skeleton:
1005 case DW_UT_split_compile: {
1006 if ((cu_ptr + sizeof(signaturedata)) > section_end_ptr) {
1007 _dwarf_error_string(dbg, error, DW_DLE_CU_LENGTH_ERROR,
1008 "DW_DLE_CU_LENGTH_ERROR: reading "
1009 "Dwarf_Sig8 signature field");
1010 local_dealloc_cu_context(dbg,cu_context);
1011 return DW_DLV_ERROR;
1012 }
1013 /* These unit types make a pair and
1014 paired units have identical signature.*/
1015 memcpy(&signaturedata,cu_ptr,sizeof(signaturedata));
1016 cu_context->cc_signature = signaturedata;
1017 cu_context->cc_signature_present = TRUE;
1018
1019 break;
1020 }
1021 /* The following with no additional fields */
1022 case DW_UT_compile:
1023 case DW_UT_partial:
1024 break;
1025 default: {
1026 /* Data corruption in libdwarf? */
1027 report_local_unit_type_error(dbg, unit_type,
1028 "",error);
1029 local_dealloc_cu_context(dbg,cu_context);
1030 return DW_DLV_ERROR;
1031 }
1032 }
1033 cu_context->cc_abbrev_hash_table =
1034 (Dwarf_Hash_Table) calloc(1,
1035 sizeof(struct Dwarf_Hash_Table_s));
1036 if (!cu_context->cc_abbrev_hash_table) {
1037 local_dealloc_cu_context(dbg,cu_context);
1038 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
1039 return DW_DLV_ERROR;
1040 }
1041 cu_context->cc_debug_offset = offset;
1042
1043 /* This is recording an overall section value for later
1044 sanity checking. */
1045 dis->de_last_offset = max_cu_global_offset;
1046 *context_out = cu_context;
1047 return DW_DLV_OK;
1048}

1300/*
1301 For a DWP/DWO the base fields
1302 of a CU are inherited from the skeleton.
1303 DWARF5 section 3.1.3
1304 "Split Full Compilation Unit Entries".
1305*/
1306static int
1307find_cu_die_base_fields(Dwarf_Debug dbg,
1308 Dwarf_CU_Context cucon,
1309 Dwarf_Die cudie,
1310 Dwarf_Bool *bad_pc_form,
1311 Dwarf_Error* error)
1312{
1313 Dwarf_CU_Context cu_context = 0;
1314 Dwarf_Attribute * alist = 0;
1315 Dwarf_Signed atcount = 0;
1316 unsigned version_stamp = 2;
1317 int alres = 0;
1318 Dwarf_Signed i = 0;
1319 Dwarf_Signed low_pc_attrnum = -1;
1320 Dwarf_Signed entry_pc_attrnum = -1;
1321 Dwarf_Signed at_addr_base_attrnum = -1;
1322
1323 cu_context = cudie->di_cu_context;
1324 version_stamp = cu_context->cc_version_stamp;
1325 alres = dwarf_attrlist(cudie, &alist,
1326 &atcount,error);
1327 if (alres != DW_DLV_OK) {
1328 /* Something is badly wrong. No attrlist! */
1329 return alres;
1330 }
1331 /* DW_AT_dwo_id and/or DW_AT_GNU_dwo_id
1332 are only found in some
1333 experimental DWARF4.
1334 Even DWARF3,4 use DW_AT_low_pc as base address
1335 DWARF5 changed CU header contents
1336 to make this attribute unnecessary.
1337 DW_AT_GNU_odr_signature is the same format,
1338 but is in a different namespace so not
1339 appropriate here..
1340 */
1341 for (i = 0; i < atcount; ++i) {
1342 Dwarf_Half attrnum = 0;
1343 Dwarf_Half form = 0;
1344 int ares = 0;
1345 int ares2 = 0;

1694finish_up_cu_context_from_cudie(Dwarf_Debug dbg,
1695 Dwarf_Unsigned offset,
1696 Dwarf_CU_Context cu_context,
1697 Dwarf_Die *cudie_return,

1701 Dwarf_Sig8 signaturedata = cu_context->cc_signature;
1702 int res = 0;

1704 /* Loads and initializes the dwarf .debug_cu_index
1705 and .debug_tu_index split dwarf package
1706 file sections */

1708 cu_context,
1709 &signaturedata,
1710 offset,

1749 (per the DWARF5 standard but for
1750 .debug_rnglists is not interited
1751 in spite of what DW5 says). */
1752 res = _dwarf_find_all_offsets_via_fission(dbg,
1753 cu_context,error);

1755 /* something seriously wrong. */
1756 return res;
1757 }
1758 }

1762/*
1763 CU_Contexts do not overlap.
1764 cu_context we see here is not in the list we
1765 are updating. See _dwarf_find_CU_Context()
1766
1767 Invariant: cc_debug_offset in strictly
1768 ascending order in the list.
1769 Never returns DW_DLV_NO_ENTRY
1770*/
1771static int
1772insert_into_cu_context_list(Dwarf_Debug_InfoTypes dis,
1773 Dwarf_CU_Context icu_context)
1774{
1775 Dwarf_Unsigned ioffset = icu_context->cc_debug_offset;
1776 Dwarf_Unsigned eoffset = 0;
1777 Dwarf_Unsigned hoffset = 0;
1778 Dwarf_Unsigned coffset = 0;
1779 Dwarf_CU_Context next = 0;
1780 Dwarf_CU_Context past = 0;
1781 Dwarf_CU_Context cur = 0;
1782
1783 /* Add the context into the section context list.
1784 This is the one and only place where it is
1785 saved for re-use and eventual dealloc. */
1786 if (!dis->de_cu_context_list) {
1787 /* First cu encountered. */
1788 dis->de_cu_context_list = icu_context;
1789 dis->de_cu_context_list_end = icu_context;
1790 return DW_DLV_OK;
1791 }
1792 if (!dis->de_cu_context_list_end) {
1793 return DW_DLV_ERROR;
1794 }
1795 eoffset = dis->de_cu_context_list_end->cc_debug_offset;
1796 if (eoffset < ioffset) {
1797 /* Normal case, add at end. */
1798 dis->de_cu_context_list_end->cc_next = icu_context;
1799 dis->de_cu_context_list_end = icu_context;
1800 return DW_DLV_OK;
1801 }
1802 hoffset = dis->de_cu_context_list->cc_debug_offset;
1803 if (hoffset > ioffset) {
1804 /* insert as new head. Unusual. */
1805 next = dis->de_cu_context_list;
1806 dis->de_cu_context_list = icu_context;
1807 dis->de_cu_context_list->cc_next = next;
1808 /* No need to touch de_cu_context_list_end */
1809 return DW_DLV_OK;
1810 }
1811 cur = dis->de_cu_context_list;
1812 past = 0;
1813 /* Insert in middle somewhere. Neither at
1814 start nor end.
1815 ASSERT: cur non-null
1816 ASSERT: past non-null */
1817 past = cur;
1818 cur = cur->cc_next;
1819 for ( ; cur ; cur = next) {
1820 next = cur->cc_next;
1821 coffset = cur->cc_debug_offset;
1822 if (coffset > ioffset) {
1823 /* Insert before cur, using past.
1824 ASSERT: past non-null */
1825 past->cc_next = icu_context;
1826 icu_context->cc_next = cur;
1827 return DW_DLV_OK;
1828 }
1829 past = cur;
1830 }
1831 /* Impossible, for end, coffset (ie, eoffset) > ioffset */
1832 /* NOTREACHED */
1833 return DW_DLV_ERROR;
1834}
