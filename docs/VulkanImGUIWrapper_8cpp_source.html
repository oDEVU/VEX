108 void VulkanImGUIWrapper::processEvent(const SDL_Event* event) {
109 ImGuiIO& io = ImGui::GetIO();
110
111 // Calculate scale factors (window-to-render)
112 float scaleX = (float)m_r_context.swapchainExtent.width /
113 (float)m_r_context.currentRenderResolution.x;
114 float scaleY = (float)m_r_context.swapchainExtent.height /
115 (float)m_r_context.currentRenderResolution.y;
116
117 switch (event->type) {
118 case SDL_EVENT_MOUSE_MOTION: {
119 io.MousePos = ImVec2(
120 event->motion.x / scaleX,
121 event->motion.y / scaleY
122 );
123 break;
124 }
125 case SDL_EVENT_MOUSE_BUTTON_DOWN:
126 case SDL_EVENT_MOUSE_BUTTON_UP: {
127 io.MousePos = ImVec2(
128 event->button.x / scaleX,
129 event->button.y / scaleY
130 );
131 ImGui_ImplSDL3_ProcessEvent(event);
132 break;
133 }
134 default: {
135 ImGui_ImplSDL3_ProcessEvent(event);
136 break;
137 }
138 }
139 }

145 void VulkanImGUIWrapper::createDescriptorPool() {
146 log("Creating ImGUI DescriptorPools");
147
148 VkDescriptorPoolSize pool_sizes[] = {
149 { VK_DESCRIPTOR_TYPE_SAMPLER, 1000 },
150 { VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, 1000 },
151 { VK_DESCRIPTOR_TYPE_SAMPLED_IMAGE, 1000 },
152 { VK_DESCRIPTOR_TYPE_STORAGE_IMAGE, 1000 },
153 { VK_DESCRIPTOR_TYPE_UNIFORM_TEXEL_BUFFER, 1000 },
154 { VK_DESCRIPTOR_TYPE_STORAGE_TEXEL_BUFFER, 1000 },
155 { VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, 1000 },
156 { VK_DESCRIPTOR_TYPE_STORAGE_BUFFER, 1000 },
157 { VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER_DYNAMIC, 1000 },
158 { VK_DESCRIPTOR_TYPE_STORAGE_BUFFER_DYNAMIC, 1000 },
159 { VK_DESCRIPTOR_TYPE_INPUT_ATTACHMENT, 1000 }
160 };
161
162 VkDescriptorPoolCreateInfo pool_info = {};
163 pool_info.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
164 pool_info.flags = VK_DESCRIPTOR_POOL_CREATE_FREE_DESCRIPTOR_SET_BIT;
165 pool_info.maxSets = 1000 * IM_ARRAYSIZE(pool_sizes);
166 pool_info.poolSizeCount = (uint32_t)IM_ARRAYSIZE(pool_sizes);
167 pool_info.pPoolSizes = pool_sizes;
168
169 if (vkCreateDescriptorPool(m_r_context.device, &pool_info, nullptr, &m_imguiPool) != VK_SUCCESS) {
170 throw_error("Failed to create descriptor pool for ImGui");
171 }
172 }
