96 enviroment env;
97 if (json.contains("environment") && json["environment"].contains("shading")) {
98 log("Loading shading settings from scene");
99 const auto& shading = json["environment"]["shading"];
100 env.gourardShading = shading.value("gouraud", env.gourardShading);
101 env.passiveVertexJitter = shading.value("passiveVertexJitter", env.passiveVertexJitter);
102 env.vertexSnapping = shading.value("vertexSnapping", env.vertexSnapping);
103 env.affineWarping = shading.value("affineTextureWarping", env.affineWarping);
104 env.colorQuantization = shading.value("colorQuantization", env.colorQuantization);
105 env.ntfsArtifacts = shading.value("ntfsArtifacts", env.ntfsArtifacts);
106 }

108 if (json.contains("environment") && json["environment"].contains("lighting")) {
109 log("Loading lighting settings from scene");
110 const auto& lighting = json["environment"]["lighting"];
111 env.ambientLightStrength = lighting.value("ambientLightStrength", env.ambientLightStrength);
112
113 if (lighting.contains("ambientLight") && lighting["ambientLight"].is_array() && lighting["ambientLight"].size() >= 3) {
114 env.ambientLight = glm::vec3(

116 lighting["ambientLight"][1].get<float>(),
117 lighting["ambientLight"][2].get<float>()
118 );
119 }
120
121 if (lighting.contains("sunLight") && lighting["sunLight"].is_array() && lighting["sunLight"].size() >= 3) {
122 env.sunLight = glm::vec3(
123 lighting["sunLight"][0].get<float>(),
124 lighting["sunLight"][1].get<float>(),
125 lighting["sunLight"][2].get<float>()

138 env.clearColor = glm::vec3(

140 lighting["clearColor"][1].get<float>(),
141 lighting["clearColor"][2].get<float>()
142 );

146 engine.setEnvironmentSettings(env);
147
148 auto objects = json["objects"];
149 if (!objects.is_array()) {
150 log("Error: Scene file must have 'objects' array");
151 return;
152 }
153
154 for (const auto& obj : objects) {
155 std::string type = obj.value("type", "");
156 std::string name = obj.value("name", "");
157 if (type.empty() || name.empty()) {
158 log("Error: Object missing type or name");
159 continue;
160 }
161
162 GameObject* gameObj = GameObjectFactory::getInstance().create(type, engine, name);
163 if (!gameObj) {
164 log("Error: Failed to create GameObject of type '%s'", type.c_str());
165 continue;
166 }
167
168 auto components = obj["components"];
169 if (!components.is_array()) {
170 log("Object '%s' has no components to load", name.c_str());
171 } else {
172 for (const auto& comp : components) {
173 std::string compType = comp.value("type", "");
174 if (compType.empty()) {
175 log("Error: Component missing type for object '%s'", name.c_str());
176 continue;
177 }
178 ComponentRegistry::getInstance().loadComponent(*gameObj, compType, comp);
179 }
180 }
181
182 std::string parent = obj.value("parent", "");
183 if (!parent.empty()) {
184 for(const auto& m_obj : m_objects) {
185 if (m_obj->GetComponent<NameComponent>().name == parent) {
186 gameObj->ParentTo(m_obj->GetEntity());
187 //gameObj->GetComponent<TransformComponent>().parent = m_obj->GetEntity();
188 log("Parented object '%s' to '%s'", gameObj->GetComponent<NameComponent>().name.c_str(), m_obj->GetComponent<NameComponent>().name.c_str());
189 break;
190 }
191 }
192 }
193
194 if (gameObj) {
195 m_objects.emplace_back(gameObj);
196 }
197 }
