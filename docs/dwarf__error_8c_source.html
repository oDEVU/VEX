59/* This function performs error handling as described in the
60 libdwarf consumer document section 3. Dbg is the Dwarf_debug
61 structure being processed. Error is a pointer to the pointer
62 to the error descriptor that will be returned. Errval is an
63 error code listed in dwarf_error.h.
64
65 If the malloc arena is exhausted we return a pointer to
66 a special static error record. This special singleton
67 is mostly ignored by dwarf_dealloc().
68 Users should not be storing Dwarf_Error pointers
69 for long so this singleton is only going to cause
70 confusion when callers try to save an out-of-memory
71 Dwarf_Error pointer.
72 If the call provides no way to handle the error
73 the function simply returns, whereas it used
74 (before July 2021) to abort in that case.
75*/
76
77/* The user provides an explanatory string, the error
78 number itself explains little.
79 This prepends DW_DLE_USER_DECLARED_ERROR to the
80 caller-provided string.
81 New in April, 2020 . Used by dwarfdump in a few
82 circumstances. */
83void
84dwarf_error_creation(Dwarf_Debug dbg,
85 Dwarf_Error *err,
86 char *errmsg)
87{
88 dwarfstring m;
89
90 if (IS_INVALID_DBG(dbg)) {
91 return;
92 }
93 dwarfstring_constructor(&m);
94 dwarfstring_append(&m,"DW_DLE_USER_DECLARED_ERROR: ");
95 dwarfstring_append(&m,errmsg);
96 _dwarf_error_string(dbg,err,
97 DW_DLE_USER_DECLARED_ERROR,
98 dwarfstring_string(&m));
99 dwarfstring_destructor(&m);
100}

115/* Errors are all added to the de_primary_dbg, never to
116 de_secondary_dbg. */
117void
118_dwarf_error_string(Dwarf_Debug dbg, Dwarf_Error * error,
119 Dwarf_Signed errval,char *msg)
120{
121 Dwarf_Error errptr = 0;
122
123 /* Allow NULL dbg on entry, since sometimes that
124 can happen and we want to report the upper-level
125 error, not the null dbg error. */
126 if (error) {
127 /* If dbg is NULL, use the alternate error struct. However,
128 this will overwrite the earlier error. */
129 if (dbg) {
130 /* ERRORs are always associated with
131 de_primary_dbg so they can be returned
132 up the tree of calls on the stack
133 safely. */
134 errptr =
135 (Dwarf_Error) _dwarf_get_alloc(dbg->de_errors_dbg,
136 DW_DLA_ERROR, 1);
137 if (!errptr) {
138 errptr = &_dwarf_failsafe_error;
139 errptr->er_static_alloc = DE_STATIC;
140 } else {
141 errptr->er_static_alloc = DE_STANDARD;
142 }
143 } else {
144 /* We have no dbg to work with. dwarf_init
145 failed. We hack
146 up a special area. */
147 errptr = _dwarf_special_no_dbg_error_malloc();
148 if (!errptr) {
149 errptr = &_dwarf_failsafe_error;
150 errptr->er_static_alloc = DE_STATIC;
151#ifdef DEBUG
152 printf("libdwarf no dbg to dwarf_error_string,"
153 " fullystatic, "
154 "using DE_STATIC alloc, addr"
155 " 0x%lx line %d %s\n",
156 (unsigned long)errptr,
157 __LINE__,__FILE__);
158#endif /* DEBUG */
159 } else {
160 errptr->er_static_alloc = DE_MALLOC;
161
162#ifdef DEBUG
163 printf("libdwarf no dbg, add to static_err_list "
164 "static DE_MALLOC alloc, addr"
165 " 0x%lx line %d %s\n",
166 (unsigned long)errptr,
167 __LINE__,__FILE__);
168#endif /* DEBUG */
169 _dwarf_add_to_static_err_list(errptr);
170 }
171 }
172
173 errptr->er_errval = errval;
174 if (msg && errptr->er_static_alloc != DE_STATIC) {
175 dwarfstring *em = 0;
176
177#ifdef DEBUG
178 printf("libdwarf ALLOC creating error string"
179 " %s errval %ld errptr 0x%lx \n",
180 msg,(long)errval,(unsigned long)errptr);
181#endif /* DEBUG */
182 em = (dwarfstring *)calloc(1,sizeof(dwarfstring));
183 if (em) {
184 dwarfstring_constructor(em);
185 dwarfstring_append(em,msg);
186 errptr->er_msg = (void*)em;
187 }
188 }
189 *error = errptr;
190 return;
191 }
192
193 if (dbg && dbg->de_errhand != NULL) {
194 errptr = (Dwarf_Error) _dwarf_get_alloc(dbg->de_errors_dbg,
195 DW_DLA_ERROR, 1);
196 if (errptr == NULL) {
197 errptr = &_dwarf_failsafe_error;
198 errptr->er_static_alloc = DE_STATIC;
199 }
200 errptr->er_errval = errval;
201 dbg->de_errhand(errptr, dbg->de_errors_dbg->de_errarg);
202 return;
203 }
204 fflush(stderr);
205 fprintf(stderr,
206 "\nlibdwarf is unable to record error %s "
207 "No error argument or handler available\n",
208 dwarf_errmsg_by_number(errval));
209 fflush(stderr);
210 return;
211}
