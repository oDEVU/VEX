377/* The LOP, WHAT_IS_OPCODE stuff is here so it can
378 be reused in 3 places. Seemed hard to keep
379 the 3 places the same without an inline func or
380 a macro.
381
382 Handling the line section where the header and the
383 file being processed do not match (unusual, but
384 planned for in the design of .debug_line)
385 is too tricky to recode this several times and keep
386 it right.
387
388*/
389#define LOP_EXTENDED 1
390#define LOP_DISCARD 2
391#define LOP_STANDARD 3
392#define LOP_SPECIAL 4
393
394/* ASSERT: sets type to one of the above 4. Never anything else. */
395#define WHAT_IS_OPCODE(type,opcode,base,opcode_length,\
396line_ptr,highest_std) \
397 if ((opcode) < (base)) { \
398 /* we know we must treat as a standard op \
399 or a special case. */ \
400 if ((opcode) == DW_EXTENDED_OPCODE) { \
401 (type) = LOP_EXTENDED; \
402 } else if (((highest_std)+1) >= (base)) { \
403 /* == Standard case: compile of \
404 dwarf_line.c and object \
405 have same standard op codes set. \
406 == Special case: compile of dwarf_line.c \
407 has things in standard op codes list \
408 in dwarf.h header not \
409 in the object: handle this as a standard \
410 op code in switch below. \
411 The header special ops overlap the \
412 object standard ops. \
413 The new standard op codes will not \
414 appear in the object. */ \
415 (type) = LOP_STANDARD; \
416 } else { \
417 /* These are standard opcodes in the object \
418 ** that were not defined in the header \
419 ** at the time dwarf_line.c \
420 ** was compiled. Provides the ability of \
421 ** out-of-date dwarf reader to read newer \
422 ** line table data transparently. \
423 */ \
424 (type) = LOP_DISCARD; \
425 } \
426 } else { \
427 /* Is a special op code. */ \
428 (type) = LOP_SPECIAL; \
429 }
