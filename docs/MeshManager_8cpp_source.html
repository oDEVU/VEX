100 MeshComponent MeshManager::loadMesh(const std::string& path) {
101 MeshData meshData;
102 MeshComponent meshComponent;
103 try {
104 std::string realPath = GetAssetPath(path);
105 log("Loading mesh data from: %s", realPath.c_str());
106
107 if (!m_vfs->file_exists(realPath)) {
108 throw_error("File not found: " + realPath);
109 }
110
111 meshData.loadFromFile(realPath, m_vfs);
112 meshData.meshPath = realPath;
113 } catch (const std::exception& e) {
114 SDL_LogCritical(SDL_LOG_CATEGORY_ERROR, "Mesh load failed");
115 handle_exception(e);
