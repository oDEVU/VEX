98 MeshComponent MeshManager::loadMesh(const std::string& path) {
99 MeshData meshData;
100 MeshComponent meshComponent;
101 std::string realPath = GetAssetPath(path);
102
103 try {
104 log("Loading mesh data from: %s", realPath.c_str());
105
106 if (!m_vfs->file_exists(realPath)) {
107 throw_error("File not found: " + realPath);
108 }
109
110 meshData.loadFromFile(realPath, m_vfs);
111 meshData.meshPath = path;
112 } catch (const std::exception& e) {
113 log(LogLevel::ERROR, "Mesh load failed: %s", path.c_str());
114 handle_exception(e);
115 }
116
117 meshComponent.meshData = std::move(meshData);
118
119 glm::vec3 min = glm::vec3(FLT_MAX);
120 glm::vec3 max = glm::vec3(-FLT_MAX);
121
122 for(auto& submesh : meshComponent.meshData.submeshes) {
123 for(auto& vertex : submesh.vertices) {
124 min = glm::min(min, vertex.position);
125 max = glm::max(max, vertex.position);
126 }
127 }
128
129 glm::vec3 center = (min + max) * 0.5f;
130 float radius = glm::length(max - center);
131
132 meshComponent.localCenter = center;
133 meshComponent.localRadius = radius;
134
135 return meshComponent;
136 }
