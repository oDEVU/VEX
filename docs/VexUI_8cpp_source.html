83void VexUI::loadFonts(Widget* w) {
84 if (!w) return;
85
86 if (!w->style.font.empty() && w->style.fontSize > 0.f) {
87 std::string key = w->style.font + "_" + std::to_string(static_cast<int>(w->style.fontSize));
88 if (m_fontAtlases.count(key) > 0) goto recurse;
89
90 auto data = m_vfs->load_file(GetAssetPath(w->style.font));
91 if (!data) {
92 log("UI: cannot load font %s", w->style.font.c_str());
93 goto recurse;
94 }
95
96 FontAtlas atlas;
97 if (!stbtt_InitFont(&atlas.info, (unsigned char*)data->data.data(), 0)) {
98 log("UI: stbtt_InitFont failed for %s", w->style.font.c_str());
99 goto recurse;
100 }
101
102 const int W = 512, H = 512;
103 std::vector<unsigned char> bitmap(W * H, 0);
104 atlas.cdata.resize(96);
105 stbtt_BakeFontBitmap((unsigned char*)data->data.data(), 0, w->style.fontSize, bitmap.data(), W, H, 32, 96, atlas.cdata.data());
106
107 std::vector<unsigned char> rgba(W * H * 4);
108 for (int i = 0; i < W * H; ++i) {
109 unsigned char v = bitmap[i];
110 rgba[i*4 + 0] = 255;
111 rgba[i*4 + 1] = 255;
112 rgba[i*4 + 2] = 255;
113 rgba[i*4 + 3] = v;
114 }
115
116 std::string texName = "ui_font_" + key;
117 m_res->createTextureFromRaw(rgba, W, H, texName);
118
119 atlas.texIdx = m_res->getTextureIndex(texName);
120 atlas.width = W;
121 atlas.height = H;
122 atlas.bakedSize = w->style.fontSize;
123 m_fontAtlases[key] = atlas;
124 }
125
126recurse:
127 for (auto* c : w->children) loadFonts(c);
128}

138Widget* VexUI::parseNode(const nlohmann::json& j) {
139 Widget* w = new Widget();
140
141 if (j.contains("type")) {
142 std::string t = j["type"].get<std::string>();
143 if (t == "label") w->type = WidgetType::Label;
144 else if (t == "image") w->type = WidgetType::Image;
145 else if (t == "button") w->type = WidgetType::Button;
146 }
147
148 w->applyJson(j);
149
150 w->yoga = YGNodeNew();
151 if (j.contains("layout")) {
152 std::string l = j["layout"].get<std::string>();
153 if (l == "row") YGNodeStyleSetFlexDirection(w->yoga, YGFlexDirectionRow);
154 else if (l == "column") YGNodeStyleSetFlexDirection(w->yoga, YGFlexDirectionColumn);
155 }
156 if (j.contains("justify")) {
157 std::string jst = j["justify"].get<std::string>();
158 if (jst == "space-between") YGNodeStyleSetJustifyContent(w->yoga, YGJustifySpaceBetween);
159 else if (jst == "center") YGNodeStyleSetJustifyContent(w->yoga, YGJustifyCenter);
160 else if (jst == "space-evenly") YGNodeStyleSetJustifyContent(w->yoga, YGJustifySpaceEvenly);
161 else if (jst == "space-around") YGNodeStyleSetJustifyContent(w->yoga, YGJustifySpaceAround);
162 else if (jst == "flex-end") YGNodeStyleSetJustifyContent(w->yoga, YGJustifyFlexEnd);
163 else if (jst == "flex-start") YGNodeStyleSetJustifyContent(w->yoga, YGJustifyFlexStart);
164 }
165 if (j.contains("align")) {
166 std::string al = j["align"].get<std::string>();
167 if (al == "center") YGNodeStyleSetAlignItems(w->yoga, YGAlignCenter);
168 else if (al == "auto") YGNodeStyleSetAlignItems(w->yoga, YGAlignAuto);
169 else if (al == "flex-start") YGNodeStyleSetAlignItems(w->yoga, YGAlignFlexStart);
170 else if (al == "flex-end") YGNodeStyleSetAlignItems(w->yoga, YGAlignFlexEnd);
171 else if (al == "stretch") YGNodeStyleSetAlignItems(w->yoga, YGAlignStretch);
172 else if (al == "baseline") YGNodeStyleSetAlignItems(w->yoga, YGAlignBaseline);
173 else if (al == "space-around") YGNodeStyleSetAlignItems(w->yoga, YGAlignSpaceAround);
174 else if (al == "space-between") YGNodeStyleSetAlignItems(w->yoga, YGAlignSpaceBetween);
175 else if (al == "space-evenly") YGNodeStyleSetAlignItems(w->yoga, YGAlignSpaceEvenly);
176 }
177 if (j.contains("padding")) YGNodeStyleSetPadding(w->yoga, YGEdgeAll, j["padding"].get<float>());
178
179 if (j.contains("children") && j["children"].is_array()) {
180 for (const auto& c : j["children"]) {
181 Widget* child = parseNode(c);
182 w->children.push_back(child);
183 YGNodeInsertChild(w->yoga, child->yoga, static_cast<uint32_t>(w->children.size() - 1));
184 }
185 }
186 return w;

253void VexUI::batch(Widget* w, std::vector<float>& verts) {
254 if (!w) return;
255 float x = YGNodeLayoutGetLeft(w->yoga);
256 float y = YGNodeLayoutGetTop(w->yoga);
257 float width = YGNodeLayoutGetWidth(w->yoga);
258 float height = YGNodeLayoutGetHeight(w->yoga);
259
260 auto pushQuad = [&](float x0, float y0, float u0, float v0,
261 float x1, float y1, float u1, float v1,
262 const glm::vec4& col, float texIdx) {
263 // Two triangles to make a quad (6 vertices)
264 // Triangle 1
265 verts.insert(verts.end(), {x0, y0, u0, v0, col.r, col.g, col.b, col.a, texIdx});
266 verts.insert(verts.end(), {x1, y0, u1, v0, col.r, col.g, col.b, col.a, texIdx});
267 verts.insert(verts.end(), {x0, y1, u0, v1, col.r, col.g, col.b, col.a, texIdx});
268
269 // Triangle 2
270 verts.insert(verts.end(), {x1, y0, u1, v0, col.r, col.g, col.b, col.a, texIdx});
271 verts.insert(verts.end(), {x1, y1, u1, v1, col.r, col.g, col.b, col.a, texIdx});
272 verts.insert(verts.end(), {x0, y1, u0, v1, col.r, col.g, col.b, col.a, texIdx});
273 };
274
275 if (w->style.bgColor.a > 0.f) {
276 pushQuad(x, y, 0, 0, x + width, y + height, 1, 1, w->style.bgColor, -1.f);
277 }
278
279 if ((w->type == WidgetType::Label || w->type == WidgetType::Button) && !w->text.empty() && !w->style.font.empty()) {
280 std::string key = w->style.font + "_" + std::to_string(static_cast<int>(w->style.fontSize));
281 auto it = m_fontAtlases.find(key);
282 if (it != m_fontAtlases.end()) {
283 const FontAtlas& a = it->second;
284
285 float cx = x;
286 float cy = y;
287
288 for (char ch : w->text) {
289 if (ch < 32 || ch > 127) continue;
290 const stbtt_bakedchar& cd = a.cdata[ch - 32];
291
292 float px = cx + cd.xoff;
293 float py = cy + cd.yoff;
294
295 float u0 = cd.x0 / float(a.width);
296 float v0 = cd.y0 / float(a.height);
297 float u1 = cd.x1 / float(a.width);
298 float v1 = cd.y1 / float(a.height);
299
300 pushQuad(px, py, u0, v0,
301 px + (cd.x1 - cd.x0), py + (cd.y1 - cd.y0), u1, v1,
302 w->style.color, static_cast<float>(a.texIdx));
303 cx += cd.xadvance;
304 }
305 }
306 }
307
308 if (w->type == WidgetType::Image && !w->image.empty()) {
309 uint32_t idx = m_res->getTextureIndex(GetAssetPath(w->image));
310 if (idx != UINT32_MAX) {
311 pushQuad(x, y, 0, 0, x + width, y + height, 1, 1, {1,1,1,1}, static_cast<float>(idx));
312 }else{
313 log("Failed to find image: %s", w->image.c_str());
314 }
315 }
316
317 for (auto* c : w->children) batch(c, verts);
318}

320void VexUI::render(VkCommandBuffer cmd, VkPipeline pipeline, VkPipelineLayout pipelineLayout, int currentFrame) {
321 if (!m_root) return;
322 layout(m_ctx.currentRenderResolution);
323
324 std::vector<float> verts;
325 verts.reserve(1024 * 9);
326 batch(m_root, verts);
327 uploadVerts(verts);
328
329 if (verts.empty()) return;
330
331 vkCmdBindPipeline(cmd, VK_PIPELINE_BIND_POINT_GRAPHICS, pipeline);
332
333 glm::mat4 ortho = glm::ortho(
334 0.0f, static_cast<float>(m_ctx.swapchainExtent.width),
335 0.0f, static_cast<float>(m_ctx.swapchainExtent.height),
336 -1.0f, 1.0f
337 );
338
339 UIPushConstants uiPC{ ortho };
340
341 vkCmdPushConstants(
342 cmd,
343 pipelineLayout,
344 VK_SHADER_STAGE_VERTEX_BIT,
345 0,
346 sizeof(UIPushConstants),
347 &uiPC);
348
349 VkDeviceSize offset = 0;
350 vkCmdBindVertexBuffers(cmd, 0, 1, &m_vb, &offset);
351
352 const uint32_t verticesPerQuad = 6;
353 const uint32_t floatsPerVertex = 9;
354 const uint32_t floatsPerQuad = verticesPerQuad * floatsPerVertex;
355
356 uint32_t totalVertices = verts.size() / floatsPerVertex;
357 uint32_t currentVertex = 0;
358
359 int currentTexIndex = -1;
360 VkDescriptorSet currentTexSet = VK_NULL_HANDLE;
361
362 while (currentVertex < totalVertices) {
363 uint32_t quadStart = currentVertex * floatsPerVertex;
364 float texIndexFloat = verts[quadStart + 8];
365
366 int texIndex = static_cast<int>(texIndexFloat);
367
368 if (texIndex != currentTexIndex) {
369 currentTexIndex = texIndex;
370
371 if (texIndex >= 0) {
372 currentTexSet = m_res->getTextureDescriptorSet(currentFrame, texIndex);
373 } else {
374 currentTexSet = m_res->getTextureDescriptorSet(currentFrame, 0);
375 }
376
377 vkCmdBindDescriptorSets(
378 cmd,
379 VK_PIPELINE_BIND_POINT_GRAPHICS,
380 pipelineLayout,
381 1,
382 1, &currentTexSet,
383 0, nullptr
384 );
385 }
386
387 vkCmdDraw(cmd, 6, 1, currentVertex, 0);
388 currentVertex += 6;
389 }
390}
