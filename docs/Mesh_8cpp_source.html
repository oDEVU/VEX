14// Custom Assimp IO stream for VPK files
15class VPKAssimpStream : public Assimp::IOStream {
16private:
17 std::vector<char> buffer_;
18 size_t position_;
19
20public:
21 VPKAssimpStream(const char* data, size_t size)
22 : buffer_(data, data + size), position_(0) {}
23
24 ~VPKAssimpStream() override = default;
25
26 size_t Read(void* pvBuffer, size_t pSize, size_t pCount) override {
27 size_t bytes_to_read = pSize * pCount;
28 size_t bytes_available = buffer_.size() - position_;
29
30 if (bytes_to_read > bytes_available) {
31 bytes_to_read = bytes_available;
32 }
33
34 if (bytes_to_read > 0) {
35 memcpy(pvBuffer, buffer_.data() + position_, bytes_to_read);
36 position_ += bytes_to_read;
37 }
38
39 return bytes_to_read / pSize; // Return number of items read
40 }
41
42 size_t Write(const void* pvBuffer, size_t pSize, size_t pCount) override {
43 return 0; // Read-only
44 }
45
46 aiReturn Seek(size_t pOffset, aiOrigin pOrigin) override {
47 size_t new_position = position_;
48
49 switch (pOrigin) {
50 case aiOrigin_SET: new_position = pOffset; break;
51 case aiOrigin_CUR: new_position = position_ + pOffset; break;
52 case aiOrigin_END: new_position = buffer_.size() + pOffset; break;
53 default: return aiReturn_FAILURE;
54 }
55
56 if (new_position > buffer_.size()) {
57 return aiReturn_FAILURE;
58 }
59
60 position_ = new_position;
61 return aiReturn_SUCCESS;
62 }
63
64 size_t Tell() const override {
65 return position_;
66 }
67
68 size_t FileSize() const override {
69 return buffer_.size();
70 }
71
72 void Flush() override {}
73};

75// Custom Assimp IO system for VPK files
76class VPKAssimpIOSystem : public Assimp::IOSystem {
77private:
78 VirtualFileSystem* vfs_;
79 std::string base_path_;
80 std::string base_dir_;
81
82public:
83 VPKAssimpIOSystem(VirtualFileSystem* vfs, const std::string& base_path)
84 : vfs_(vfs), base_path_(base_path) {
85 // Extract directory from base path
86 std::filesystem::path path_obj(base_path);
87 base_dir_ = path_obj.parent_path().string();
88 if (!base_dir_.empty() && base_dir_.back() != '/') {
89 base_dir_ += '/';
90 }
91 }
92
93 ~VPKAssimpIOSystem() override = default;
94
95 bool Exists(const char* pFile) const override {
96 std::string file_path(pFile);
97
98 log("Assimp checking if file exists: '%s'", file_path.c_str());
99
100 // First try the path as-is
101 if (vfs_->file_exists(file_path)) {
102 log("File exists as-is: '%s'", file_path.c_str());
103 return true;
104 }
105
106 // Try relative to the base directory
107 std::string relative_path = base_dir_ + file_path;
108 if (vfs_->file_exists(relative_path)) {
109 log("File exists relative to base: '%s'", relative_path.c_str());
110 return true;
111 }
112
113 // Try just the filename (for cases like "scene.bin")
114 std::filesystem::path path_obj(file_path);
115 std::string just_filename = path_obj.filename().string();
116 if (vfs_->file_exists(just_filename)) {
117 log("File exists as filename only: '%s'", just_filename.c_str());
118 return true;
119 }
120
121 log("File not found: '%s'", file_path.c_str());
122 return false;
123 }
124
125 char getOsSeparator() const override {
126 return '/';
127 }
128
129 Assimp::IOStream* Open(const char* pFile, const char* pMode) override {
130 // Only support read mode
131 if (std::strstr(pMode, "r") == nullptr) {
132 return nullptr;
133 }
134
135 std::string file_path(pFile);
136 log("Assimp trying to open: '%s'", file_path.c_str());
137
138 std::string final_path;
139
140 // Try different path resolutions in order:
141
142 // 1. Try as-is
143 if (vfs_->file_exists(file_path)) {
144 final_path = file_path;
145 log("Opening file as-is: '%s'", final_path.c_str());
146 }
147 // 2. Try relative to base directory
148 else if (vfs_->file_exists(base_dir_ + file_path)) {
149 final_path = base_dir_ + file_path;
150 log("Opening file relative to base: '%s'", final_path.c_str());
151 }
152 // 3. Try just the filename
153 else {
154 std::filesystem::path path_obj(file_path);
155 std::string just_filename = path_obj.filename().string();
156 if (vfs_->file_exists(just_filename)) {
157 final_path = just_filename;
158 log("Opening file as filename only: '%s'", final_path.c_str());
159 } else {
160 log("Failed to find file: '%s'", file_path.c_str());
161 return nullptr;
162 }
163 }
164
165 // Read the file data into memory first
166 auto file_data = vfs_->load_file(final_path);
167 if (!file_data) {
168 log("Failed to load file data: '%s'", final_path.c_str());
169 return nullptr;
170 }
171
172 log("Successfully loaded file: '%s' (%zu bytes)", final_path.c_str(), file_data->size);
173
174 // Create stream from memory buffer
175 return new VPKAssimpStream(reinterpret_cast<const char*>(file_data->data.data()), file_data->size);
176 }
177
178 void Close(Assimp::IOStream* pFile) override {
179 delete pFile;
180 }

183 void MeshData::loadFromFile(const std::string& path, VirtualFileSystem* vfs) {

185
186 // Temporary debug code
187 log("Using Assimp version: %d.%d.%d",
188 aiGetVersionMajor(),
189 aiGetVersionMinor(),
190 aiGetVersionRevision());

192 log("Creating assimp importer...");
193 Assimp::Importer importer;
194
195 std::string realPath = path;
196
197 #if NDEBUG
198 // Set up custom IO system if using VFS
199 if (vfs) {
200 importer.SetIOHandler(new VPKAssimpIOSystem(vfs, realPath));
201 }
202 #endif
203
204 if (!vfs->file_exists(realPath)){
205 throw_error("File: [" + realPath + "] doesnt exists");
206 }
207
208 const aiScene* scene = nullptr;
209
210
211
212#if NDEBUG
213 auto fileData = vfs->load_file(realPath);
214 if (!fileData) {
215 throw_error("Failed to load file from VFS: " + realPath);
216 }
217
218 log("Loading from VFS memory buffer, size: %zu", fileData->size);
219
220 // Get file extension for format hint
221 std::string extension = std::filesystem::path(realPath).extension().string();
222 const char* formatHint = extension.c_str();
223
224 // Use ReadFileFromMemory instead of ReadFile
225 scene = importer.ReadFileFromMemory(
226 fileData->data.data(),
227 fileData->size,
228 aiProcess_Triangulate | aiProcess_GenNormals | aiProcess_FlipUVs,
229 formatHint);
230#else
231 scene = importer.ReadFile(realPath,
232 aiProcess_Triangulate |
233 aiProcess_GenNormals |
234 aiProcess_FlipUVs);
235
236#endif
237 if (!scene) {
238 throw_error("Assimp failed to load file: " + std::string(importer.GetErrorString()));
239 }
240
241 if (scene->mFlags & AI_SCENE_FLAGS_INCOMPLETE) {
242 throw_error("Assimp scene incomplete: " + std::string(importer.GetErrorString()));
243 }
244 if (!scene->mRootNode) {
245 throw_error("Assimp scene has no root node");
246 }
247 if (!scene->mMeshes) {
248 throw_error("Assimp scene has invalid mesh array");
249 }
250 if (scene->mNumMeshes == 0) {
251 throw_error("Model contains no meshes");
252 }
253
254 submeshes.clear();
255 submeshes.resize(scene->mNumMeshes);
256 for (unsigned m = 0; m < scene->mNumMeshes; m++) {
257 log("Processing mesh %i...", m);
258 aiMesh* aiMesh = scene->mMeshes[m];
259 Submesh& submesh = submeshes[m];
260
261 // Vertices
262 log("Loading verticies...");
263 submesh.vertices.resize(aiMesh->mNumVertices);
264 for (unsigned i = 0; i < aiMesh->mNumVertices; i++) {
265 submesh.vertices[i].position = {
266 aiMesh->mVertices[i].x,
267 aiMesh->mVertices[i].y,
268 aiMesh->mVertices[i].z
269 };
270
271 submesh.vertices[i].normal = {
272 aiMesh->mNormals[i].x,
273 aiMesh->mNormals[i].y,
274 aiMesh->mNormals[i].z
275 };
276
277 if (aiMesh->mTextureCoords[0]) {
278 submesh.vertices[i].uv = {
279 aiMesh->mTextureCoords[0][i].x,
280 aiMesh->mTextureCoords[0][i].y
281 };
282 } else {
283 submesh.vertices[i].uv = glm::vec2(-100000.f);
284 }
285 }
286
287 // Indices
288 log("Loading indices...");
289 submesh.indices.resize(aiMesh->mNumFaces * 3);
290 for (unsigned i = 0; i < aiMesh->mNumFaces; i++) {
291 aiFace face = aiMesh->mFaces[i];
292 for (unsigned j = 0; j < face.mNumIndices; j++) {
293 submesh.indices[i*3 + j] = face.mIndices[j];
294 }
295 }
296
297 // Textures
298 log("Getting texture path...");
299 std::filesystem::path meshPath(realPath);
300 meshPath.remove_filename();
301
302 if (aiMesh->mMaterialIndex >= 0) {
303 aiMaterial* material = scene->mMaterials[aiMesh->mMaterialIndex];
304 aiString texPath;
305 if (material->GetTexture(aiTextureType_DIFFUSE, 0, &texPath) == AI_SUCCESS) {
306 submesh.texturePath = meshPath.string() + texPath.C_Str();
307 }
308 }
309 }
310
311 // Clean up custom IO system
312 if (vfs) {
313 importer.SetIOHandler(nullptr);
314 }
315 }
