33 void VulkanMesh::upload(const MeshData& meshData) {
34 log("Uploading mesh with %zu submeshes", meshData.submeshes.size());
35
36 m_submeshBuffers.reserve(meshData.submeshes.size());
37 m_submeshTextures.reserve(meshData.submeshes.size());
38
39 for (const auto& srcSubmesh : meshData.submeshes) {
40 SubmeshBuffers buffers{};
41
42 VkBufferCreateInfo bufferInfo{};
43 bufferInfo.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
44 bufferInfo.size = srcSubmesh.vertices.size() * sizeof(Vertex);
45 bufferInfo.usage = VK_BUFFER_USAGE_VERTEX_BUFFER_BIT;
46
47 VmaAllocationCreateInfo allocInfo{};
48 allocInfo.usage = VMA_MEMORY_USAGE_CPU_TO_GPU;
49 vmaCreateBuffer(m_r_context.allocator, &bufferInfo, &allocInfo,
50 &buffers.vertexBuffer, &buffers.vertexAlloc, nullptr);
51
52 void* data;
53 vmaMapMemory(m_r_context.allocator, buffers.vertexAlloc, &data);
54 memcpy(data, srcSubmesh.vertices.data(), bufferInfo.size);
55 vmaUnmapMemory(m_r_context.allocator, buffers.vertexAlloc);
56
57 bufferInfo.size = srcSubmesh.indices.size() * sizeof(uint32_t);
58 bufferInfo.usage = VK_BUFFER_USAGE_INDEX_BUFFER_BIT;
59 vmaCreateBuffer(m_r_context.allocator, &bufferInfo, &allocInfo,
60 &buffers.indexBuffer, &buffers.indexAlloc, nullptr);
61
62 vmaMapMemory(m_r_context.allocator, buffers.indexAlloc, &data);
63 memcpy(data, srcSubmesh.indices.data(), bufferInfo.size);
64 vmaUnmapMemory(m_r_context.allocator, buffers.indexAlloc);
65
66 buffers.indexCount = static_cast<uint32_t>(srcSubmesh.indices.size());
67
68 m_submeshBuffers.push_back(buffers);
69 m_submeshTextures.push_back(srcSubmesh.texturePath);
70
71 log("Uploaded submesh: %zu vertices, %u indices, texture: '%s'",
72 srcSubmesh.vertices.size(), buffers.indexCount,
73 srcSubmesh.texturePath.c_str());
74 }
75 }
