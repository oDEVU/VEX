33/*
34 This struct contains header information for a set of pubnames.
35 Essentially, they contain the context for a set of pubnames
36 belonging to a compilation-unit.
37
38 Also used for the sgi-specific
39 weaknames, typenames, varnames, funcnames data:
40 the structs for those are incomplete and
41 instances of this are used instead.
42
43 Also used for DWARF3,4 .debug_pubtypes and DWARF5 .debug_names.
44
45 These never refer to .debug_types, only to .debug_info.
46*/
47struct Dwarf_Global_Context_s {
48 unsigned pu_is_dnames; /* 0 if not .debug_names. */
49 unsigned pu_alloc_type; /* DW_DLA something */
50 Dwarf_Debug pu_dbg;
51
52 /* One of DW_GL_GLOBAL,DW_GL_PUBTYPES,etc */
53 int pu_global_category;
54
55 /* The following set with the DW_TAG of the DIE
56 involved. Left 0 with .debug_pubnames. */
57 Dwarf_Bool pu_is_debug_names;
58
59 /* For DWARF5 .debug_names the following
60 are irellevant, left 0. */
61 /* For this context, size of a length (offset). 4 or 8 */
62 unsigned char pu_length_size;
63
64 /* Size of the data section for the CU */
65 unsigned char pu_length;
66
67 /* For this CU, size of the extension. 0 except
68 for dwarf2 extension (IRIX) 64bit, in which case is 4. */
69 unsigned char pu_extension_size;
70
71 Dwarf_Half pu_version; /* 2,3,4 or 5 */
72
73 /* offset in pubnames or debug_names of the pu header. */
74 Dwarf_Off pu_pub_offset;
75 /* One past end of the section entries for this CU. */
76 Dwarf_Byte_Ptr pu_pub_entries_end_ptr;
77
78 /* Offset into .debug_info of the compilation-unit header
79 (not DIE) for this set of pubnames. */
80 Dwarf_Off pu_offset_of_cu_header;
81
82 /* Size of compilation-unit that these pubnames are in. */
83 Dwarf_Unsigned pu_info_length;
84
85};

87/* This struct contains information for a single pubname. */
88struct Dwarf_Global_s {
89
90 /* Offset from the start of the corresponding compilation-unit of
91 the DIE for the given pubname CU. */
92 Dwarf_Off gl_named_die_offset_within_cu;
93
94 /* Points to the given pubname. */
95 Dwarf_Small *gl_name;
96
97 /* Context for this pubname. */
98 Dwarf_Global_Context gl_context;
99
100 Dwarf_Half gl_alloc_type; /* DW_DLA something */
101 Dwarf_Half gl_tag; /* .debug_names only. Else 0. */
102};
