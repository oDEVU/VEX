52/* Used to add the specific information for a debug related section
53 Called on each section of interest by section name.
54 DWARF_MAX_DEBUG_SECTIONS must be large enough to allow
55 that all sections of interest fit in the table.
56 returns DW_DLV_ERROR or DW_DLV_OK.
57 */
58static int
59add_debug_section_info(Dwarf_Debug dbg,
60 /* Name as seen in object file. */
61 const char *name,
62 const char *standard_section_name,
63 Dwarf_Unsigned obj_sec_num,
64 struct Dwarf_Section_s *secdata,
65 unsigned groupnum,
66 /* The have_dwarf flag is a somewhat imprecise
67 way to determine if there is at least one 'meaningful'
68 DWARF information section present in the object file.
69 If not set on some section we claim (later) that there
70 is no DWARF info present. see 'foundDwarf' in this file */
71 int duperr,int emptyerr,int have_dwarf,
72 int havezdebug,
73 int *err)
74{
75 unsigned total_entries = dbg->de_debug_sections_total_entries;
76 if (secdata->dss_is_in_use) {
77 *err = duperr;
78 return DW_DLV_ERROR;
79 }
80 if (total_entries < DWARF_MAX_DEBUG_SECTIONS) {
81 struct Dwarf_dbg_sect_s *debug_section =
82 &dbg->de_debug_sections[total_entries];
83 secdata->dss_is_in_use = TRUE;
84 debug_section->ds_name = name;
85 debug_section->ds_number = obj_sec_num;
86 debug_section->ds_secdata = secdata;
87 debug_section->ds_groupnumber = groupnum;
88 secdata->dss_name = name; /* Actual name from object file. */
89 secdata->dss_standard_name = standard_section_name;
90 secdata->dss_number = obj_sec_num;
91 secdata->dss_zdebug_requires_decompress =
92 (Dwarf_Small)havezdebug;
93 secdata->dss_computed_mmap_offset = 0;
94 secdata->dss_computed_mmap_len = 0;
95 secdata->dss_mmap_realarea = 0;
96 secdata->dss_was_alloc = FALSE;
97 /* Just gets current global pref */
98#ifndef TESTING
99 secdata->dss_load_preference = dwarf_set_load_preference(0);
100#endif /* TESTING*/
101 /* We don't yet know about SHF_COMPRESSED */
102 debug_section->ds_duperr = duperr;
103 debug_section->ds_emptyerr = emptyerr;
104 debug_section->ds_have_dwarf = have_dwarf;
105 debug_section->ds_have_zdebug = havezdebug;
106 ++dbg->de_debug_sections_total_entries;
107 return DW_DLV_OK;
108 }
109 /* This represents a bug in libdwarf.
110 Mis-setup-DWARF_MAX_DEBUG_SECTIONS.
111 Or possibly a use of section groups that is
112 not supported. */
113 *err = DW_DLE_TOO_MANY_DEBUG;
114 return DW_DLV_ERROR;
115}

117/* Avoid adding offest to null s2.
118 This function avoids a compiler warning:
119 error: 'strcmp' reading 1 or more bytes
120 from a region of size 0
121 Offset is a fixed small positive number. */
122static int
123both_strings_nonempty(const char *s1, const char *s2, int offset)
124{
125 const char *s3 = 0;
126 if (!s1 || !s2) {
127 return FALSE;
128 }
129 s3 = s2 + offset;
130 if (!s1[0] || !s3[0]) {
131 return FALSE;
132 }
133 return TRUE;
134}

136/* Return DW_DLV_OK etc.
137 PRECONDITION: secname and targname are non-null
138 pointers to strings. */
139static int
140set_up_section(Dwarf_Debug dbg,
141 /* Section name from object format.
142 Might start with .zdebug not .debug if compressed section. */
143 const char *secname,
144 /* Standard section name, such as .debug_info */
145 const char *sec_standard_name,
146 /* Section number from object format */
147 Dwarf_Unsigned obj_sec_num,
148 /* The name associated with this secdata in libdwarf */
149 const char *targname,
150 /* DW_GROUPNUMBER_ANY or BASE or DWO or some other group num */
151 unsigned groupnum_of_sec,
152 struct Dwarf_Section_s *secdata,
153 int duperr,int emptyerr,int have_dwarf,
154 int *err)
155{
156 /* Here accommodate the .debug or .zdebug version, (and of
157 course non- .debug too, but those never zlib) .
158 SECNAMEMAX should be a little bigger than any section
159 name we care about as possibly compressed, which
160 is to say bigger than any standard section name. */
161#define SECNAMEMAX 30
162 size_t secnamelen = strlen(secname);
163 /* static const char *dprefix = ".debug_"; */
164#define DPREFIXLEN 7
165 static const char *zprefix = ".zdebug_";
166#define ZPREFIXLEN 8
167 int havezdebug = FALSE;
168 int namesmatch = FALSE;
169 const char *postzprefix = 0;
170
171 /* For example, if the secname is .zdebug_info
172 we update the finaltargname to .debug_info
173 to match with the particular (known, predefined)
174 object section name.
175 We add one character, so check
176 to see if it will, in the end, fit.
177 See the SET_UP_SECTION macro. */
178
179 if (secnamelen >= SECNAMEMAX) {
180 /* This is not the target section.
181 our caller will keep looking. */
182 return DW_DLV_NO_ENTRY;
183 }
184 havezdebug = !strncmp(secname,zprefix,ZPREFIXLEN);
185 if (havezdebug) {
186 postzprefix = secname+ZPREFIXLEN;
187 }
188 /* With Alpine gcc
189 12.2.1_git20220924-r4) 12.2.1 20220924
190 and some other gcc versions when compiling
191 with -Werror and -fsanitize:
192 we get
193 error: 'strcmp' reading 1 or more bytes
194 from a region of size 0 [-Werror=stringop-overread]
195 So we add -Wnostringop-overread to the build as the error is
196 a false positive. We had to drop stringop-overread
197 references in compiler options, such turned off
198 valuable warnings. Oct 2024
199 refined the test to notice empty string */
200 if (both_strings_nonempty(postzprefix,targname,DPREFIXLEN) &&
201 !strcmp(postzprefix,targname+DPREFIXLEN)) {
202 /* zprefix version matches the object section
203 name so the section is compressed and is
204 the section this targname applies to. */
205 namesmatch = TRUE;
206 } else if (!strcmp(secname,targname)) {
207 namesmatch = TRUE;
208 } else { /* Fall to next statement */ }
209#undef ZPREFIXLEN
210#undef DPREFIXLEN
211#undef SECNAMEMAX
212 if (!namesmatch) {
213 /* This is not the target section.
214 our caller will keep looking. */
215 return DW_DLV_NO_ENTRY;
216 }
217
218 /* SETUP_SECTION. See also BUILDING_SECTIONS, BUILDING_MAP */
219 {
220 /* The section name is a match with targname, or
221 the .zdebug version of targname. */
222 int sectionerr = 0;
223
224 sectionerr = add_debug_section_info(dbg,secname,
225 sec_standard_name,
226 obj_sec_num,
227 secdata,
228 groupnum_of_sec,
229 duperr,emptyerr, have_dwarf,
230 havezdebug,err);
231 if (sectionerr != DW_DLV_OK) {
232 /* *err is set already */
233 return sectionerr;
234 }
235 }
236 return DW_DLV_OK;
237}

239#define SET_UP_SECTION(mdbg,mname,mtarg,mgrp,minfo,me1,me2,mdw,mer) \
240 { \
241 int lerr = 0; \
242 lerr = set_up_section((mdbg), \
243 (mname), /* actual section name */ \
244 (mtarg), /* std section name */ \
245 /* scn_number from macro use context */ \
246 scn_number,(mtarg),(mgrp), \
247 (minfo), \
248 (me1),(me2),(mdw),(mer)); \
249 if (lerr != DW_DLV_NO_ENTRY) { \
250 return lerr; \
251 } /* else fall through. */ \
252 }
