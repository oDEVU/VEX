18 * - Only GNUC
19 * - Only ELF
20 * - Only x86-64, i386, aarch64 and risc-v.
21 * Also, explicitly disable on platforms known not to work so they aren't
22 * forgotten in the future.
23 */

25 defined(__GNUC__) && defined(__ELF__) && \
26 (defined(__x86_64__) || defined(_M_X64) || defined(__i386__) || \
27 defined(_M_IX86) || defined(__aarch64__) || defined(__riscv)) && \
28 !defined(__APPLE__) && !defined(_WIN32) && !defined(__MINGW32__) && \
29 !defined(__CYGWIN__) && !defined(_AIX)
30# define ZSTD_HAVE_WEAK_SYMBOLS 1

32# define ZSTD_HAVE_WEAK_SYMBOLS 0
