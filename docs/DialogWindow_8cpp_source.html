27 void DialogWindow::render() {
28 auto cameraEntity = getCamera();
29 if (cameraEntity == entt::null){
30 log("No camera found. Skipping frame.");
31 return;
32 }
33
34 if(getResolutionMode() != ResolutionMode::NATIVE){
35 setResolutionMode(ResolutionMode::NATIVE);
36 }
37
38 if(getInputMode() != InputMode::UI){
39 setInputMode(InputMode::UI);
40 }
41
42 try {
43 SceneRenderData renderData{};
44 if (!m_interface->getRenderer().beginFrame(m_resolutionManager->getWindowResolution(), renderData)) {
45 log("Skipping frame.");
46 return;
47 }
48 renderData.imguiTextureID = m_interface->getRenderer().getImGuiTextureID(*m_imgui);
49 m_interface->getRenderer().renderScene(renderData, cameraEntity, m_registry, m_frame);
50 m_imgui->beginFrame();
51 m_imgui->executeUIFunctions();
52
53 glm::uvec2 newRes = m_resolutionManager->getWindowResolution();
54 drawDialogWindowLayout(renderData, newRes);
55
56 m_imgui->endFrame();
57 m_interface->getRenderer().composeFrame(renderData, *m_imgui, true);
58 m_interface->getRenderer().endFrame(renderData);
59
60 } catch (const std::exception& e) {
61 SDL_LogError(SDL_LOG_CATEGORY_ERROR, "Dialogwindow render failed");
62 handle_exception(e);
63 }
