118void EditorMenuBar::OpenProjectSelector() {
119 std::filesystem::path binDir = vex::GetExecutableDir();
120 std::filesystem::path selectorPath = binDir / "VexProjectSelector";
121
122 #ifdef _WIN32
123 selectorPath += ".exe";
124 #endif
125
126 if (!std::filesystem::exists(selectorPath)) {
127 vex::log("Error: Could not find ProjectSelector at %s", selectorPath.string().c_str());
128 return;
129 }
130
131 std::string pathStr = selectorPath.string();
132
133 #ifdef _WIN32
134 ShellExecuteA(NULL, "open", pathStr.c_str(), NULL, NULL, SW_SHOW);
135 #else
136 pid_t pid = fork();
137
138 if (pid == 0) {
139 if (setsid() < 0) {
140 exit(EXIT_FAILURE);
141 }
142 execl(pathStr.c_str(), pathStr.c_str(), (char*)NULL);
143 exit(EXIT_FAILURE);
144 }
145 #endif
146}

314std::string EditorMenuBar::GetProjectName() {
315 std::filesystem::path projectFile = std::filesystem::path(m_editor.getProjectBinaryPath()) / "VexProject.json";
316
317 std::ifstream file(projectFile);
318 if (file.is_open()) {
319 try {
320 nlohmann::json j;
321 file >> j;
322 if (j.contains("project_name")) {
323 return j["project_name"].get<std::string>();
324 }
325 } catch (...) {
326 vex::log("Error: Failed to parse VexProject.json");
