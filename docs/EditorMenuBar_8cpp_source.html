180void EditorMenuBar::OpenProjectSelector() {
181 std::filesystem::path binDir = vex::GetExecutableDir();
182 std::filesystem::path selectorPath = binDir / "VexProjectSelector";
183
184 #ifdef _WIN32
185 selectorPath += ".exe";
186 #endif
187
188 if (!std::filesystem::exists(selectorPath)) {
189 vex::log("Error: Could not find ProjectSelector at %s", selectorPath.string().c_str());
190 return;
191 }
192
193 std::string pathStr = selectorPath.string();
194
195 #ifdef _WIN32
196 ShellExecuteA(NULL, "open", pathStr.c_str(), NULL, NULL, SW_SHOW);
197 #else
198 pid_t pid = fork();
199
200 if (pid == 0) {
201 if (setsid() < 0) {
202 exit(EXIT_FAILURE);
203 }
204 execl(pathStr.c_str(), pathStr.c_str(), (char*)NULL);
205 exit(EXIT_FAILURE);
206 }
207 #endif
208}

376std::string EditorMenuBar::GetProjectName() {
377 std::filesystem::path projectFile = std::filesystem::path(m_editor.getProjectBinaryPath()) / "VexProject.json";
378
379 std::ifstream file(projectFile);
380 if (file.is_open()) {
381 try {
382 nlohmann::json j;
383 file >> j;
384 if (j.contains("project_name")) {
385 return j["project_name"].get<std::string>();
386 }
387 } catch (...) {
388 vex::log("Error: Failed to parse VexProject.json");
389 }
390 }

394void EditorMenuBar::RunBuild(bool isDebug, bool runAfter) {
395 struct BuildState {
396 std::string logs = "";
397 bool isFinished = false;
398 std::mutex logMutex;
399 bool autoScroll = true;
400 };
401 auto state = std::make_shared<BuildState>();
402
403 std::string projectPath = m_editor.getProjectBinaryPath();
404 std::string configFlag = isDebug ? " -d" : " -r";
405 std::string command;
406
407 #ifdef _WIN32
408 command = "..\\..\\BuildTools\\build\\ProjectBuilder.exe \"" + projectPath + "\"" + configFlag;
409 #else
410 command = "../../BuildTools/build/ProjectBuilder \"" + projectPath + "\"" + configFlag;
411 #endif
412
413 vex::log("Starting Build: %s", command.c_str());
414
415 std::shared_ptr<BasicEditorWindow> buildWindow = std::make_shared<BasicEditorWindow>();
416 std::weak_ptr<BasicEditorWindow> weakWindow = buildWindow;
417
418 buildWindow->Create = [this, weakWindow, state, isDebug](vex::ImGUIWrapper& wrapper) {
419 wrapper.addUIFunction([=]() {
420 auto window = weakWindow.lock(); if (!window || !window->isOpen) return;
421
422 ImGui::SetNextWindowSize(ImVec2(600, 400), ImGuiCond_FirstUseEver);
423
424 std::string title = isDebug ? "Building Debug..." : "Building Release...";
425 if (state->isFinished) title += " (Finished)";
426
427 if (ImGui::Begin(title.c_str(), &window->isOpen)) {
428
429 if(ImGui::Button("Clear")) {
430 std::lock_guard<std::mutex> lock(state->logMutex);
431 state->logs.clear();
432 }
433 ImGui::SameLine();
434 ImGui::Checkbox("Auto-scroll", &state->autoScroll);
435
436 ImGui::Separator();
437
438 ImGui::BeginChild("BuildLogRegion", ImVec2(0, 0), false, ImGuiWindowFlags_HorizontalScrollbar);
439
440 {
441 std::lock_guard<std::mutex> lock(state->logMutex);
442 ImGui::TextUnformatted(state->logs.c_str());
443 }
444
445 if (state->autoScroll && ImGui::GetScrollY() >= ImGui::GetScrollMaxY()) {
446 ImGui::SetScrollHereY(1.0f);
447 }
448
449 ImGui::EndChild();
450 }
451 ImGui::End();
452 });
453 };
454
455 m_Windows.push_back(buildWindow);
456 buildWindow->Create(m_ImGUIWrapper);
457
458 std::thread([command, state, this, isDebug, runAfter]() {
459 executeCommandRealTime(command,
460 [state](const std::string& line) {
461 std::lock_guard<std::mutex> lock(state->logMutex);
462 state->logs += line;
463 }
464 );
465
466 if (runAfter) {
467 std::string projectName = GetProjectName();
468 if (projectName.empty()) {
469 std::lock_guard<std::mutex> lock(state->logMutex);
470 state->logs += "\n[Error] Could not retrieve Project Name from VexProject.json. Cannot run.\n";
471 } else {
472 std::filesystem::path runPath(m_editor.getProjectBinaryPath());
473 runPath = runPath / "Build" / (isDebug ? "Debug" : "Release");
474 runPath = runPath / projectName;
475
476 #ifdef _WIN32
477 runPath += ".exe";
478 #endif
479
480 if (std::filesystem::exists(runPath)) {
481 {
482 std::lock_guard<std::mutex> lock(state->logMutex);
483 state->logs += "\n=== Build Complete. Launching: " + runPath.string() + " ===\n";
484 }
485
486 std::string runCmd = "\"" + runPath.string() + "\"";
487
488 executeCommandRealTime(runCmd,
489 [state](const std::string& line) {
490 std::lock_guard<std::mutex> lock(state->logMutex);
491 state->logs += line;
492 }
493 );
494 } else {
495 std::lock_guard<std::mutex> lock(state->logMutex);
496 state->logs += "\n[Error] Executable not found at: " + runPath.string() + "\n";
497 }
498 }
499 }
500
501 {
502 std::lock_guard<std::mutex> lock(state->logMutex);
503 state->logs += "\n=== Build Finished ===\n";
504 state->isFinished = true;
505 }
506 }).detach();
507}

509void EditorMenuBar::BuildDist() {
510 struct BuildState {
511 std::string logs = "";
512 bool isFinished = false;
513 std::mutex logMutex;
514 bool autoScroll = true;
515 };
516 auto state = std::make_shared<BuildState>();
517
518 std::string projectPath;
519 std::string enginePath;
520
521 try {
522 projectPath = std::filesystem::canonical(m_editor.getProjectBinaryPath()).string();
523 enginePath = std::filesystem::canonical(vex::GetExecutableDir() / ".." / "..").string();
524 } catch (const std::exception& e) {
525 vex::log("Error resolving paths for Dist Build: %s", e.what());
526 return;
527 }
528
529 std::string command;
530
531 #ifdef _WIN32
532 command = "..\\..\\BuildTools\\build\\ProjectBuilder.exe \"" + projectPath + "\" -dist";
533 vex::log("Starting Windows Distribution Build: %s", command.c_str());
534
535 #else
536 std::string image = "vex-builder:latest";
537 std::string containerEnginePath = "/VexEngine";
538 std::string containerProjectPath = "/VexProject";
539
540 std::string shellCmd =
541 "export VEX_CXX_COMPILER=clang++-17 && "
542 "echo '>> Compiling ProjectBuilder (Container)...' && "
543 "cd " + containerEnginePath + "/BuildTools && "
544 "chmod +x rebuild-buildtools-linux.sh && "

546 "echo '>> Starting Project Distribution Build...' && "
547 "./build_dist/ProjectBuilder " + containerProjectPath + " -dist";
548
549 std::stringstream cmdBuilder;
550
551 bool isPodman = (system("which podman > /dev/null 2>&1") == 0);
552 if (isPodman) cmdBuilder << "podman";
553 else cmdBuilder << "docker";
554
555 std::string mountOpts = ":rw,z";
556
557 cmdBuilder << " run --rm ";
558
559 if (isPodman) {
560 cmdBuilder << "--userns=keep-id ";
561 } else {
562 cmdBuilder << "--user $(id -u):$(id -g) ";
563 }
564
565 cmdBuilder << "-v \"" << enginePath << ":" << containerEnginePath << mountOpts << "\" "
566 << "-v \"" << projectPath << ":" << containerProjectPath << mountOpts << "\" "
567 << "-w " << containerEnginePath << " "
568 << image << " "
569 << "/bin/bash -c \"" << shellCmd << "\"";
570
571 command = cmdBuilder.str() + " 2>&1";
572
573 vex::log("Starting Linux (Sniper) Distribution Build...");
574 #endif

621 try {
622 executeCommandRealTime(command,
623 [state](const std::string& line) {
624 {
625 std::lock_guard<std::mutex> lock(state->logMutex);

627 }
628
629 if (line.length() > 1) {
630 vex::log("%s", line.c_str());
631 }
632 }
633 );
634 } catch (const std::exception& e) {
635 std::lock_guard<std::mutex> lock(state->logMutex);
636 state->logs += "\n[CRITICAL ERROR] Execution failed: ";
637 state->logs += e.what();
638 }
639
640 {
641 std::lock_guard<std::mutex> lock(state->logMutex);
642 state->logs += "\n=== Distribution Build Finished ===\n";
643 state->isFinished = true;
644 }
645 }).detach();
646}
