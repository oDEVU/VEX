53size_t ZBUFF_decompressContinue(ZBUFF_DCtx* zbd,
54 void* dst, size_t* dstCapacityPtr,
55 const void* src, size_t* srcSizePtr)
56{
57 ZSTD_outBuffer outBuff;
58 ZSTD_inBuffer inBuff;
59 size_t result;
60 outBuff.dst = dst;
61 outBuff.pos = 0;
62 outBuff.size = *dstCapacityPtr;
63 inBuff.src = src;
64 inBuff.pos = 0;
65 inBuff.size = *srcSizePtr;
66 result = ZSTD_decompressStream(zbd, &outBuff, &inBuff);
67 *dstCapacityPtr = outBuff.pos;
68 *srcSizePtr = inBuff.pos;
69 return result;
70}
