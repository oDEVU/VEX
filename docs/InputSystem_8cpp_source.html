34 void InputSystem::processEvent(const SDL_Event& event, float deltaTime) {
35 if (m_inputMode == InputMode::UI && event.type != SDL_EVENT_KEY_DOWN && event.type != SDL_EVENT_KEY_UP) {
36 return;
37 }
38
39 if (event.type == SDL_EVENT_KEY_DOWN || event.type == SDL_EVENT_KEY_UP) {
40 SDL_Scancode scancode = event.key.scancode;
41 bool isPressed = (event.type == SDL_EVENT_KEY_DOWN);
42
43 auto& keyState = keyStates[scancode];
44 bool wasPressed = keyState.isPressed;
45
46 keyState.isPressed = isPressed;
47 if (isPressed && !wasPressed) {
48 keyState.wasProcessedAsPressed = false;
49 }
50
51 auto view = m_registry.view<InputComponent>();
52 for (auto entity : view) {
53 auto& inputComp = view.get<InputComponent>(entity);
54 for (const auto& binding : inputComp.bindings) {
55 if (binding.scancode == scancode) {
56 if (isPressed && binding.state == InputActionState::Pressed && !wasPressed && !keyState.wasProcessedAsPressed) {
57 binding.action(deltaTime);
58 keyState.wasProcessedAsPressed = true;
59 } else if (!isPressed && binding.state == InputActionState::Released) {
60 binding.action(deltaTime);
61 } else if (!isPressed && binding.state == InputActionState::Held) {
62 binding.action(0.0f);
63 }
64 }
65 }
66 }
67 } else if (event.type == SDL_EVENT_MOUSE_MOTION && m_inputMode != InputMode::UI) {
68 auto view = m_registry.view<InputComponent>();
69 for (auto entity : view) {
70 auto& inputComp = view.get<InputComponent>(entity);
71 for (const auto& binding : inputComp.mouseAxisBindings) {
72 float delta = (binding.axis == MouseAxis::X) ? event.motion.xrel : event.motion.yrel;
73 binding.action(delta);
74 }
75 }
76 } else if (event.type == SDL_EVENT_GAMEPAD_BUTTON_DOWN || event.type == SDL_EVENT_GAMEPAD_BUTTON_UP) {
77 bool isPressed = (event.type == SDL_EVENT_GAMEPAD_BUTTON_DOWN);
78 SDL_GamepadButton button = (SDL_GamepadButton)event.gbutton.button;
79
80 gamepadButtonStates[button].isPressed = isPressed;
81
82 auto view = m_registry.view<InputComponent>();
83 for (auto entity : view) {
84 auto& inputComp = view.get<InputComponent>(entity);
85 for (const auto& binding : inputComp.gamepadButtonBindings) {
86 if (binding.button == button) {
87 if (isPressed && binding.state == InputActionState::Pressed) {
88 binding.action(deltaTime);
89 } else if (!isPressed && binding.state == InputActionState::Released) {
90 binding.action(deltaTime);
91 }
92 }
93 }
94 }
95 } else if (event.type == SDL_EVENT_GAMEPAD_AXIS_MOTION) {
96 float normalizedValue = event.gaxis.value / 32767.0f;
97
98 auto view = m_registry.view<InputComponent>();
99 for (auto entity : view) {
100 auto& inputComp = view.get<InputComponent>(entity);
101 for (const auto& binding : inputComp.gamepadAxisBindings) {
102 if (binding.axis == (SDL_GamepadAxis)event.gaxis.axis) {
103 binding.action(normalizedValue);
104 }
105 }
106 }
107 }
108 }

110 void InputSystem::update(float deltaTime) {
111 auto view = m_registry.view<InputComponent>();
112 for (auto entity : view) {
113 auto& inputComp = view.get<InputComponent>(entity);
114 for (const auto& binding : inputComp.bindings) {
115 if (binding.state == InputActionState::Held && keyStates[binding.scancode].isPressed) {
116 binding.action(deltaTime);
117 }
118 }
119 for (const auto& binding : inputComp.gamepadButtonBindings) {
120 if (binding.state == InputActionState::Held && gamepadButtonStates[binding.button].isPressed) {
121 binding.action(deltaTime);
122 }
123 }
124 }
