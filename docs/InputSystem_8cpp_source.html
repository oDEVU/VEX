24 void InputSystem::processEvent(const SDL_Event& event, float deltaTime) {
25 if (m_inputMode == InputMode::UI && event.type != SDL_EVENT_KEY_DOWN && event.type != SDL_EVENT_KEY_UP) {
26 return;
27 }
28
29 if (event.type == SDL_EVENT_KEY_DOWN || event.type == SDL_EVENT_KEY_UP) {
30 SDL_Scancode scancode = event.key.scancode;
31 bool isPressed = (event.type == SDL_EVENT_KEY_DOWN);
32
33 auto& keyState = keyStates[scancode];
34 bool wasPressed = keyState.isPressed;
35
36 keyState.isPressed = isPressed;
37 if (isPressed && !wasPressed) {
38 keyState.wasProcessedAsPressed = false;
39 }
40
41 auto view = m_registry.view<InputComponent>();
42 for (auto entity : view) {
43 auto& inputComp = view.get<InputComponent>(entity);
44 for (const auto& binding : inputComp.bindings) {
45 if (binding.scancode == scancode) {
46 if (isPressed && binding.state == InputActionState::Pressed && !wasPressed && !keyState.wasProcessedAsPressed) {
47 binding.action(deltaTime);
48 keyState.wasProcessedAsPressed = true;
49 } else if (!isPressed && binding.state == InputActionState::Released) {
50 binding.action(deltaTime);
51 }
52 }
53 }
54 }
55 } else if (event.type == SDL_EVENT_MOUSE_MOTION && m_inputMode != InputMode::UI) {
56 auto view = m_registry.view<InputComponent>();
57 for (auto entity : view) {
58 auto& inputComp = view.get<InputComponent>(entity);
59 for (const auto& binding : inputComp.mouseAxisBindings) {
60 float delta = (binding.axis == MouseAxis::X) ? event.motion.xrel : event.motion.yrel;
61 binding.action(delta);
62 }
63 }
64 }
65 }

67 void InputSystem::update(float deltaTime) {

69 for (auto entity : view) {
70 auto& inputComp = view.get<InputComponent>(entity);
71 for (const auto& binding : inputComp.bindings) {
72 if (binding.state == InputActionState::Held && keyStates[binding.scancode].isPressed) {
73 binding.action(deltaTime);
74 }
