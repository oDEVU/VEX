45/* Returns DW_DLV_OK DW_DLV_NO_ENTRY or DW_DLV_ERROR
46 crc32 used for debuglink crc calculation.
47 Caller passes pointer to an
48 uninitialized array of 4 unsigned char
49 and if this returns DW_DLV_OK that is filled in.
50 The crc is calculated based on reading
51 the entire current open
52 Dwarf_Debug dbg object file and all bytes in
53 the file are read to create the crc.
54
55 In Windows, where unsigned int is 16 bits, this
56 produces different output than on 32bit ints.
57
58 Since lseek() (all systems/versions) returns
59 a signed value, we check for < 0 as error
60 rather than just check for -1. So it is clear
61 to symbolic execution that conversion to
62 unsigned does not lose bits.
63*/
64int
65dwarf_crc32 (Dwarf_Debug dbg,unsigned char *crcbuf,
66 Dwarf_Error *error)
67{
68 /* off_t is signed, defined by POSIX */
69 /* ssize_t is signed, defined in POSIX */
70
71 /* size_t is unsigned, defined in C89. */
72 Dwarf_Unsigned fsize = 0;
73 /* Named with u to remind the reader that this is
74 an unsigned value. */
75 Dwarf_Unsigned readlenu = 10000;
76 Dwarf_Unsigned size_left = 0;
77 const unsigned char *readbuf = 0;
78 unsigned int tcrc = 0;
79 unsigned int init = 0;
80 int fd = -1;
81 Dwarf_Unsigned sz = 0;
82 int res = 0;
83
84 CHECK_DBG(dbg,error,"dwarf_crc32()");
85 if (!crcbuf) {
86 return DW_DLV_NO_ENTRY;
87 }
88 if (!dbg->de_owns_fd) {
89 return DW_DLV_NO_ENTRY;
90 }
91 fd = dbg->de_fd;
92 if (fd < 0) {
93 return DW_DLV_NO_ENTRY;
94 }
95 fd = dbg->de_fd;
96 if (dbg->de_filesize) {
97 fsize = (size_t)dbg->de_filesize;
98 } else {
99 res = _dwarf_seekr(fd,0,SEEK_END,&sz);
100 if (res != DW_DLV_OK) {
101 _dwarf_error_string(dbg,error,DW_DLE_SEEK_ERROR,
102 "DW_DLE_SEEK_ERROR: dwarf_crc32 seek "
103 "to end fails");
104 return DW_DLV_ERROR;
105 }
106 fsize = sz;
107 }
108 if (fsize <= (Dwarf_Unsigned)500) {
109 /* Not a real object file.
110 A random length check. */
111 return DW_DLV_NO_ENTRY;
112 }
113 size_left = fsize;
114 res = _dwarf_seekr(fd,0,SEEK_SET,0);
115 if (res != DW_DLV_OK) {
116 _dwarf_error_string(dbg,error,DW_DLE_SEEK_ERROR,
117 "DW_DLE_SEEK_ERROR: dwarf_crc32 seek "
118 "to start fails");
119 return DW_DLV_ERROR;
120 }
121 readbuf = (unsigned char *)malloc(readlenu);
122 if (!readbuf) {
123 _dwarf_error_string(dbg,error,DW_DLE_ALLOC_FAIL,
124 "DW_DLE_ALLOC_FAIL: dwarf_crc32 read buffer"
125 " alloc fails");
126 return DW_DLV_ERROR;
127 }
128 while (size_left > 0) {
129 if (size_left < readlenu) {
130 readlenu = size_left;
131 }
132 res = _dwarf_readr(fd,(char *)readbuf,readlenu,0);
133 if (res != DW_DLV_OK) {
134 _dwarf_error_string(dbg,error,DW_DLE_READ_ERROR,
135 "DW_DLE_READ_ERROR: dwarf_crc32 read fails ");
136 free((unsigned char*)readbuf);
137 return DW_DLV_ERROR;
138 }
139 /* Call the public API function so it gets tested too. */
140 tcrc = (unsigned int)dwarf_basic_crc32(readbuf,
141 (unsigned long)readlenu,
142 (unsigned long)init);
143 init = tcrc;
144 size_left -= readlenu;
145 }
146 /* endianness issues? */
147 free((unsigned char*)readbuf);
148 memcpy(crcbuf,(void *)&tcrc,4);
149 return DW_DLV_OK;
150}
