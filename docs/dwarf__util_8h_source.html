268/*
269 READ_AREA LENGTH reads the length (the older way
270 of pure 32 or 64 bit
271 or the dwarf v3 64bit-extension way)
272
273 It reads the bits from where rw_src_data_p points to
274 and updates the rw_src_data_p to point past what was just read.
275
276 It updates w_length_size (to the size of an offset, either 4 or 8)
277 and w_exten_size (set 0 unless this frame has the DWARF3
278 and later 64bit
279 extension, in which case w_exten_size is set to 4).
280
281 r_dbg is just the current dbg pointer.
282 w_target is the output length field.
283 r_targtype is the output type. Always Dwarf_Unsigned so far.
284
285*/
286/* This one handles the v3 64bit extension
287 and 32bit (and SGI/MIPS fixed 64 bit via the
288 dwarf_init-set r_dbg->de_length_size)..
289 It does not recognize any but the one distinguished value
290 (the only one with defined meaning).
291 It assumes that no CU will have a length
292 0xffffffxx (32bit length)
293 or
294 0xffffffxx xxxxxxxx (64bit length)
295 which makes possible auto-detection of the extension.
296
297 This depends on knowing that only a non-zero length
298 is legitimate (AFAICT), and for IRIX non-standard -64
299 dwarf that the first 32 bits of the 64bit offset will be
300 zero (because the compiler could not handle a truly large
301 value as of Jan 2003 and because no app has that much debug
302 info anyway, at least not in the IRIX case).
303
304 At present not testing for '64bit elf' here as that
305 does not seem necessary (none of the 64bit length seems
306 appropriate unless it's ident[EI_CLASS] == ELFCLASS64).
307*/
308/* The w_target > r_sectionlen compare is done without adding in case
309 the w_target value read is so large any addition would overflow.
310 A basic value sanity check. */
311#define READ_AREA_LENGTH_CK(r_dbg,w_target,r_targtype, \
312 rw_src_data_p,w_length_size,w_exten_size,w_error, \
313 r_sectionlen,r_endptr) \
314 do { \
315 READ_UNALIGNED_CK(r_dbg,w_target,r_targtype, \
316 rw_src_data_p, ORIGINAL_DWARF_OFFSET_SIZE, \
317 w_error,r_endptr); \
318 if ((w_target) == DISTINGUISHED_VALUE) { \
319 /* dwarf3 64bit extension */ \
320 (w_length_size) = DISTINGUISHED_VALUE_OFFSET_SIZE; \
321 (rw_src_data_p) += ORIGINAL_DWARF_OFFSET_SIZE; \
322 (w_exten_size) = ORIGINAL_DWARF_OFFSET_SIZE; \
323 READ_UNALIGNED_CK(r_dbg,w_target,r_targtype, \
324 rw_src_data_p, DISTINGUISHED_VALUE_OFFSET_SIZE,\
325 w_error,r_endptr); \
326 if ((w_target) > (r_sectionlen)) { \
327 _dwarf_create_area_len_error((r_dbg),(w_error),\
328 (w_target),(r_sectionlen)); \
329 return DW_DLV_ERROR; \
330 } \
331 (rw_src_data_p) += DISTINGUISHED_VALUE_OFFSET_SIZE; \
332 } else { \
333 if (!(w_target) && (r_dbg)->de_big_endian_object) {\
334 /* Might be IRIX: We have to distinguish between */\
335 /* 32-bit DWARF format and IRIX 64-bit \
336 DWARF format. */ \
337 if ((r_dbg)->de_length_size == 8) { \
338 /* IRIX 64 bit, big endian. This test */ \
339 /* is not a truly precise test, a precise test*/ \
340 /* would check if the target was IRIX. */ \
341 READ_UNALIGNED_CK(r_dbg, w_target , \
342 r_targtype, \
343 rw_src_data_p, \
344 DISTINGUISHED_VALUE_OFFSET_SIZE, \
345 (w_error),(r_endptr)); \
346 if ((w_target) > (r_sectionlen)) { \
347 _dwarf_create_area_len_error((r_dbg), \
348 (w_error), \
349 (w_target),(r_sectionlen)); \
350 return DW_DLV_ERROR; \
351 } \
352 (w_length_size) = \
353 DISTINGUISHED_VALUE_OFFSET_SIZE; \
354 (rw_src_data_p) += \
355 DISTINGUISHED_VALUE_OFFSET_SIZE; \
356 (w_exten_size) = 0; \
357 } else { \
358 /* 32 bit, big endian */ \
359 (w_length_size) = ORIGINAL_DWARF_OFFSET_SIZE;\
360 (rw_src_data_p) += (w_length_size); \
361 (w_exten_size) = 0; \
362 } \
363 } else { \
364 if ((w_target) > (r_sectionlen)) { \
365 _dwarf_create_area_len_error((r_dbg), \
366 (w_error), \
367 (w_target),(r_sectionlen)); \
368 return DW_DLV_ERROR; \
369 } \
370 /* Standard 32 bit dwarf2/dwarf3 */ \
371 (w_exten_size) = 0; \
372 (w_length_size) = ORIGINAL_DWARF_OFFSET_SIZE; \
373 (rw_src_data_p) += (w_length_size); \
374 } \
375 } \
376 } while (0)

393/*
394 Dwarf_Hash_Table_s is the base for the 'hash' table.
395 The table occurs exactly once per CU.
396
397 The intent is that once the total_abbrev_count across
398 one should build a new Dwarf_Hash_Table_Base_s, rehash
399 all the existing entries, and delete the old table and entries.
400 One (500MB) application had
401 127000 abbreviations in one compilation unit but
402 that is clearly an outlier, so this goes for
403 good performance for more normal compilation units.
404 The incoming 'code' is an abbrev number and those simply
405 increase linearly so the hashing is perfect always.
406 As a result we have
407 tb_highest_used_entry to tell us the highest
408 hash value seen, shorting some operations, like
409 dealloc.
410*/
411struct Dwarf_Hash_Table_s {
412 unsigned long tb_table_entry_count;
413 unsigned long tb_total_abbrev_count;
414 unsigned long tb_highest_used_entry;
415 /* Each table entry is a pointer to
416 a list of abbrev-codes and their details.
417 Each Dwarf_Abbrev_List pointer in the array here,
418 and in each singly-linked list starting
419 there points to the entries for one abbrev code. */
420 Dwarf_Abbrev_List *tb_entries;
421};
