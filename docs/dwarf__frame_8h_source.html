30/* The dwarf 2.0 standard dictates that only the following
31 fields can be read when an unexpected augmentation string
32 (in the cie) is encountered: CIE length, CIE_id, version and
33 augmentation; FDE: length, CIE pointer, initial location and
34 address range. Unfortunately, with the above restrictions, it
35 is impossible to read the instruction table from a CIE or a FDE
36 when a new augmentation string is encountered.
37 To fix this problem, the following layout is used, if the
38 augmentation string starts with the string "z".
39 CIE FDE
40 length length
41 CIE_id CIE_pointer
42 version initial_location
43 augmentation address_range
44
45 - length_of_augmented_fields (*NEW*)
46 code_alignment_factor Any new fields as necessary
47 data_alignment_factor instruction_table
48 return_address
49 length_of_augmented fields
50 Any new fields as necessary
51 initial_instructions
52
53 The type of all the old data items are the same as what is
54 described in dwarf 2.0 standard. The length_of_augmented_fields
55 is an LEB128 data item that denotes the size (in bytes) of
56 the augmented fields (not including the size of
57 "length_of_augmented_fields" itself).
58
59 Handling of cie augmentation strings is necessarily a heuristic.
60 See dwarf_frame.c for the currently known augmentation strings.
61
62 ---START SGI-ONLY COMMENT:
63 SGI-IRIX versions of cie or fde were intended
64 to use "z1", "z2" as the
65 augmenter strings if required for new augmentation.
66 However, that never happened (as of March 2005).
67
68 The fde's augmented by the string "z" have a new field
69 (signed constant, 4 byte field)
70 called offset_into_exception_tables, following the
71 length_of_augmented field. This field contains an offset
72 into the "_MIPS_eh_region", which describes
73 the IRIX CC exception handling tables.
74 ---END SGI-ONLY COMMENT
75
76 GNU .eh_frame has an augmentation string of z[RLP]* (gcc 3.4)
77 The similarity to IRIX 'z' (and proposed but never
78 implemented IRIX z1, z2 etc) was confusing things.
79 If the section is .eh_frame then 'z' means GNU exception
80 information 'Augmentation Data' not IRIX 'z'.
81 See The Linux Standard Base Core Specification version 3.0
82*/

140/*
141 This struct denotes the rule for a register in a row of
142 the frame table. In other words, it is one element of
143 the table.
144*/
145struct Dwarf_Reg_Rule_s {
146
147 /* Is a flag indicating whether the rule includes the offset
148 field, ie whether the ru_soffset field is valid or not.
149 Applies only if DW_EXPR_OFFSET or DW_EXPR_VAL_OFFSET.
150 It is important, since reg+offset (offset of 0)
151 is different from
152 just 'register' since the former means 'read memory at address
153 given by the sum of register contents plus offset to get the
154 value'. whereas the latter
155 means 'the value is in the register'.
156 */
157 Dwarf_Sbyte ru_is_offset;
158
159 /* This has to do with evaluating register
160 instructions, not with printing frame instruction.
161 DW_EXPR_OFFSET 0 ( DWARF2)
162 DW_EXPR_VAL_OFFSET 1 (dwarf2/3)
163 DW_EXPR_EXPRESSION 2 (dwarf2/3)
164 DW_EXPR_VAL_EXPRESSION 3 (dwarf2/3)
165 DW_EXPR_ARGS_SIZE 4 (GNU)
166 */
167 Dwarf_Sbyte ru_value_type;
168
169 /* Register involved in this rule, real or non-real-register.
170 ru_value_type is DW_EXPR_OFFSET or DW_EXPR_VAL_OFFSET.
171 */
172 Dwarf_Unsigned ru_register;
173
174 /* Offset to add to register, if indicated by ru_is_offset.
175 ru_value_type is DW_EXPR_OFFSET */
176 Dwarf_Signed ru_offset;
177 Dwarf_Unsigned ru_args_size; /* DW_CFA_GNU_args_size */
178 /* If ru_value_type is DW_EXPR_EXPRESSION
179 or DW_EXPR_VAL_EXPRESSION this is filled in. */
180 Dwarf_Block ru_block;
181
182};

225/* See dwarf_frame.c for the heuristics used to set the
226 Dwarf_Cie ci_augmentation_type.
227
228 This succinctly helps interpret the size and
229 meaning of .debug_frame
230 and (for gcc) .eh_frame.
231
232 In the case of gcc .eh_frame (gcc 3.3, 3.4)
233 z may be followed by one or more of
234 L R P.
235
236*/
237enum Dwarf_augmentation_type {
238 aug_empty_string, /* Default empty augmentation string. */
239 aug_irix_exception_table, /* IRIX plain "z",
240 for exception handling, IRIX CC compiler.
241 Proposed z1 z2 ... never implemented. */
242 aug_gcc_eh_z, /* gcc z augmentation, (including
243 L R P variations). gcc 3.3 3.4 exception
244 handling in eh_frame. */
245 aug_irix_mti_v1, /* IRIX "mti v1" augmentation string. Probably
246 never in any released SGI-IRIX compiler. */
247 aug_eh, /* For gcc .eh_frame, "eh" is the string.,
248 gcc 1,2, egcs. Older values. */
249 aug_armcc, /* "armcc+" meaning the cfa calculation
250 is corrected to be standard (output by
251 Arm C RVCT 3.0 SP1 and later). See
252 http://sourceware.org/ml/gdb-patches/2006-12/msg00249.html
253 for details. */
254 aug_unknown, /* Unknown augmentation, we cannot do much. */
255
256 /* HC, From http://sourceforge.net/p/elftoolchain/tickets/397/ */
257 aug_metaware,
258
259 aug_past_last
