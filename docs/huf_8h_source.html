55/* static allocation of HUF's Compression Table */
56/* this is a private definition, just exposed for allocation and strict aliasing purpose. never EVER access its members directly */
57typedef size_t HUF_CElt; /* consider it an incomplete type */
58#define HUF_CTABLE_SIZE_ST(maxSymbolValue) ((maxSymbolValue)+2) /* Use tables of size_t, for proper alignment */
59#define HUF_CTABLE_SIZE(maxSymbolValue) (HUF_CTABLE_SIZE_ST(maxSymbolValue) * sizeof(size_t))
60#define HUF_CREATE_STATIC_CTABLE(name, maxSymbolValue) \
61 HUF_CElt name[HUF_CTABLE_SIZE_ST(maxSymbolValue)] /* no final ; */
62
63/* static allocation of HUF's DTable */
64typedef U32 HUF_DTable;
65#define HUF_DTABLE_SIZE(maxTableLog) (1 + (1<<(maxTableLog)))
66#define HUF_CREATE_STATIC_DTABLEX1(DTable, maxTableLog) \
67 HUF_DTable DTable[HUF_DTABLE_SIZE((maxTableLog)-1)] = { ((U32)((maxTableLog)-1) * 0x01000001) }
68#define HUF_CREATE_STATIC_DTABLEX2(DTable, maxTableLog) \
69 HUF_DTable DTable[HUF_DTABLE_SIZE(maxTableLog)] = { ((U32)(maxTableLog) * 0x01000001) }
