80 /* Total length of the macro data for this

94 /* the following three set from the bits in mc_flags */

108 the standardops may or may not be
109 defined in the header. */
110 Dwarf_Bool mc_operands_table_flag;
111
112 /* Count of the Dwarf_Macro_Forms_s structs pointed to by
113 mc_opcode_forms. These from the header. */
114 Dwarf_Small mc_opcode_count;
115 struct Dwarf_Macro_Forms_s *mc_opcode_forms;
116
117 /* mc_ops must be free()d, but pointers inside
118 mc_ops are to static or section data so must not
119 be freed. */
120 Dwarf_Unsigned mc_macro_ops_count;
121 Dwarf_Unsigned mc_ops_data_length;

124 Dwarf_Small * mc_macro_header;
125 Dwarf_Small * mc_macro_ops;
126
127 /* These are malloc space, not _dwarf_get_alloc()

130 char ** mc_srcfiles;

132

139 */

142 /* The following is malloc,so macro_context_s destructor

146 Dwarf_Debug mc_dbg;
147 Dwarf_CU_Context mc_cu_context;
