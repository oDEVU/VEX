25std::filesystem::path GetExecutableDir() {
26 std::filesystem::path exePath;
27
28#ifdef _WIN32
29 wchar_t wPath[MAX_PATH];
30 DWORD len = GetModuleFileNameW(NULL, wPath, MAX_PATH);
31 if (len == 0 || len == MAX_PATH) {
32 vex::throw_error("Failed to get executable path on Windows");
33 }
34 int utf8Size = WideCharToMultiByte(CP_UTF8, 0, wPath, -1, nullptr, 0, nullptr, nullptr);
35 if (utf8Size == 0) {
36 vex::throw_error("Failed to convert executable path to UTF-8");
37 }
38 std::string utf8Path(utf8Size, '\0');
39 WideCharToMultiByte(CP_UTF8, 0, wPath, -1, utf8Path.data(), utf8Size, nullptr, nullptr);
40 exePath = std::filesystem::path(utf8Path);
41#elif defined(__APPLE__)
42 char pathBuf[MAXPATHLEN];
43 uint32_t size = sizeof(pathBuf);
44 if (_NSGetExecutablePath(pathBuf, &size) != 0) {
45 vex::throw_error("Failed to get executable path on macOS");
46 }
47 exePath = std::filesystem::canonical(pathBuf);
48#else
49 char pathBuf[1024];
50 ssize_t len = ::readlink("/proc/self/exe", pathBuf, sizeof(pathBuf) - 1);
51 if (len == -1) {
52 vex::throw_error("Failed to read /proc/self/exe on Linux");
53 }
54 pathBuf[len] = '\0';
55 exePath = std::filesystem::path(pathBuf);
56#endif
57
58 return exePath.parent_path();
59}
