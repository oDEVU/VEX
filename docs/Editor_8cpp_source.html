303 float Editor::raySphereIntersect(const glm::vec3& rayOrigin, const glm::vec3& rayDir, const glm::vec3& sphereCenter, float sphereRadius) {
304 glm::vec3 L = sphereCenter - rayOrigin;
305 float tca = glm::dot(L, rayDir);
306 if (tca < 0) return -1.0f;
307 float d2 = glm::dot(L, L) - tca * tca;
308 float radius2 = sphereRadius * sphereRadius;
309 if (d2 > radius2) return -1.0f;
310 float thc = sqrt(radius2 - d2);
311 return tca - thc;
312 }

314 float Editor::rayTriangleIntersect(
315 const glm::vec3& rayOrigin, const glm::vec3& rayDir,
316 const glm::vec3& v0, const glm::vec3& v1, const glm::vec3& v2)
317 {
318 const float EPSILON = 0.0000001f;
319 glm::vec3 edge1 = v1 - v0;
320 glm::vec3 edge2 = v2 - v0;
321 glm::vec3 h = glm::cross(rayDir, edge2);
322 float a = glm::dot(edge1, h);
323
324 if (a > -EPSILON && a < EPSILON) return -1.0f;
325
326 float f = 1.0f / a;
327 glm::vec3 s = rayOrigin - v0;
328 float u = f * glm::dot(s, h);
329 if (u < 0.0f || u > 1.0f) return -1.0f;
330
331 glm::vec3 q = glm::cross(s, edge1);
332 float v = f * glm::dot(rayDir, q);
333 if (v < 0.0f || u + v > 1.0f) return -1.0f;
334
335 float t = f * glm::dot(edge2, q);
336 if (t > EPSILON) return t;
337 else return -1.0f;
338 }
