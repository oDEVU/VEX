273 setInputMode(InputMode::UI);

276 glm::uvec2 viewportRes = (m_viewportSize.x > 0 && m_viewportSize.y > 0)
277 ? m_viewportSize
278 : m_resolutionManager->getRenderResolution();
279
280 try {
281 SceneRenderData renderData{};
282 if (!m_interface->getRenderer().beginFrame(viewportRes, renderData)) {
283 return;

492 float Editor::raySphereIntersect(const glm::vec3& rayOrigin, const glm::vec3& rayDir, const glm::vec3& sphereCenter, float sphereRadius) {
493 glm::vec3 L = sphereCenter - rayOrigin;
494 float tca = glm::dot(L, rayDir);
495 if (tca < 0) return -1.0f;
496 float d2 = glm::dot(L, L) - tca * tca;
497 float radius2 = sphereRadius * sphereRadius;
498 if (d2 > radius2) return -1.0f;
499 float thc = sqrt(radius2 - d2);
500 return tca - thc;
501 }

503 float Editor::rayTriangleIntersect(
504 const glm::vec3& rayOrigin, const glm::vec3& rayDir,
505 const glm::vec3& v0, const glm::vec3& v1, const glm::vec3& v2)
506 {
507 const float EPSILON = 0.0000001f;
508 glm::vec3 edge1 = v1 - v0;
509 glm::vec3 edge2 = v2 - v0;
510 glm::vec3 h = glm::cross(rayDir, edge2);
511 float a = glm::dot(edge1, h);
512
513 if (a > -EPSILON && a < EPSILON) return -1.0f;
514
515 float f = 1.0f / a;
516 glm::vec3 s = rayOrigin - v0;
517 float u = f * glm::dot(s, h);
518 if (u < 0.0f || u > 1.0f) return -1.0f;
519
520 glm::vec3 q = glm::cross(s, edge1);
521 float v = f * glm::dot(rayDir, q);
522 if (v < 0.0f || u + v > 1.0f) return -1.0f;
523
524 float t = f * glm::dot(edge2, q);
525 if (t > EPSILON) return t;
526 else return -1.0f;
527 }

734 const auto& v1 = submesh.vertices[submesh.indices[i+1]].position;

754 hitEntity = entity;
755 }
756 }
