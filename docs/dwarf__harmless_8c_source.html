30/* This implements _dwarf_insert_harmless_error
31 and related helper functions for recording
32 compiler errors that need not make the input
33 unusable.
34
35 Applications can use dwarf_get_harmless_error_list to
36 find (and possibly print) a warning about such errors.
37
38 The initial error reported here is
39 DW_DLE_DEBUG_FRAME_LENGTH_NOT_MULTIPLE which was a
40 bug in a specific compiler.
41
42 It is a fixed length circular list to constrain
43 the space used for errors.
44
45 The assumption is that these errors are exceedingly
46 rare, and indicate a broken compiler (the one that
47 produced the object getting the error(s)).
48
49 dh_maxcount is recorded internally as 1 greater than
50 requested. Hiding the fact we always leave one
51 slot unused (at least). So a user request for
52 N slots really gives the user N usable slots. */
53

78/* The pointers returned here through errmsg_ptrs_array
79 become invalidated by any call to libdwarf. Any call.
80*/
81int
82dwarf_get_harmless_error_list(Dwarf_Debug dbg,
83 unsigned count,
84 const char ** errmsg_ptrs_array,
85 unsigned * errs_count)
86{
87
88 struct Dwarf_Harmless_s *dhp = 0;
89
90 if (IS_INVALID_DBG(dbg)) {
91 return DW_DLV_NO_ENTRY;
92 }
93 dhp = &dbg->de_harmless_errors;
94 if (!dhp->dh_errors) {
95 dhp->dh_errs_count = 0;
96 return DW_DLV_NO_ENTRY;
97 }
98 if (dhp->dh_errs_count == 0) {
99 return DW_DLV_NO_ENTRY;
100 }
101 if (errs_count) {
102 *errs_count = dhp->dh_errs_count;
103 }
104 if (count) {
105 /* NULL terminate the array of pointers */
106 --count;
107 errmsg_ptrs_array[count] = 0;
108
109 if (dhp->dh_next_to_use != dhp->dh_first) {
110 unsigned i = 0;
111 unsigned cur = dhp->dh_first;
112 for (i = 0; cur != dhp->dh_next_to_use;
113 ++i, cur = (cur +1) % dhp->dh_maxcount) {
114 if (i >= count ) {
115 /* All output spaces are used. */
116 break;
117 }
118 errmsg_ptrs_array[i] = dhp->dh_errors[cur];
119 }
120 errmsg_ptrs_array[i] = 0;
121 }
122 }
123 dhp->dh_next_to_use = 0;
124 dhp->dh_first = 0;
125 dhp->dh_errs_count = 0;
126 return DW_DLV_OK;
127}

129/* Insertion made public is only for testing the harmless error code,
130 it is not necessarily useful for libdwarf client code aside
131 from code testing libdwarf. */
132void
133dwarf_insert_harmless_error(Dwarf_Debug dbg,
134 char *newerror)
135{
136 struct Dwarf_Harmless_s *dhp = 0;
137 unsigned next = 0;
138 unsigned cur = 0;
139 char *msgspace = 0;
140
141 if (IS_INVALID_DBG(dbg)) {
142 return;
143 }
144 dhp = &dbg->de_harmless_errors;
145 cur = dhp->dh_next_to_use;
146 if (!dhp->dh_errors) {
147 dhp->dh_errs_count++;
148 return;
149 }
150 msgspace = dhp->dh_errors[cur];
151 _dwarf_safe_strcpy(msgspace,
152 DW_HARMLESS_ERROR_MSG_STRING_SIZE,
153 newerror,
154 strlen(newerror));
155 next = (cur+1) % dhp->dh_maxcount;
156 dhp->dh_errs_count++;
157 dhp->dh_next_to_use = next;
158 if (dhp->dh_next_to_use == dhp->dh_first) {
159 /* Array is full set full invariant. */
160 dhp->dh_first = (dhp->dh_first+1) % dhp->dh_maxcount;
161 }
162}

164/* The size of the circular list of strings may be set
165 and reset as desired. Returns the previous size of
166 the list. If the list is shortened excess error entries
167 are simply dropped.
168 If the reallocation fails the list size is left unchanged.
169 Do not make this a long list!
170
171 Remember the maxcount we record is 1 > the user count,
172 so we adjust it so it looks like the user count.
173*/
174unsigned
175dwarf_set_harmless_error_list_size(Dwarf_Debug dbg,
176 unsigned maxcount )
177{
178 struct Dwarf_Harmless_s *dhp = 0;
179 unsigned prevcount = 0;
180
181 if (IS_INVALID_DBG(dbg)) {
182 return 0;
183 }
184 dhp = &dbg->de_harmless_errors;
185 prevcount = dhp->dh_maxcount;
186 if (maxcount != 0) {
187 ++maxcount;
188 if (maxcount != dhp->dh_maxcount) {
189 /* Assign transfers 'ownership' of the malloc areas
190 to oldarray. */
191 struct Dwarf_Harmless_s oldarray = *dhp;
192 /* Do not double increment the max, the init() func
193 increments it too. */
194 _dwarf_harmless_init(dhp,maxcount-1);
195 if (oldarray.dh_next_to_use != oldarray.dh_first) {
196 unsigned i = 0;
197 for (i = oldarray.dh_first;
198 i != oldarray.dh_next_to_use;
199 i = (i+1)%oldarray.dh_maxcount) {
200 dwarf_insert_harmless_error(dbg,
201 oldarray.dh_errors[i]);
202 }
203 if (oldarray.dh_errs_count > dhp->dh_errs_count) {
204 dhp->dh_errs_count = oldarray.dh_errs_count;
205 }
206 }
207 _dwarf_harmless_cleanout(&oldarray);
208 }
209 }
210 return prevcount-1;
211}

213/* Only callable from within libdwarf (as a practical matter)
214*/
215void
216_dwarf_harmless_init(struct Dwarf_Harmless_s *dhp,unsigned size)
217{
218 unsigned i = 0;
219 memset(dhp,0,sizeof(*dhp));
220 dhp->dh_maxcount = size +1;
221 dhp->dh_errors = (char **)calloc(dhp->dh_maxcount,
222 sizeof(char *));
223 if (!dhp->dh_errors) {
224 dhp->dh_maxcount = 0;
225 return;
226 }
227 for (i = 0; i < dhp->dh_maxcount; ++i) {
228 char *newstr =
229 (char *)calloc(1, DW_HARMLESS_ERROR_MSG_STRING_SIZE);
230 dhp->dh_errors[i] = newstr;
231#if 0 /* Commentary about avoiding leak */
232 /* BAD IDEA. just use the NULL pointer,
233 so we avoid problems later with
234 freeing. */
235 if (!newstr) {
236 dhp->dh_maxcount = 0;
237 /* Let it leak, the leak is a constrained amount. */
238 free(dhp->dh_errors);
239 dhp->dh_errors = 0;
240 return;
241 }
242#endif /* 0 */
243 dhp->dh_errors[i] = newstr;
244 }
245}

247void
248_dwarf_harmless_cleanout(struct Dwarf_Harmless_s *dhp)
249{
250 unsigned i = 0;
251 if (!dhp->dh_errors) {
252 return;
253 }
254 for (i = 0; i < dhp->dh_maxcount; ++i) {
255 free(dhp->dh_errors[i]);
256 dhp->dh_errors[i] = 0;
257 }
258 free(dhp->dh_errors);
259 dhp->dh_errors = 0;
260 dhp->dh_maxcount = 0;
261}
