63static int
64_dwarf_find_CU_Context_given_sig(Dwarf_Debug dbg,
65 int context_level,
66 Dwarf_Sig8 *sig_in,
67 Dwarf_CU_Context *cu_context_out,
68 Dwarf_Bool *is_info_out,
69 Dwarf_Error *error)
70{
71 Dwarf_CU_Context cu_context = 0;
72 Dwarf_Bool is_info = FALSE;
73 int loopcount = 0;
74 int lres = 0;
75 Dwarf_Debug_InfoTypes dis = 0;
76 struct Dwarf_Section_s *secdp = 0;
77
78 /* Loop once with is_info, once with !is_info.
79 Then stop. */
80 for ( ; loopcount < 2; ++loopcount) {
81 Dwarf_CU_Context prev_cu_context = 0;
82 Dwarf_Unsigned section_size = 0;
83 Dwarf_Unsigned new_cu_offset = 0;
84
85 is_info = !is_info;
86
87 if (is_info) {
88 dis = &dbg->de_info_reading;
89 secdp = &dbg->de_debug_info;
90 } else {
91 dis = &dbg->de_types_reading;
92 secdp = &dbg->de_debug_types;
93 }
94 lres = _dwarf_load_die_containing_section(dbg,is_info,error);
95 if (lres == DW_DLV_ERROR) {
96 return lres;
97 }
98 if (lres == DW_DLV_NO_ENTRY ) {
99 continue;
100 }
101 /* Lets see if we already have the CU we need. */
102 for (cu_context = dis->de_cu_context_list;
103 cu_context; cu_context = cu_context->cc_next) {
104 prev_cu_context = cu_context;
105
106 if (memcmp(sig_in,&cu_context->cc_signature,
107 sizeof(Dwarf_Sig8))) {
108 continue;
109 }
110 if (cu_context->cc_unit_type == DW_UT_split_type||
111 cu_context->cc_unit_type == DW_UT_type) {
112 *cu_context_out = cu_context;
113 *is_info_out = cu_context->cc_is_info;
114 return DW_DLV_OK;
115 }
116 }
117 if (context_level > 0) {
118 /* Make no attempt to create new context,
119 we are finishing cu die base fields
120 on one already.
121 Just look for the other context,
122 DWARF4 debug_types */
123 continue;
124 }
125 if (prev_cu_context) {
126 Dwarf_CU_Context lcu_context = prev_cu_context;
127 new_cu_offset =
128 _dwarf_calculate_next_cu_context_offset(
129 lcu_context);
130 } else {
131 new_cu_offset = 0;
132 }
133 section_size = secdp->dss_size;
134 for ( ; new_cu_offset < section_size;
135 new_cu_offset =
136 _dwarf_calculate_next_cu_context_offset(
137 cu_context)) {
138#if 0 /* unnecessary load section call, we think. */
139 lres = _dwarf_load_die_containing_section(dbg,
140 is_info,error);
141 if (lres == DW_DLV_ERROR) {
142 return lres;
143 }
144 if (lres == DW_DLV_NO_ENTRY) {
145 continue;
146 }
147#endif /*0*/
148 lres = _dwarf_create_a_new_cu_context_record_on_list(
149 dbg, dis,is_info,section_size,new_cu_offset,
150 &cu_context,NULL,error);
151 if (lres == DW_DLV_ERROR) {
152 return lres;
153 }
154 if (lres == DW_DLV_NO_ENTRY) {
155 break;
156 }
157 if (memcmp(sig_in,&cu_context->cc_signature,
158 sizeof(Dwarf_Sig8))) {
159 continue;
160 }
161 if (cu_context->cc_unit_type == DW_UT_split_type||
162 cu_context->cc_unit_type == DW_UT_type) {
163 *cu_context_out = cu_context;
164 *is_info_out = cu_context->cc_is_info;
165 return DW_DLV_OK;
166 }
167 }
168 } /* Loop-end. */
169 /* Not found */
170 return DW_DLV_NO_ENTRY;
171}

173/* We will search to find a CU with the indicated signature
174 The attribute leading us here is often
175 We are looking for a DW_UT_split_type or DW_UT_type
176 CU.
177 DW_AT_type and if DWARF4 that means our first look is
178 to !is_info */
179int
180dwarf_find_die_given_sig8(Dwarf_Debug dbg,
181 Dwarf_Sig8 *ref,
182 Dwarf_Die *die_out,
183 Dwarf_Bool *is_info,
184 Dwarf_Error *error)
185{
186 int res = 0;
187 CHECK_DBG(dbg,error,"dwarf_find_die_given_sig8()");
188 res = _dwarf_internal_find_die_given_sig8(
189 dbg,0,ref,die_out,is_info,error);
190 return res;
191}

193/* If context level > 0 restrict what we will do
194 to avoid recursion creating CU Contexts */
195int
196_dwarf_internal_find_die_given_sig8(Dwarf_Debug dbg,
197 int context_level,
198 Dwarf_Sig8 *ref,
199 Dwarf_Die *die_out,
200 Dwarf_Bool *is_info,
201 Dwarf_Error *error)
202{
203 int res = 0;
204 Dwarf_Die ndi = 0;
205 Dwarf_CU_Context context = 0;
206 Dwarf_Bool result_is_info = FALSE;
207 Dwarf_Unsigned dieoffset = 0;
208
209 res =_dwarf_find_CU_Context_given_sig(dbg,
210 context_level,
211 ref, &context, &result_is_info,error);
212 if (res != DW_DLV_OK) {
213 return res;
214 }
215 dieoffset = context->cc_debug_offset +
216 context->cc_signature_offset;
217 res = dwarf_offdie_b(dbg,dieoffset,result_is_info,
218 &ndi,error);
219 if (res == DW_DLV_OK) {
220 *die_out = ndi;
221 *is_info = result_is_info;
222 }
223 return res;
224}
