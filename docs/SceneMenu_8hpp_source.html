26inline std::string Demangle(const char* name) {
27#ifdef _WIN32
28 std::string s = name;
29 const std::string prefix_class = "class ";
30 const std::string prefix_struct = "struct ";
31
32 if (s.rfind(prefix_class, 0) == 0) return s.substr(prefix_class.length());
33 if (s.rfind(prefix_struct, 0) == 0) return s.substr(prefix_struct.length());
34 return s;
35#else
36 int status = -1;
37 std::unique_ptr<char, void(*)(void*)> res {
38 abi::__cxa_demangle(name, NULL, NULL, &status),
39 std::free
40 };
41 return (status == 0) ? res.get() : name;
42#endif
43}
