12void EditorCameraObject::processEvent(const SDL_Event& event, float deltaTime) {

14 bool isPressed = (event.type == SDL_EVENT_KEY_DOWN);
15 const SDL_KeyboardEvent& key_event = event.key;
16 keyStates[key_event.scancode] = isPressed;
17 }
18
19 if (event.type == SDL_EVENT_MOUSE_WHEEL && m_isFlying) {
20 if (event.wheel.y > 0) {
21 m_movementSpeed *= 1.2f;
22 } else if (event.wheel.y < 0) {
23 m_movementSpeed *= 0.8f;
24 }
25 }
26
27 if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN) {
28 if (event.button.button == SDL_BUTTON_RIGHT && m_viewportHovered) {
29 m_isFlying = true;
30 SDL_SetWindowRelativeMouseMode(m_window, true);
31 }
32 }
33
34 if (event.type == SDL_EVENT_MOUSE_BUTTON_UP) {
35 if (event.button.button == SDL_BUTTON_RIGHT) {
36 m_isFlying = false;
37 SDL_SetWindowRelativeMouseMode(m_window, false);
38 }
39 }
40
41 if (event.type == SDL_EVENT_MOUSE_MOTION && m_isFlying) {
42 auto& tc = GetComponent<vex::TransformComponent>();
43
44 float deltaYaw = -event.motion.xrel * m_mouseSensitivity;
45 float deltaPitch = -event.motion.yrel * m_mouseSensitivity;
46

52void EditorCameraObject::Update(float deltaTime) {

54 bool isShiftHeld = keyStates[SDL_SCANCODE_LSHIFT] || keyStates[SDL_SCANCODE_RSHIFT];
55
56 if(isShiftHeld){

59
60 if (keyStates[SDL_SCANCODE_W] && (m_viewportHovered || m_isFlying)) {

62 }
63
64 if (keyStates[SDL_SCANCODE_S] && (m_viewportHovered || m_isFlying)) {

66 }
