244 void ProjectSelector::moveProject(int index, const std::string& newParentDir) {
245 if (index < 0 || index >= m_projects.size()) return;
246
247 auto& proj = m_projects[index];
248 std::filesystem::path oldPath(proj.path);
249 std::filesystem::path newParent(newParentDir);
250 std::filesystem::path newPath = newParent / oldPath.filename();
251
252 if (!std::filesystem::exists(oldPath)) {
253 log(LogLevel::ERROR, "Source path does not exist!");
254 return;
255 }
256 if (std::filesystem::exists(newPath)) {
257 log(LogLevel::ERROR, "Destination already exists: %s", newPath.string().c_str());
258 return;
259 }
260
261 try {
262 if (!std::filesystem::exists(newParent)) std::filesystem::create_directories(newParent);
263
264 std::filesystem::rename(oldPath, newPath);
265 log("Project moved to: %s", newPath.string().c_str());
266
267 proj.path = newPath.string();
268
269 saveKnownProjects();

305 void ProjectSelector::scanAndAddProject(const std::string& pathStr) {
306 std::filesystem::path path(pathStr);
307 std::filesystem::path projectFile = path / "VexProject.json";
308
309 if (std::filesystem::exists(projectFile)) {
310 try {
311 std::ifstream f(projectFile);
312 nlohmann::json j;
313 f >> j;
314
315 std::string pName = j.value("project_name", "Unknown Project");
316 std::string pVer = j.value("version", "1");
317
318 for(auto& p : m_projects) {
319 if(p.path == pathStr) return;
320 }
321
322 m_projects.push_back({pName, pathStr, pVer});
323 saveKnownProjects();
324 log("Added project: %s", pName.c_str());
325
326 } catch(const std::exception& e) {
327 log(LogLevel::ERROR, "Invalid VexProject.json at %s: %s", pathStr.c_str(), e.what());
328 }
329 } else {
330 log(LogLevel::ERROR, "No VexProject.json found at %s", pathStr.c_str());
331 }

381 void ProjectSelector::createNewProject() {
382 std::string name(m_newProjName);
383 std::string parent(m_newProjParentDir);
384
385 if (name.empty() || parent.empty()) return;
386
387 std::filesystem::path exeDir = GetExecutableDir();
388 std::filesystem::path templatePath = exeDir / ".." / "Assets" / "default" / "NewProject";
389
390 if (!std::filesystem::exists(templatePath)) {
391 log(LogLevel::ERROR, "Missing template: %s", templatePath.string().c_str());
392 return;
393 }
394
395 std::filesystem::path destParent(parent);
396 std::filesystem::path newProjectPath = destParent / name;
397
398 if (std::filesystem::exists(newProjectPath)) {
399 log(LogLevel::ERROR, "Project already exists at: %s", newProjectPath.string().c_str());
400 return;
401 }
402
403 try {
404 if (!std::filesystem::exists(destParent)) {
405 std::filesystem::create_directories(destParent);
406 }
407
408 std::filesystem::copy(templatePath, newProjectPath, std::filesystem::copy_options::recursive);
409
410 std::filesystem::path jsonPath = newProjectPath / "VexProject.json";
411 if (std::filesystem::exists(jsonPath)) {
412 std::ifstream i(jsonPath);
413 nlohmann::json j;
414 i >> j;
415
416 j["project_name"] = name;
417
418 std::ofstream o(jsonPath);
419 o << j.dump(4);
420 }
421
422 scanAndAddProject(newProjectPath.string());
423 log("Created project: %s", name.c_str());
424
425 } catch (const std::exception& e) {
426 log(LogLevel::ERROR, "Creation failed: %s", e.what());
427 }
