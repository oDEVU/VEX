161 void ProjectSelector::scanAndAddProject(const std::string& pathStr) {
162 std::filesystem::path path(pathStr);
163 std::filesystem::path projectFile = path / "VexProject.json";
164
165 if (std::filesystem::exists(projectFile)) {
166 try {
167 std::ifstream f(projectFile);
168 nlohmann::json j;
169 f >> j;
170
171 std::string pName = j.value("project_name", "Unknown Project");
172 std::string pVer = j.value("version", "1");
173
174 for(auto& p : m_projects) {
175 if(p.path == pathStr) return;
176 }
177
178 m_projects.push_back({pName, pathStr, pVer});
179 saveKnownProjects();
180 log("Added project: %s", pName.c_str());
181
182 } catch(const std::exception& e) {
183 log(LogLevel::ERROR, "Invalid VexProject.json at %s: %s", pathStr.c_str(), e.what());
184 }
185 } else {
186 log(LogLevel::ERROR, "No VexProject.json found at %s", pathStr.c_str());
187 }
188 }
