208 // @brief Creates a mesh-shaped physics component.
209 static PhysicsComponent Mesh(MeshComponent& mesh, BodyType bodyType = BodyType::STATIC, float mass = 1.0f, float friction = 0.5f, float bounce = 0.1f) {
210 PhysicsComponent pc;
211 pc.shape = ShapeType::MESH;
212 pc.bodyType = bodyType;
213 pc.mass = mass;
214 pc.friction = friction;
215 pc.bounce = bounce;
216 if (!mesh.meshData.submeshes.empty()) {
217 pc.meshVertices.clear();
218 pc.meshIndices.clear();
219
220 size_t vertexOffset = 0;
221
222 for (const auto& sm : mesh.meshData.submeshes) {
223 for (const auto& v : sm.vertices) {
224 pc.meshVertices.push_back(v.position);
225 }
226 for (auto idx : sm.indices) {
227 pc.meshIndices.push_back(static_cast<uint32_t>(idx + vertexOffset));
228 }
229 vertexOffset += sm.vertices.size();
230 }
231 }
232 return pc;
233 }
