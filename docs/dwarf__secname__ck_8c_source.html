45/* Mainly important for Elf */
46
47int
48_dwarf_startswith(const char * input,const char* ckfor)
49{
50 size_t cklen = strlen(ckfor);
51
52 if (! strncmp(input,ckfor,cklen)) {
53 return TRUE;
54 }
55 return FALSE;
56}

58static int
59_dwarf_startswithx(const char * input,const char* ckfor)
60{
61 size_t cklen = strlen(ckfor);
62 int res = 0;
63
64 res = strncmp(input,ckfor,cklen);
65 return res;
66}

96/* These help us ignore some sections that are
97 irrelevant to libdwarf. */
98int
99_dwarf_ignorethissection(const char *scn_name)
100{
101 int l = 0;
102 int h = sizeof(nonsec)/sizeof(const char *) -1;
103 int res = 0;
104
105 while (l <= h) {
106 int m = (l+h)/2;
107 const char *s = nonsec[m];
108
109 res = _dwarf_startswithx(scn_name,s);
110 if (res < 0) {
111 h = m -1;
112 } else if (res > 0) {
113 l = m + 1;
114 } else {
115 return TRUE;
116 }
117 }
118 return FALSE;
119}
