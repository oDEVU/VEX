76 void PhysicsSystem::update(float deltaTime, entt::registry& registry) {
77 if (!m_physicsSystem) return;
78
79 m_accumulator += deltaTime;
80 while (m_accumulator >= m_fixedDt) {
81 m_physicsSystem->Update(m_fixedDt, 1, m_tempAllocator, m_jobSystem);
82 m_accumulator -= m_fixedDt;
83 }
84
85 auto view = registry.view<PhysicsComponent, TransformComponent>();
86 for (auto e : view) {
87 auto& pc = view.get<PhysicsComponent>(e);
88 if (pc.bodyId.IsInvalid()){
89 CreateBodyForEntity(e, registry, pc);
90 }else{
91 SyncBodyToTransform(e, registry, pc.bodyId);
92 }
93 }

96 std::optional<JPH::BodyID> PhysicsSystem::CreateBodyForEntity(entt::entity e, entt::registry& r, PhysicsComponent& pc) {
97 if (!m_physicsSystem) return std::nullopt;
98 if (!r.all_of<TransformComponent>(e)) return std::nullopt;
99
100 auto& t = r.get<TransformComponent>(e);
101 JPH::Vec3 pos(t.position.x, t.position.y, t.position.z);
102 JPH::Quat rot = EulerToQuat(t.rotation);
103
104 JPH::Ref<JPH::Shape> shape;
105 if (pc.shape == ShapeType::BOX)
106 shape = new JPH::BoxShape(JPH::Vec3(pc.boxHalfExtents.x, pc.boxHalfExtents.y, pc.boxHalfExtents.z));
107 else if (pc.shape == ShapeType::SPHERE)
108 shape = new JPH::SphereShape(pc.sphereRadius);
109 else if (pc.shape == ShapeType::CAPSULE)
110 shape = new JPH::CapsuleShape(pc.capsuleRadius, pc.capsuleHeight);
111 else if (pc.shape == ShapeType::CYLINDER)
112 shape = new JPH::CylinderShape(pc.cylinderRadius, pc.cylinderHeight);
113
114 JPH::EMotionType motion = pc.isStatic ? JPH::EMotionType::Static : JPH::EMotionType::Dynamic;
115 JPH::BodyCreationSettings settings(shape, JPH::RVec3(pos), rot, motion, pc.objectLayer);
116 settings.mOverrideMassProperties = JPH::EOverrideMassProperties::CalculateInertia;
117 settings.mMassPropertiesOverride.mMass = pc.mass;
118
119 auto& bodyInterface = m_physicsSystem->GetBodyInterface();
120 JPH::BodyID bodyId = bodyInterface.CreateAndAddBody(settings, JPH::EActivation::DontActivate);
121 if (bodyId.IsInvalid()) return std::nullopt;
122
123 bodyInterface.SetFriction(bodyId, pc.friction);
124
125 pc.bodyId = bodyId;
126 bodyInterface.ActivateBody(bodyId);
127 return bodyId;
128 }

138 JPH::Quat PhysicsSystem::EulerToQuat(const glm::vec3& eulerDeg) {
139 glm::mat4 mat(1.0f);
140 mat = glm::rotate(mat, glm::radians(eulerDeg.y), glm::vec3(0.0f, 1.0f, 0.0f)); // Y first
141 mat = glm::rotate(mat, glm::radians(eulerDeg.x), glm::vec3(1.0f, 0.0f, 0.0f)); // X
142 mat = glm::rotate(mat, glm::radians(eulerDeg.z), glm::vec3(0.0f, 0.0f, 1.0f)); // Z last
143 glm::quat q = glm::quat_cast(mat);
144 return JPH::Quat(q.x, q.y, q.z, q.w);
145 }

147 glm::vec3 PhysicsSystem::QuatToEuler(const JPH::Quat& q) {
148 glm::quat gq(q.GetW(), q.GetX(), q.GetY(), q.GetZ());
149 glm::mat4 mat = glm::mat4_cast(gq);
150 float ex, ey, ez;

153 //log("Quaterion: {%f,%f,%f,%f}", q.GetW(), q.GetX(), q.GetY(), q.GetZ());
154 //log("Euler: {%f,%f,%f}", glm::degrees(ex), glm::degrees(ey), glm::degrees(ez));
155
