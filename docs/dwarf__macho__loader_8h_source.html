306/*
307* The load commands directly follow the mach_header.
308* The total size of all
309* of the commands is given by the sizeofcmds
310* field in the mach_header. All
311* load commands must have as their first two
312* fields cmd and cmdsize. The cmd
313* field is filled in with a constant for that command type.
314* Each command type
315* has a structure specifically for it.
316* The cmdsize field is the size in bytes
317* of the particular load command structure
318* plus anything that follows it that
319* is a part of the load command (i.e.
320* section structures, strings, etc.). To
321* advance to the next load command the
322* cmdsize can be added to the offset or
323* pointer of the current load command.
324* The cmdsize for 32-bit architectures
325* MUST be a multiple of 4 bytes and for
326* 64-bit architectures MUST be a multiple
327* of 8 bytes (these are forever the maximum
328* alignment of any load commands).
329* The padded bytes must be zero.
330* All tables in the object file must also
331* follow these rules so the file can be memory mapped.
332* Otherwise the pointers
333* to these tables will not work well or at all
334* on some machines. With all
335* padding zeroed like objects will compare byte for byte.
336*/
337struct load_command {
338 TYP(cmd,4); /* type of load command */
339 TYP(cmdsize,4); /* total size of command in bytes */
340};

590/*
591* A segment is made up of zero or more sections.
592* Non-MH_OBJECT files have
593* all of their segments with the proper sections
594* in each, and padded to the
595* specified segment alignment when produced by
596* the link editor. The first
597* segment of a MH_EXECUTE and MH_FVMLIB format file
598* contains the mach_header
599* and load commands of the object file before its
600* first section. The zero
601* fill sections are always last in their segment
602* (in all formats). This
603* allows the zeroed segment padding to be mapped
604* into memory where zero fill
605* sections might be. The gigabyte zero fill sections,
606* those with the section
607* type S_GB_ZEROFILL, can only be in a segment
608* with sections of this type.
609* These segments are then placed after all other segments.
610*
611* The MH_OBJECT format has all of its sections in one segment for
612* compactness. There is no padding to a specified
613* segment boundary and the
614* mach_header and load commands are not part of the segment.
615*
616* Sections with the same section name, sectname,
617* going into the same segment,
618* segname, are combined by the link editor.
619* The resulting section is aligned
620* to the maximum alignment of the combined sections
621* and is the new section's
622* alignment. The combined sections are aligned to
623* their original alignment in
624* the combined section. Any padded bytes to get
625* the specified alignment are
626* zeroed.
627*
628* The format of the relocation entries referenced by
629* the reloff and nreloc
630* fields of the section structure for mach
631* object files is described in the
632* header file <reloc.h>.
633*/
634struct section { /* for 32-bit architectures */
635 char sectname[16]; /* name of this section */
636 char segname[16]; /* segment this section goes in */
637 TYP(addr,4); /* memory address of this section */
638 TYP(size,4); /* size in bytes of this section */
639 TYP(offset,4); /* file offset of this section */
640 TYP(align,4); /* section alignment (power of 2) */
641 TYP(reloff,4); /* file offset of relocation entries */
642 TYP(nreloc,4); /* number of relocation entries */
643 TYP(flags,4); /* flags (section type and attributes)*/
644 TYP(reserved1,4); /* reserved (for offset or index) */
645 TYP(reserved2,4); /* reserved (for count or sizeof) */
646};

649 char sectname[16]; /* name of this section */
650 char segname[16]; /* segment this section goes in */
651 TYP(addr,8); /* memory address of this section */
652 TYP(size,8); /* size in bytes of this section */
653 TYP(offset,4); /* file offset of this section */
654 TYP(align,4); /* section alignment (power of 2) */
655 TYP(reloff,4); /* file offset of relocation entries */
656 TYP(nreloc,4); /* number of relocation entries */
657 TYP(flags,4); /* flags (section type and attributes)*/

663/*

665* separated into two parts a section
666* type and section attributes.
667* The section types are mutually exclusive (it
668* can only have one type) but the section attributes
669* are not (it may have more
