24static unsigned long total_alloc;
25static unsigned long alloc_count;
26static unsigned long largest_alloc;
27static unsigned long free_count;
28#define xmalloc malloc
29#define xrealloc realloc
30#define xcalloc calloc
31#define xfree free

33void * _libdwarf_malloc(size_t s)
34{
35 ++alloc_count;
36 total_alloc += s;
37 if ((unsigned long)s > largest_alloc) {
38 largest_alloc = s;
39 printf("LIBDWARF_MALLOC line %d largest_alloc %lu\n",
40 __LINE__,
41 largest_alloc);
42 fflush(stdout);
43#if 0 /* For specific malloc issue detection. Ignore. */
44 if (s > 1000000000) {
45 abort();
46 }
47#endif
48 }
49 {
50 void *r = xmalloc(s);
51 printf("LIBDWARF_MALLOC alloc malloc ret %p size %lu\n",r,
52 (unsigned long)s);
53 fflush(stdout);
54 return r;
55 }
56}
57void * _libdwarf_calloc(size_t n, size_t s)
58{
59 ++alloc_count;
60 total_alloc += n*s;
61 if (n > largest_alloc ) {
62 largest_alloc = n*s;
63 printf("LIBDWARF_MALLOC line %d largest_alloc %lu "
64 " n=%lu s=%lu\n",
65 __LINE__,
66 largest_alloc,
67 (unsigned long)n,
68 (unsigned long)s);
69 fflush(stdout);
70 }
71 if (s > largest_alloc ) {
72 largest_alloc = n*s;
73 printf("LIBDWARF_MALLOC line %d largest_alloc %lu "
74 " n=%lu s=%lu\n",
75 __LINE__,
76 largest_alloc,
77 (unsigned long)n,
78 (unsigned long)s);
79 fflush(stdout);
80 }
81 if ((n * s) > largest_alloc) {
82 largest_alloc = s*n;
83 printf("LIBDWARF_MALLOC line %d largest_alloc %lu "
84 " n=%lu s=%lu\n",
85 __LINE__,
86 largest_alloc,
87 (unsigned long)n,
88 (unsigned long)s);
89 fflush(stdout);
90 }
91 {
92 void *r = xcalloc(n,s);
93 printf("LIBDWARF_MALLOC alloc calloc ret %p size %lu\n",r,
94 (unsigned long)s);
95 fflush(stdout);
96 return r;
97 }
98}
99void * _libdwarf_realloc(void * p, size_t s)
100{
101 ++alloc_count;
102 total_alloc += s;
103 if ((s) > largest_alloc) {
104 largest_alloc = s;
105 printf("LIBDWARF_MALLOC line %d largest_alloc %lu\n",
106 __LINE__,
107 largest_alloc);
108 fflush(stdout);
109 }
110 {
111 void *r = xrealloc(p,s);
112 printf("LIBDWARF_MALLOC alloc realloc ret %p size %lu\n",r,
113 (unsigned long)s);
114 fflush(stdout);
115 return r;
116 }
117}
118void _libdwarf_free(void *p)
119{
120 free_count++;
121 printf("LIBDWARF_MALLOC alloc free %p\n",p);
122 fflush(stdout);
123 xfree(p);
124}
125void _libdwarf_finish(void)
126{
127 printf("LIBDWARF_MALLOC at finish total alloc %lu\n",
128 total_alloc);
129 printf("LIBDWARF_MALLOC at finish largest alloc %lu\n",
130 largest_alloc);
131 printf("LIBDWARF_MALLOC at finish alloc count %lu\n",
132 alloc_count);
133 printf("LIBDWARF_MALLOC at finish free count %lu\n",
134 free_count);
135 fflush(stdout);
136 total_alloc = 0;
137 largest_alloc = 0;
138 alloc_count = 0;
139 free_count = 0;
140}
