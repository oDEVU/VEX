88 void GatherHierarchy(vex::GameObject* obj, std::vector<vex::GameObject*>& outList) {
89 if (!obj) return;
90 outList.push_back(obj);
91
92 auto& registry = m_engine.getRegistry();
93 auto view = registry.view<vex::TransformComponent>();
94 entt::entity parentID = obj->GetEntity();
95
96 for (auto entity : view) {
97 if (view.get<vex::TransformComponent>(entity).getParent() == parentID) {
98 auto* childObj = m_engine.getSceneManager()->GetScene(m_sceneName)->GetGameObjectByEntity(entity);
99 if (childObj) {
100 GatherHierarchy(childObj, outList);
101 }
102 }
103 }
104 }

106 public:
108 DeleteCommand(vex::Engine& engine, std::vector<vex::GameObject*> rootsToDelete)
109 : m_engine(engine)
110 {
111 m_sceneName = engine.getSceneManager()->getLastSceneName();
112
113 std::vector<vex::GameObject*> allObjects;
114 for (auto* root : rootsToDelete) {
115 GatherHierarchy(root, allObjects);
116 }
117
118 for (auto* obj : allObjects) {
119 SerializedObject data;
120 data.name = obj->GetComponent<vex::NameComponent>().name;
121 data.type = obj->getObjectType();
122 data.originalID = obj->GetEntity();
123
124 if (obj->HasComponent<vex::TransformComponent>()) {
125 data.originalParentID = obj->GetComponent<vex::TransformComponent>().getParent();
126 } else {
127 data.originalParentID = entt::null;
128 }
129
130 const auto& regNames = vex::ComponentRegistry::getInstance().getRegisteredNames();
131 for (const auto& compName : regNames) {
132 nlohmann::json json = vex::ComponentRegistry::getInstance().saveComponent(*obj, compName);
133 if (!json.is_null()) {
134 data.components[compName] = json;
135 }
136 }
137 m_serializedData.push_back(data);
138 }
139 }
140
142 void Execute() override {
143 if (m_runtimeIDs.empty()) return;
144
145 auto* scene = m_engine.getSceneManager()->GetScene(m_sceneName);
146 if (!scene) return;
147
148 m_engine.WaitForGpu();
149
150 for (entt::entity entity : m_runtimeIDs) {
151 if (m_engine.getRegistry().valid(entity)) {
152 auto* obj = scene->GetGameObjectByEntity(entity);
153 if (obj) {
154 scene->DestroyGameObject(obj);
155 } else {
156 m_engine.getRegistry().destroy(entity);
157 }
158 }
159 }
160
161 m_runtimeIDs.clear();
162 m_engine.refreshForObject();
163 }
164
166 void Undo() override {
167 auto* scene = m_engine.getSceneManager()->GetScene(m_sceneName);
168 if (!scene) return;
169
170 std::unordered_map<entt::entity, entt::entity> oldToNewIDMap;
171 m_runtimeIDs.clear();
172
173 for (const auto& data : m_serializedData) {
174 vex::GameObject* newObj = vex::GameObjectFactory::getInstance().create(data.type, m_engine, data.name);
175 if (!newObj) continue;
176
177 for (const auto& [compName, jsonData] : data.components) {
178 vex::ComponentRegistry::getInstance().loadComponent(*newObj, compName, jsonData);
179 }
180
181 scene->AddEditorGameObject(newObj);
182
183 entt::entity newID = newObj->GetEntity();
184 oldToNewIDMap[data.originalID] = newID;
185 m_runtimeIDs.push_back(newID);
186 }
187
188 auto& registry = m_engine.getRegistry();
189
190 for (const auto& data : m_serializedData) {
191 if (!oldToNewIDMap.contains(data.originalID)) continue;
192
193 entt::entity newEntityID = oldToNewIDMap[data.originalID];
194
195 if (registry.all_of<vex::TransformComponent>(newEntityID)) {
196 auto& tc = registry.get<vex::TransformComponent>(newEntityID);
197 entt::entity oldParent = data.originalParentID;
198
199 if (oldToNewIDMap.contains(oldParent)) {
200 tc.setParent(oldToNewIDMap[oldParent]);
201 } else {
202 tc.setParent(oldParent);
203 }
204 }
205 }
206
207 m_engine.refreshForObject();
208 }
209 };
