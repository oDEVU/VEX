141 glm::quat getWorldQuaternion() const {
142 return glm::normalize(glm::quat_cast(matrix()));
143 }

147 void setWorldQuaternion(glm::quat targetWorldQuat) {
148 if (parent != entt::null && m_registry.valid(parent) && m_registry.all_of<TransformComponent>(parent)) {
149 glm::quat parentWorldQuat = glm::quat_cast(m_registry.get<TransformComponent>(parent).matrix());
150 glm::quat parentInverse = glm::inverse(parentWorldQuat);
151 m_rotationQuat = parentInverse * targetWorldQuat;
152 } else {
153 m_rotationQuat = targetWorldQuat;
154 }
155 m_rotationQuat = glm::normalize(m_rotationQuat);
156 }

161 glm::mat4 matrix() const {
162 glm::mat4 local = glm::mat4(1.0f);
163 local = glm::translate(local, position); // Najpierw translacja
164 local *= glm::mat4_cast(m_rotationQuat); // Następnie rotacja (mnożenie macierzy)
165 local = glm::scale(local, scale); // Na koniec skala (mnożenie macierzy)
166
167 if (parent != entt::null && m_registry.valid(parent) && m_registry.all_of<TransformComponent>(parent)) {
168 return m_registry.get<TransformComponent>(parent).matrix() * local;
169 }
170 return local;
171 }
