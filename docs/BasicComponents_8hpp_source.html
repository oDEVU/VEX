154 return glm::normalize(glm::quat_cast(matrix()));

159 void setWorldQuaternion(glm::quat targetWorldQuat) {
160 if (parent != entt::null && m_registry.valid(parent) && m_registry.all_of<TransformComponent>(parent)) {
161 glm::quat parentWorldQuat = glm::quat_cast(m_registry.get<TransformComponent>(parent).matrix());
162 glm::quat parentInverse = glm::inverse(parentWorldQuat);
163 m_rotationQuat = parentInverse * targetWorldQuat;
164 } else {
165 m_rotationQuat = targetWorldQuat;
166 }
167 m_rotationQuat = glm::normalize(m_rotationQuat);
168 }

173 glm::mat4 matrix() const {
174 glm::mat4 local = glm::mat4(1.0f);
175 local = glm::translate(local, position); // Najpierw translacja
176 local *= glm::mat4_cast(m_rotationQuat); // Następnie rotacja (mnożenie macierzy)
177 local = glm::scale(local, scale); // Na koniec skala (mnożenie macierzy)
178
179 if (parent != entt::null && m_registry.valid(parent) && m_registry.all_of<TransformComponent>(parent)) {
180 return m_registry.get<TransformComponent>(parent).matrix() * local;
181 }
182 return local;
183 }

203 worldScale *= m_registry.get<TransformComponent>(current).scale;
204 current = m_registry.get<TransformComponent>(current).parent;
