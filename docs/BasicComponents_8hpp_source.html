222 void setWorldQuaternion(glm::quat targetWorldQuat) {
223 enableLastTransformed();
224 if (parent != entt::null && m_registry->valid(parent) && m_registry->all_of<TransformComponent>(parent)) {
225 glm::quat parentWorldQuat = m_registry->get<TransformComponent>(parent).getWorldQuaternion();

228 m_rotationQuat = parentInverse * targetWorldQuat;

237 glm::mat4 recalculateMatrix(){
238 glm::mat4 local = glm::mat4(1.0f);
239 local = glm::translate(local, position);
240 local *= glm::mat4_cast(m_rotationQuat);
241 local = glm::scale(local, scale);
242
243 if (parent != entt::null && m_registry && m_registry->valid(parent) && m_registry->all_of<TransformComponent>(parent)) {
244 cachedMatrix = m_registry->get<TransformComponent>(parent).recalculateMatrix() * local;
245 }else{
246 cachedMatrix = local;
247 }
248
249 return cachedMatrix;
250 }
