201 void setWorldQuaternion(glm::quat targetWorldQuat) {
202 if (parent != entt::null && m_registry->valid(parent) && m_registry->all_of<TransformComponent>(parent)) {
203 glm::quat parentWorldQuat = m_registry->get<TransformComponent>(parent).getWorldQuaternion();
204
205 glm::quat parentInverse = glm::inverse(parentWorldQuat);
206 m_rotationQuat = parentInverse * targetWorldQuat;

216 glm::mat4 matrix() const {
217 glm::mat4 local = glm::mat4(1.0f);
218 local = glm::translate(local, position); // Najpierw translacja
219 local *= glm::mat4_cast(m_rotationQuat); // Następnie rotacja (mnożenie macierzy)
220 local = glm::scale(local, scale); // Na koniec skala (mnożenie macierzy)
221
222 if (parent != entt::null && m_registry && m_registry->valid(parent) && m_registry->all_of<TransformComponent>(parent)) {
223 return m_registry->get<TransformComponent>(parent).matrix() * local;
224 }
225 return local;
226 }
