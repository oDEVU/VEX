116 glm::mat4 matrix() const {
117 glm::mat4 local(1.0f);
118 local = glm::translate(local, position);
119 local = glm::rotate(local, glm::radians(rotation.x), {1, 0, 0});
120 local = glm::rotate(local, glm::radians(rotation.y), {0, 1, 0});
121 local = glm::rotate(local, glm::radians(rotation.z), {0, 0, 1});
122 local = glm::scale(local, scale);
123 if (parent != entt::null && m_registry.valid(parent) && m_registry.all_of<TransformComponent>(parent)) {
124 return m_registry.get<TransformComponent>(parent).matrix() * local;
125 }
126 return local;
127 }

221 glm::vec3 getForwardVector() const {
222 float pitch = glm::radians(getWorldRotation().x);
223 float yaw = glm::radians(getWorldRotation().y);
224
225 return glm::normalize(glm::vec3(
226 cos(yaw) * cos(pitch),
227 sin(pitch),
228 sin(yaw) * cos(pitch)
229 ));
230 }
231
234 glm::vec3 getRightVector() {
235 glm::vec3 forward = getForwardVector();
236 glm::vec3 worldUp = glm::vec3(0.0f, 1.0f, 0.0f);
237 return glm::normalize(glm::cross(forward, worldUp));
238 }
239
242 glm::vec3 getUpVector() {
243 glm::vec3 forward = getForwardVector();
244 glm::vec3 right = getRightVector();
245 return glm::normalize(glm::cross(right, forward));
246 }
