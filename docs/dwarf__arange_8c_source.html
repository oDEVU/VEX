53static void
54free_aranges_chain(Dwarf_Debug dbg, Dwarf_Chain head)
55{
56 Dwarf_Chain cur = head;
57 Dwarf_Chain next = 0;
58
59 if (!head) {
60 return;
61 }
62 next = head->ch_next;
63 for ( ;cur; cur = next) {
64 void *item = cur->ch_item;
65 int type = cur->ch_itemtype;
66
67 next = cur->ch_next;
68 if (item && type) {
69 dwarf_dealloc(dbg,item,type);
70 cur->ch_item = 0;
71 dwarf_dealloc(dbg,cur,DW_DLA_CHAIN);
72 }
73 }
74}

76/* Common code for two user-visible routines to share.
77 Errors here result in memory leaks, but errors here
78 are serious (making aranges unusable) so we assume
79 callers will not repeat the error often or mind the leaks.
80*/
81static int
82_dwarf_get_aranges_list(Dwarf_Debug dbg,
83 Dwarf_Chain * chain_out,
84 Dwarf_Signed * chain_count_out,
85 Dwarf_Error * error)
86{
87 /* Sweeps through the arange. */
88 Dwarf_Small *arange_ptr = 0;
89 Dwarf_Small *arange_ptr_start = 0;
90
91 /* Start of arange header.
92 Used for rounding offset of arange_ptr
93 to twice the tuple size. Libdwarf requirement. */
94 Dwarf_Small *header_ptr = 0;
95
96 /* Version of .debug_aranges header. */
97 Dwarf_Unsigned version = 0;
98
99 /* Offset of current set of aranges into .debug_info. */
100 Dwarf_Off info_offset = 0;
101 /* Size in bytes of addresses in target. */
102 Dwarf_Small address_size = 0;
103 /* Size in bytes of segment offsets in target. */
104 Dwarf_Small segment_sel_size = 0;
105 /* Count of total number of aranges. */
106 Dwarf_Signed arange_count = 0;
107 Dwarf_Arange arange = 0;
108 Dwarf_Unsigned section_size = 0;
109 Dwarf_Byte_Ptr arange_end_section = 0;
110 /* Used to chain Dwarf_Aranges structs. */
111 Dwarf_Chain curr_chain = NULL;
112 Dwarf_Chain head_chain = NULL;
113 Dwarf_Chain *plast = &head_chain;
114
115 if (!dbg->de_debug_aranges.dss_size) {
116 return DW_DLV_NO_ENTRY;
117 }
118 arange_ptr = dbg->de_debug_aranges.dss_data;
119 arange_ptr_start = arange_ptr;
120 section_size = dbg->de_debug_aranges.dss_size;
121 arange_end_section = arange_ptr + section_size;
122
123 do {
124 /* Length of current set of aranges.
125 This is local length, which begins just
126 after the length field itself. */
127 Dwarf_Unsigned area_length = 0;
128 Dwarf_Unsigned remainder = 0;
129 Dwarf_Unsigned range_entry_size = 0;
130 int local_length_size;
131 int local_extension_size = 0;
132 Dwarf_Small *end_this_arange = 0;
133 int res = 0;
134
135 header_ptr = arange_ptr;
136 if (header_ptr >= arange_end_section) {
137 free_aranges_chain(dbg,head_chain);
138 _dwarf_error(dbg, error,DW_DLE_ARANGES_HEADER_ERROR);
139 return DW_DLV_ERROR;
140 }
141 res = _dwarf_read_area_length_ck_wrapper(dbg,&area_length,
142 &arange_ptr,&local_length_size,&local_extension_size,
143 section_size,arange_end_section,error);
144 if (res != DW_DLV_OK) {
145 free_aranges_chain(dbg,head_chain);
146 return res;
147 }
148 /* arange_ptr has been incremented appropriately past
149 the length field by READ_AREA_LENGTH. */
150
151 if (area_length > dbg->de_debug_aranges.dss_size ||
152 (area_length +local_length_size+local_extension_size)
153 > dbg->de_debug_aranges.dss_size ) {
154 free_aranges_chain(dbg,head_chain);
155 _dwarf_error(dbg, error,DW_DLE_ARANGES_HEADER_ERROR);
156 return DW_DLV_ERROR;
157 }
158 if ((area_length + local_length_size + local_extension_size) >
159 dbg->de_debug_aranges.dss_size) {
160 free_aranges_chain(dbg,head_chain);
161 _dwarf_error(dbg, error, DW_DLE_ARANGES_HEADER_ERROR);
162 return DW_DLV_ERROR;
163 }
164
165 end_this_arange = arange_ptr + area_length;
166 if (end_this_arange > arange_end_section) {
167 free_aranges_chain(dbg,head_chain);
168 _dwarf_error(dbg, error,DW_DLE_ARANGES_HEADER_ERROR);
169 return DW_DLV_ERROR;
170 }
171 if (!area_length) {
172 /* We read 4 bytes of zero, so area-length zero.
173 Keep scanning. First seen Nov 27, 2018
174 in GNU-cc in windows dll. */
175 continue;
176 }
177
178 res = _dwarf_read_unaligned_ck_wrapper(dbg,&version,
179 arange_ptr,DWARF_HALF_SIZE,end_this_arange,error);
180 if (res != DW_DLV_OK) {
181 free_aranges_chain(dbg,head_chain);
182 return res;
183 }
184 arange_ptr += DWARF_HALF_SIZE;
185 if (arange_ptr >= end_this_arange) {
186 free_aranges_chain(dbg,head_chain);
187 _dwarf_error(dbg, error, DW_DLE_ARANGES_HEADER_ERROR);
188 return DW_DLV_ERROR;
189 }
190 if (version != DW_ARANGES_VERSION2) {
191 free_aranges_chain(dbg,head_chain);
192 _dwarf_error(dbg, error, DW_DLE_VERSION_STAMP_ERROR);
193 return DW_DLV_ERROR;
194 }
195 res = _dwarf_read_unaligned_ck_wrapper(dbg,&info_offset,
196 arange_ptr,local_length_size,end_this_arange,error);
197 if (res != DW_DLV_OK) {
198 free_aranges_chain(dbg,head_chain);
199 return res;
200 }
201
202 arange_ptr += local_length_size;
203 if (arange_ptr >= end_this_arange) {
204 free_aranges_chain(dbg,head_chain);
205 _dwarf_error(dbg, error, DW_DLE_ARANGES_HEADER_ERROR);
206 return DW_DLV_ERROR;
207 }
208 /* This applies to debug_info only, not to debug_types. */
209 if (info_offset >= dbg->de_debug_info.dss_size) {
210 FIX_UP_OFFSET_IRIX_BUG(dbg, info_offset,
211 "arange info offset.a");
212 if (info_offset >= dbg->de_debug_info.dss_size) {
213 free_aranges_chain(dbg,head_chain);
214 _dwarf_error(dbg, error, DW_DLE_ARANGE_OFFSET_BAD);
215 return DW_DLV_ERROR;
216 }
217 }
218
219 address_size = *(Dwarf_Small *) arange_ptr;
220 if (address_size > sizeof(Dwarf_Addr)) {
221 free_aranges_chain(dbg,head_chain);
222 _dwarf_error(dbg, error, DW_DLE_ADDRESS_SIZE_ERROR);
223 return DW_DLV_ERROR;
224 }
225 if (address_size == 0) {
226 free_aranges_chain(dbg,head_chain);
227 _dwarf_error(dbg, error, DW_DLE_ADDRESS_SIZE_ZERO);
228 return DW_DLV_ERROR;
229 }
230 /* It is not an error if the sizes differ.
231 Unusual, but not an error. */
232 arange_ptr = arange_ptr + sizeof(Dwarf_Small);
233
234 /* The following deref means we better
235 check the pointer for off-end. */
236 if (arange_ptr >= end_this_arange) {
237 free_aranges_chain(dbg,head_chain);
238 _dwarf_error(dbg, error, DW_DLE_ARANGE_OFFSET_BAD);
239 return DW_DLV_ERROR;
240 }
241
242 /* Even DWARF2 had a segment_sel_size field here,
243 meaning
244 size in bytes of a segment selector/descriptor
245 on the target system.
246 In reality it is unlikely any non-zero
247 value will work sensibly for the user. */
248 segment_sel_size = *(Dwarf_Small *) arange_ptr;
249 if (segment_sel_size > 0) {
250 free_aranges_chain(dbg,head_chain);
251 _dwarf_error_string(dbg, error,
252 DW_DLE_SEGMENT_SIZE_BAD,
253 "DW_DLE_SEGMENT_SIZE_BAD: "
254 "segment selector size > 0 is not supported");
255 return DW_DLV_ERROR;
256 }
257 arange_ptr = arange_ptr + sizeof(Dwarf_Small);
258 /* Code below will check for == end_this_arange
259 as appropriate. */
260 if (arange_ptr > end_this_arange) {
261 free_aranges_chain(dbg,head_chain);
262 _dwarf_error(dbg, error, DW_DLE_ARANGE_OFFSET_BAD);
263 return DW_DLV_ERROR;
264 }
265 range_entry_size = 2*address_size + segment_sel_size;
266 /* Round arange_ptr offset to next multiple of
267 address_size. */
268 remainder = (Dwarf_Unsigned) ((arange_ptr - header_ptr) %
269 (range_entry_size));
270 if (remainder != 0) {
271 arange_ptr = arange_ptr + (2 * address_size) - remainder;
272 }
273
274 do {
275 Dwarf_Addr range_address = 0;
276 Dwarf_Unsigned segment_selector = 0;
277 Dwarf_Unsigned range_length = 0;
278 /* For segmented address spaces, the first field to
279 read is a segment selector (new in DWARF4).
280 The version number DID NOT CHANGE from 2, which
281 is quite surprising.
282 Also surprising since the segment_sel_size
283 was always there
284 in the table header! */
285 /* We want to test cu_version here
286 and segment_sel_size, but
287 currently with no way segment_sel_size
288 can be other than zero.
289 We just hope no one using
290 segment_selectors, really. FIXME */
291 res = _dwarf_read_unaligned_ck_wrapper(dbg,&range_address,
292 arange_ptr,address_size,end_this_arange,error);
293 if (res != DW_DLV_OK) {
294 free_aranges_chain(dbg,head_chain);
295 return res;
296 }
297 arange_ptr += address_size;
298
299 res = _dwarf_read_unaligned_ck_wrapper(dbg,&range_length,
300 arange_ptr,address_size,end_this_arange,error);
301 if (res != DW_DLV_OK) {
302 free_aranges_chain(dbg,head_chain);
303 return res;
304 }
305
306 arange_ptr += address_size;
307
308 {
309 /* We used to suppress all-zero entries, but
310 now we return all aranges entries so we show
311 the entire content. March 31, 2010. */
312
313 arange = (Dwarf_Arange)
314 _dwarf_get_alloc(dbg, DW_DLA_ARANGE, 1);
315 if (arange == NULL) {
316 free_aranges_chain(dbg,head_chain);
317 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
318 return DW_DLV_ERROR;
319 }
320
321 arange->ar_segment_selector = segment_selector;
322 arange->ar_segment_selector_size =
323 segment_sel_size;
324 arange->ar_address = range_address;
325 arange->ar_length = range_length;
326 arange->ar_info_offset = info_offset;
327 arange->ar_dbg = dbg;
328 arange_count++;
329
330 curr_chain = (Dwarf_Chain)
331 _dwarf_get_alloc(dbg, DW_DLA_CHAIN, 1);
332 if (curr_chain == NULL) {
333 dwarf_dealloc(dbg,arange,DW_DLA_ARANGE);
334 free_aranges_chain(dbg,head_chain);
335 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
336 return DW_DLV_ERROR;
337 }
338
339 curr_chain->ch_item = arange;
340 curr_chain->ch_itemtype = DW_DLA_ARANGE;
341 (*plast) = curr_chain;
342 plast = &(curr_chain->ch_next);
343 }
344 /* The current set of ranges is terminated by
345 range_address 0 and range_length 0, but that
346 does not necessarily terminate the ranges for this CU!
347 There can be multiple sets in that DWARF
348 does not explicitly forbid multiple sets.
349 DWARF2,3,4 section 7.20
350 We stop short to avoid overrun of the
351 end of the CU. */
352 } while (end_this_arange >= (arange_ptr + range_entry_size));
353
354 /* A compiler could emit some padding bytes here. dwarf2/3
355 (dwarf4 sec 7.20) does not clearly make extra padding
356 bytes illegal. */
357 if (end_this_arange < arange_ptr) {
358 Dwarf_Unsigned pad_count = arange_ptr - end_this_arange;
359 Dwarf_Unsigned offset = arange_ptr - arange_ptr_start;
360 dwarfstring aramsg;
361
362 dwarfstring_constructor(&aramsg);
363 /* Safe. Length strictly limited. */
364 dwarfstring_append_printf_u(&aramsg,
365 "DW_DLE_ARANGE_LENGTH_BAD."
366 " 0x%" DW_PR_XZEROS DW_PR_DUx,
367 pad_count);
368 dwarfstring_append_printf_u(&aramsg,
369 " pad bytes at offset 0x%" DW_PR_XZEROS DW_PR_DUx
370 " in .debug_aranges",
371 offset);
372 dwarf_insert_harmless_error(dbg,
373 dwarfstring_string(&aramsg));
374 dwarfstring_destructor(&aramsg);
375 }
376 /* For most compilers, arange_ptr == end_this_arange at
377 this point. But not if there were padding bytes */
378 arange_ptr = end_this_arange;
379 } while (arange_ptr < arange_end_section);
380
381 if (arange_ptr != arange_end_section) {
382 free_aranges_chain(dbg,head_chain);
383 _dwarf_error(dbg, error, DW_DLE_ARANGE_DECODE_ERROR);
384 return DW_DLV_ERROR;
385 }
386 *chain_out = head_chain;
387 *chain_count_out = arange_count;
388 return DW_DLV_OK;
389}

391/*
392 This function returns the count of the number of
393 aranges in the .debug_aranges section. It sets
394 aranges to point to a block of Dwarf_Arange's
395 describing the arange's. It returns DW_DLV_ERROR
396 on error.
397
398 Must be identical in most aspects to
399 dwarf_get_aranges_addr_offsets!
400
401*/
402int
403dwarf_get_aranges(Dwarf_Debug dbg,
404 Dwarf_Arange ** aranges,
405 Dwarf_Signed * returned_count, Dwarf_Error * error)
406{
407 /* Count of total number of aranges. */
408 Dwarf_Signed arange_count = 0;
409
410 Dwarf_Arange *arange_block = 0;
411
412 /* Used to chain Dwarf_Aranges structs. */
413 Dwarf_Chain curr_chain = NULL;
414 Dwarf_Chain head_chain = NULL;
415 Dwarf_Signed i = 0;
416 int res = DW_DLV_ERROR;
417
418 /* ***** BEGIN CODE ***** */
419
420 CHECK_DBG(dbg,error,"dwarf_get_aranges()");
421 res = _dwarf_load_section(dbg, &dbg->de_debug_aranges, error);
422 if (res != DW_DLV_OK) {
423 return res;
424 }
425 /* aranges points in to info, so if info needs expanding
426 we have to load it. */
427 res = _dwarf_load_debug_info(dbg, error);
428 if (res != DW_DLV_OK) {
429 return res;
430 }
431
432 res = _dwarf_get_aranges_list(dbg,&head_chain,
433 &arange_count,error);
434 if (res != DW_DLV_OK) {
435 free_aranges_chain(dbg,head_chain);
436 return res;
437 }
438
439 arange_block = (Dwarf_Arange *)
440 _dwarf_get_alloc(dbg, DW_DLA_LIST, arange_count);
441 if (arange_block == NULL) {
442 _dwarf_error(dbg, error, DW_DLE_ALLOC_FAIL);
443 free_aranges_chain(dbg,head_chain);
444 return DW_DLV_ERROR;
445 }
446
447 /* See also free_aranges_chain() above */
448 curr_chain = head_chain;
449 for (i = 0; i < arange_count; i++) {
450 Dwarf_Chain prev = 0;
451
452 /* Copies pointers. No dealloc of ch_item, */
453 *(arange_block + i) = curr_chain->ch_item;
454 curr_chain->ch_item = 0;
455 prev = curr_chain;
456 curr_chain = curr_chain->ch_next;
457 dwarf_dealloc(dbg, prev, DW_DLA_CHAIN);
458 }
459 *aranges = arange_block;
460 *returned_count = (arange_count);
461 return DW_DLV_OK;
462}

563/*
564 This function takes a pointer to a block
565 of Dwarf_Arange's, and a count of the
566 length of the block. It checks if the
567 given address is within the range of an
568 address range in the block. If yes, it
569 returns the appropriate Dwarf_Arange.
570 If no, it returns DW_DLV_NO_ENTRY;
571 On error it returns DW_DLV_ERROR.
572*/
573int
574dwarf_get_arange(Dwarf_Arange * aranges,
575 Dwarf_Unsigned arange_count,
576 Dwarf_Addr address,
577 Dwarf_Arange * returned_arange, Dwarf_Error * error)
578{
579 Dwarf_Arange curr_arange = 0;
580 Dwarf_Unsigned i = 0;
581
582 if (!aranges) {
583 _dwarf_error(NULL, error, DW_DLE_ARANGES_NULL);
584 return DW_DLV_ERROR;
585 }
586 for (i = 0; i < arange_count; i++) {
587 curr_arange = *(aranges + i);
588 if (address >= curr_arange->ar_address &&
589 address <
590 curr_arange->ar_address + curr_arange->ar_length) {
591 *returned_arange = curr_arange;
592 return DW_DLV_OK;
593 }
594 }
595 return DW_DLV_NO_ENTRY;
596}

598/*
599 This function takes an Dwarf_Arange,
600 and returns the offset of the first
601 die in the compilation-unit that the
602 arange belongs to. Returns DW_DLV_ERROR
603 on error.
604
605 For an arange, the cu_die can only be from debug_info,
606 not debug_types, it seems.
607*/
608int
609dwarf_get_cu_die_offset(Dwarf_Arange arange,
610 Dwarf_Off * returned_offset,
611 Dwarf_Error * error)
612{
613 Dwarf_Debug dbg = 0;
614 Dwarf_Off offset = 0;
615 Dwarf_Unsigned headerlen = 0;
616 int cres = 0;
617
618 if (arange == NULL) {
619 _dwarf_error(NULL, error, DW_DLE_ARANGE_NULL);
620 return DW_DLV_ERROR;
621 }
622 dbg = arange->ar_dbg;
623 offset = arange->ar_info_offset;
624 /* This applies to debug_info only, not to debug_types. */
625 if (!dbg->de_debug_info.dss_data) {
626 int res = _dwarf_load_debug_info(dbg, error);
627
628 if (res != DW_DLV_OK) {
629 return res;
630 }
631 }
632 cres = _dwarf_length_of_cu_header(dbg, offset,
633 TRUE, &headerlen,error);
634 if (cres != DW_DLV_OK) {
635 return cres;
636 }
637 *returned_offset = headerlen + offset;
638 return DW_DLV_OK;
639}

641/* This function takes an Dwarf_Arange,
642 and returns the offset of the CU header
643 in the compilation-unit that the
644 arange belongs to. Returns DW_DLV_ERROR
645 on error.
646 Ensures .debug_info loaded so
647 the cu_offset is meaningful. */
648int
649dwarf_get_arange_cu_header_offset(Dwarf_Arange arange,
650 Dwarf_Off * cu_header_offset_returned,
651 Dwarf_Error * error)
652{
653 Dwarf_Debug dbg = 0;
654 if (arange == NULL) {
655 _dwarf_error(NULL, error, DW_DLE_ARANGE_NULL);
656 return DW_DLV_ERROR;
657 }
658 dbg = arange->ar_dbg;
659 /* This applies to debug_info only, not to debug_types. */
660 /* Like dwarf_get_arange_info_b() this ensures debug_info loaded:
661 the cu_header is in debug_info and will be used else
662 we would not call dwarf_get_arange_cu_header_offset. */
663 if (!dbg->de_debug_info.dss_data) {
664 int res = _dwarf_load_debug_info(dbg, error);

673/*
674 This function takes a Dwarf_Arange, and returns
675 TRUE if it is not NULL. It also stores the start
676 address of the range in *start, the length of the
677 range in *length, and the offset of the first die
678 in the compilation-unit in *cu_die_offset. It
679 returns FALSE on error.
680 If cu_die_offset returned ensures .debug_info loaded so
681 the cu_die_offset is meaningful.
682
683 New for DWARF4, entries may have segment information.
684 *segment is only meaningful
685 if *segment_entry_size is non-zero.
686 But segment_selectors are not fully defined so
687 a non-zero segment_entry_size is not actually
688 usable. */
689int
690dwarf_get_arange_info_b(Dwarf_Arange arange,
691 Dwarf_Unsigned* segment,
692 Dwarf_Unsigned* segment_entry_size,
693 Dwarf_Addr * start,
694 Dwarf_Unsigned* length,
695 Dwarf_Off * cu_die_offset,
696 Dwarf_Error * error)
697{
698 if (arange == NULL) {
699 _dwarf_error(NULL, error, DW_DLE_ARANGE_NULL);
700 return DW_DLV_ERROR;
701 }
702
703 if (segment != NULL) {
704 *segment = arange->ar_segment_selector;
705 }
706 if (segment_entry_size != NULL) {
707 *segment_entry_size = arange->ar_segment_selector_size;
708 }
709 if (start != NULL)
710 *start = arange->ar_address;
711 if (length != NULL)
712 *length = arange->ar_length;
713 if (cu_die_offset != NULL) {
714 Dwarf_Debug dbg = arange->ar_dbg;
715 Dwarf_Off offset = arange->ar_info_offset;
716 Dwarf_Unsigned headerlen = 0;
717 int cres = 0;
718
719 /* This applies to debug_info only, not to debug_types. */
720 if (!dbg->de_debug_info.dss_data) {
721 int res = _dwarf_load_debug_info(dbg, error);
722 if (res != DW_DLV_OK) {
723 return res;
724 }
725 }
726 cres = _dwarf_length_of_cu_header(dbg, offset,
727 TRUE, &headerlen,error);
728 if (cres != DW_DLV_OK) {
729 return cres;
730 }
731 *cu_die_offset = offset + headerlen;
732
733 }
734 return DW_DLV_OK;
735}
