55struct generic_ehdr {
56 unsigned char ge_ident[EI_NIDENT];
57 Dwarf_Unsigned ge_type;
58 Dwarf_Unsigned ge_machine;
59 Dwarf_Unsigned ge_version;
60 Dwarf_Unsigned ge_entry;
61 Dwarf_Unsigned ge_phoff;
62 Dwarf_Unsigned ge_shoff;
63 Dwarf_Unsigned ge_flags;
64 Dwarf_Unsigned ge_ehsize;
65 Dwarf_Unsigned ge_phentsize;
66 Dwarf_Unsigned ge_phnum;
67 Dwarf_Unsigned ge_shentsize;
68 Dwarf_Unsigned ge_shnum;
69 /* if ge_shnum >= 0xff00 SHN_LORESERVE
70 Once section zero is read we put the sh_size
71 member as the true count and set ge_shnum_in_shnum TRUE.
72 ge_shnum_extended is TRUE if the object used the extension
73 mechanism */
74 Dwarf_Bool ge_shnum_in_shnum;
75 Dwarf_Bool ge_shnum_extended;
76
77 /* if section num of sec strings >= 0xff SHN_LORESERVE
78 this member holds SHN_XINDEX (0xffff) and the real
79 section string index is the sh_link value of section
80 0. ge_sstrndx_extended is TRUE if the object used
81 the extension mechanism */
82 Dwarf_Unsigned ge_shstrndx;
83 Dwarf_Bool ge_strndx_in_strndx;
84 Dwarf_Bool ge_strndx_extended;
85};

96struct generic_shdr {
97 Dwarf_Unsigned gh_secnum;
98 Dwarf_Unsigned gh_name;
99 const char * gh_namestring;
100 Dwarf_Unsigned gh_type;
101 Dwarf_Unsigned gh_flags;
102 Dwarf_Unsigned gh_addr;
103 Dwarf_Unsigned gh_offset;
104 Dwarf_Unsigned gh_size;
105 Dwarf_Unsigned gh_link;
106 /* Section index (in an SHT_REL or SHT_RELA section)
107 of the target section from gh_link. Otherwise 0. */
108 Dwarf_Unsigned gh_reloc_target_secnum;
109 Dwarf_Unsigned gh_info;
110 Dwarf_Unsigned gh_addralign;
111 Dwarf_Unsigned gh_entsize;
112
113 /* Zero unless content read in. Malloc space
114 of size gh_size, in bytes.
115 or if load type Dwarf_Alloc_Mmap
116 gh_content is a pointer to the user data.
117 free() or
118 unmap this this if not null*/
119 char * gh_content;
120 /* Actual load type */
121 enum Dwarf_Sec_Alloc_Pref gh_load_type;
122 /* Normally TRUE, meaning do free/munmap in dwarf_finish() */
123 Dwarf_Small gh_was_alloc;
124 char * gh_mmap_realarea;
125 Dwarf_Unsigned gh_computed_mmaplen;
126
127 /* If a .rel or .rela section this will point
128 to generic relocation records if such
129 have been loaded.
130 free() this if not null, always malloc. */
131 Dwarf_Unsigned gh_relcount;
132 struct generic_rela * gh_rels;
133
134 /* For SHT_GROUP based grouping, which
135 group is this section in. 0 unknown,
136 1 DW_GROUP_NUMBER_BASE base DWARF,
137 2 DW_GROUPNUMBER_DWO dwo sections, 3+
138 are in an SHT_GROUP. GNU uses this.
139 set with group number (3+) from SHT_GROUP
140 and the flags should have SHF_GROUP set
141 if in SHT_GROUP. Must only be in one group? */
142 Dwarf_Unsigned gh_section_group_number;
143
144 /* Content of an SHT_GROUP section as an array
145 of integers. [0] is the version, which
146 can only be one(1) . Free this on dwarf_finish() */
147 Dwarf_Unsigned * gh_sht_group_array;
148 /* Number of elements in the gh_sht_group_array. */
149 Dwarf_Unsigned gh_sht_group_array_count;
150
151 /* TRUE if .debug_info .eh_frame etc. */
152 char gh_is_dwarf;
153};
