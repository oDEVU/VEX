108public:
109 using ComponentLoader = std::function<void(GameObject&, const nlohmann::json&)>;

111 using ComponentInspector = std::function<void(GameObject&)>;

113 using ComponentCreator = std::function<void(GameObject&)>;
114
115 static ComponentRegistry& getInstance();

126 template<typename T>
127 void registerComponent(const std::string& name, bool isDynamic = false) {
128 loaders[name] = [](GameObject& obj, const nlohmann::json& j) {
129 if (!obj.HasComponent<T>()) {
130 if constexpr (std::is_default_constructible_v<T>) {
131 obj.AddComponent<T>(T());
132 }
133 }
134 if (obj.HasComponent<T>()) {
135 T& comp = obj.GetComponent<T>();
136 j.get_to(comp);
137 }
138 };
139
140 savers[name] = [](GameObject& obj) -> nlohmann::json {
141 if (obj.HasComponent<T>()) {
142 return obj.GetComponent<T>();
143 }
144 return nullptr;
145 };
146
147 inspectors[name] = &GenericComponentInspector<T>;
148
149 checkers[name] = [](const GameObject& obj) {
150 return obj.HasComponent<T>();
151 };
152
153 creators[name] = [](GameObject& obj) {
154 if (!obj.HasComponent<T>()) {
155 if constexpr (std::is_default_constructible_v<T>) {
156 obj.AddComponent<T>(T());
157 } else {
158 log("Error: Component '%s' is not default constructible.", typeid(T).name());
159 }
160 }
161 };
162
163 registeredNames.push_back(name);
164
165 if (isDynamic) {
166 dynamicComponents.push_back(name);
167 }
168 }
