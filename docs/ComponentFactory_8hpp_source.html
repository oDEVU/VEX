66public:

74 static ComponentRegistry instance;
75 return instance;

78 // Generic Register Function
79 template<typename T>
80 void registerComponent(const std::string& name) {
81 // 1. Loader: Relies on from_json(json, T) being defined (either Auto or Manual)
82 loaders[name] = [](GameObject& obj, const nlohmann::json& j) {
83 if (!obj.HasComponent<T>()) {
84 // Try default construction. Note: TransformComponent needs handling if no default ctor.
85 // Assuming you added a default ctor or logic to handle it.
86 if constexpr (std::is_default_constructible_v<T>) {
87 obj.AddComponent<T>(T());
88 }
89 }
90 if (obj.HasComponent<T>()) {
91 T& comp = obj.GetComponent<T>();
92 j.get_to(comp); // Calls global from_json
93 }
94 };
95
96 // 2. Saver: Relies on to_json(json, T) being defined
97 savers[name] = [](GameObject& obj) -> nlohmann::json {
98 if (obj.HasComponent<T>()) {
99 return obj.GetComponent<T>(); // Calls global to_json
100 }
101 return nullptr;
102 };
103
104 // 3. Inspector: Registers the Generic template.
105 // If you specialize this template for a type (like PhysicsComponent),
106 // the compiler will pick your specialization automatically.
107 inspectors[name] = &GenericComponentInspector<T>;
108
109 checkers[name] = [](const GameObject& obj) {
110 return obj.HasComponent<T>();
111 };
112
113 creators[name] = [](GameObject& obj) {
114 if (!obj.HasComponent<T>()) {
115 if constexpr (std::is_default_constructible_v<T>) {
116 obj.AddComponent<T>(T());
117 } else {
118 log("Error: Component '%s' is not default constructible.", typeid(T).name());
119 }
120 }
121 };
122
123 registeredNames.push_back(name);
124 }
125
126 bool loadComponent(GameObject& obj, const std::string& type, const nlohmann::json& json) {
127 auto it = loaders.find(type);
128 if (it != loaders.end()) {
129 it->second(obj, json);
130 return true;
131 }
132 log("Error: Component type '%s' not registered", type.c_str());
133 return false;
134 }
135
136 nlohmann::json saveComponent(GameObject& obj, const std::string& type) {
137 if (savers.find(type) != savers.end()) return savers[type](obj);
138 return nullptr;
139 }
140
141 void drawInspectorForObject(GameObject& obj) {
142 for (auto& [name, inspector] : inspectors) {
143 inspector(obj);
144 }
145 }
146
147 std::vector<std::string> getMissingComponents(const GameObject& obj) {
148 std::vector<std::string> missing;
149 for (const auto& name : registeredNames) {
150 if (checkers.find(name) != checkers.end() && !checkers[name](obj)) {
151 missing.push_back(name);
152 }
153 }
154 return missing;
155 }
156
157 void createComponent(GameObject& obj, const std::string& name) {
158 if (creators.find(name) != creators.end()) {
159 creators[name](obj);
160 } else {
161 log("Error: Cannot create component '%s', not registered.", name.c_str());
162 }
163 }
164
165 const std::unordered_map<std::string, ComponentInspector>& getInspectors() const {
166 return inspectors;
167 }
168
169 const std::vector<std::string>& getRegisteredNames() const { return registeredNames; }
170
171private:
172 ComponentRegistry() = default;
173 std::vector<std::string> registeredNames;
174 std::unordered_map<std::string, ComponentLoader> loaders;
175 std::unordered_map<std::string, ComponentSaver> savers;
176 std::unordered_map<std::string, ComponentInspector> inspectors;
177 std::unordered_map<std::string, ComponentChecker> checkers;
178 std::unordered_map<std::string, ComponentCreator> creators;
179};
