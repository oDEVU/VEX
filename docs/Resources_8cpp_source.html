508 log("Creating %d texture descriptor sets", m_r_context.MAX_FRAMES_IN_FLIGHT * m_r_context.MAX_TEXTURES);
509
510 m_textureDescriptorSets.resize(m_r_context.MAX_FRAMES_IN_FLIGHT * m_r_context.MAX_TEXTURES);

513 m_r_context.MAX_FRAMES_IN_FLIGHT * m_r_context.MAX_TEXTURES,
514 m_r_context.textureDescriptorSetLayout
515 );

518 allocInfo.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;

522
523 VkResult result = vkAllocateDescriptorSets(m_r_context.device, &allocInfo, m_textureDescriptorSets.data());
524 if (result != VK_SUCCESS) {
525 throw_error("Failed to allocate texture descriptor sets: " + std::to_string(result));
526 }
527
528 VkDescriptorImageInfo defaultImageInfo{};
529 defaultImageInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
530 defaultImageInfo.sampler = m_textureSampler;
531 defaultImageInfo.imageView = getTextureView("default");
532
533 std::vector<VkWriteDescriptorSet> writes;
534 for (uint32_t frame = 0; frame < m_r_context.MAX_FRAMES_IN_FLIGHT; ++frame) {
535 for (uint32_t texIdx = 0; texIdx < m_r_context.MAX_TEXTURES; ++texIdx) {
536 VkWriteDescriptorSet write{};
537 write.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
538 write.dstSet = getTextureDescriptorSet(frame, texIdx);
539 write.dstBinding = 0;
540 write.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
541 write.descriptorCount = 1;
542 write.pImageInfo = &defaultImageInfo;
543 writes.push_back(write);
544 }
545 }
546 vkUpdateDescriptorSets(m_r_context.device, writes.size(), writes.data(), 0, nullptr);
547 log("Initialized %d texture descriptor sets with default texture", writes.size());
548 }

581 void VulkanResources::loadTexture(const std::string& path, const std::string& name, VirtualFileSystem *vfs) {
582 std::string fullPath = path;//"Assets/" + std::string(path.c_str());
583
584 if (m_r_context.textureIndices.size() >= m_r_context.MAX_TEXTURES) {
585 SDL_LogError(SDL_LOG_CATEGORY_ERROR,
586 "Maximum texture count (%u) reached!",
587 m_r_context.MAX_TEXTURES);
588 return;
589 }
590
591 // Use VFS to check if file exists and get data
592 if (!vfs->file_exists(fullPath)) {
593 SDL_LogError(SDL_LOG_CATEGORY_ERROR, "Texture file not found in VFS: %s", fullPath.c_str());
594 throw_error("Missing texture in VFS: " + fullPath);
595 }
596
597 log("VFS loading image...");
598 int texWidth, texHeight, texChannels;
599
600 // Load file data through VFS
601 auto fileData = vfs->load_file(fullPath); // This returns std::unique_ptr<FileData>
602 if (!fileData) {
603 SDL_LogError(SDL_LOG_CATEGORY_ERROR, "VFS failed to load texture: %s", fullPath.c_str());
604 throw_error("VFS failed to load texture: " + fullPath);
605 }
606
607 // Use stbi_load_from_memory instead of stbi_load
608 stbi_uc* pixels = stbi_load_from_memory(
609 reinterpret_cast<const stbi_uc*>(fileData->data.data()), // fileData is a pointer
610 static_cast<int>(fileData->size), // fileData->size is member
611 &texWidth, &texHeight, &texChannels, STBI_rgb_alpha
612 );
613
614 if (!pixels) {
615 SDL_LogError(SDL_LOG_CATEGORY_ERROR, "STBI failed: %s", stbi_failure_reason());
616 throw_error("Failed to load texture pixels from VFS data");
617 }
618
619 log("Image loaded from VFS: %dx%d, %d channels", texWidth, texHeight, texChannels);
620
621 VkDeviceSize imageSize = texWidth * texHeight * 4;
622
623 if (!pixels) {
624 throw_error("Failed to load texture: " + fullPath);
625 }
626
627 log("Creating staging buffer (%zu bytes)...",
628 static_cast<size_t>(texWidth * texHeight * 4));
629 VkBuffer stagingBuffer;
630 VmaAllocation stagingAlloc;
631 VkBufferCreateInfo bufferInfo{};
632 bufferInfo.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
633 bufferInfo.size = imageSize;
634 bufferInfo.usage = VK_BUFFER_USAGE_TRANSFER_SRC_BIT;
635
636 VmaAllocationCreateInfo allocInfo{};
637 allocInfo.usage = VMA_MEMORY_USAGE_CPU_TO_GPU;
638
639 vmaCreateBuffer(m_r_context.allocator, &bufferInfo, &allocInfo, &stagingBuffer, &stagingAlloc, nullptr);
640
641 void* data;
642 vmaMapMemory(m_r_context.allocator, stagingAlloc, &data);
643 memcpy(data, pixels, static_cast<size_t>(imageSize));
644 vmaUnmapMemory(m_r_context.allocator, stagingAlloc);
645 stbi_image_free(pixels);
646
647 log("Creating Vulkan image...");
648 VkImageCreateInfo imageInfo{};
649 imageInfo.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
650 imageInfo.imageType = VK_IMAGE_TYPE_2D;
651 imageInfo.extent = {static_cast<uint32_t>(texWidth), static_cast<uint32_t>(texHeight), 1};
652 imageInfo.mipLevels = 1;
653 imageInfo.arrayLayers = 1;
654 imageInfo.format = VK_FORMAT_R8G8B8A8_SRGB;
655 imageInfo.tiling = VK_IMAGE_TILING_OPTIMAL;
656 imageInfo.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
657 imageInfo.usage = VK_IMAGE_USAGE_TRANSFER_DST_BIT | VK_IMAGE_USAGE_SAMPLED_BIT;
658 imageInfo.samples = VK_SAMPLE_COUNT_1_BIT;
659
660 VmaAllocationCreateInfo imageAllocInfo{};
661 imageAllocInfo.usage = VMA_MEMORY_USAGE_GPU_ONLY;
662
663 VkImage textureImage;
664 VmaAllocation textureAlloc;
665 vmaCreateImage(m_r_context.allocator, &imageInfo, &imageAllocInfo, &textureImage, &textureAlloc, nullptr);
666
667 VkCommandBuffer commandBuffer = m_r_context.beginSingleTimeCommands();
668
669 VkImageMemoryBarrier barrier{};
670 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
671 barrier.oldLayout = VK_IMAGE_LAYOUT_UNDEFINED;
672 barrier.newLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
673 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
674 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
675 barrier.image = textureImage;
676 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
677 barrier.subresourceRange.baseMipLevel = 0;
678 barrier.subresourceRange.levelCount = 1;
679 barrier.subresourceRange.baseArrayLayer = 0;
680 barrier.subresourceRange.layerCount = 1;
681 barrier.srcAccessMask = 0;
682 barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
683
684
685 vkCmdPipelineBarrier(commandBuffer,
686 VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT,
687 VK_PIPELINE_STAGE_TRANSFER_BIT,
688 0,
689 0, nullptr,
690 0, nullptr,
691 1, &barrier);
692
693 VkBufferImageCopy region{};
694 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
695 region.imageSubresource.mipLevel = 0;
696 region.imageSubresource.baseArrayLayer = 0;
697 region.imageSubresource.layerCount = 1;
698 region.imageExtent = imageInfo.extent;
699
700
701 vkCmdCopyBufferToImage(commandBuffer,
702 stagingBuffer,
703 textureImage,
704 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
705 1,
706 &region);
707
708 barrier.oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
709 barrier.newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
710 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
711 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
712
713 vkCmdPipelineBarrier(commandBuffer,
714 VK_PIPELINE_STAGE_TRANSFER_BIT,
715 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT,
716 0,
717 0, nullptr,
718 0, nullptr,
719 1, &barrier);
720
721 m_r_context.endSingleTimeCommands(commandBuffer);
722
723 vkDeviceWaitIdle(m_r_context.device);
724
725 vmaDestroyBuffer(m_r_context.allocator, stagingBuffer, stagingAlloc);
726
727 VkImageViewCreateInfo viewInfo{};
728 viewInfo.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
729 viewInfo.image = textureImage;
730 viewInfo.viewType = VK_IMAGE_VIEW_TYPE_2D;
731 viewInfo.format = VK_FORMAT_R8G8B8A8_SRGB;
732 viewInfo.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
733 viewInfo.subresourceRange.levelCount = 1;
734 viewInfo.subresourceRange.layerCount = 1;
735
736 VkImageView textureView;
737 log("Creating vulkan texture view...");
738 if (vkCreateImageView(m_r_context.device, &viewInfo, nullptr, &textureView) != VK_SUCCESS) {
739 throw_error("Failed to create texture image view!");
740 }
741
742 if (m_r_context.textureIndices.contains(name)) {
743 log("Texture '%s' already exists at index %u", name.c_str(), m_r_context.textureIndices[name]);
744 return;
745 }
746
747 m_r_context.textureIndices[name] = m_r_context.nextTextureIndex++;
748 log("Assigned texture '%s' to index %u", name.c_str(), m_r_context.textureIndices[name]);
749
750 m_textures[name] = textureView;
751 m_textureImages[name] = textureImage;
752 m_textureAllocations[name] = textureAlloc;
753 m_textureViews[name] = textureView;
754
755 if (m_r_context.textureIndices.size() >= m_r_context.MAX_TEXTURES) {
756 throw_error("Exceeded maximum texture count");
757 }
758 m_r_context.textureIndices[name] = m_r_context.textureIndices.size();
759 log("Assigned texture '%s' to index %d", name.c_str(), m_r_context.textureIndices[name]);
760
761 VkDescriptorImageInfo imageDescInfo{};
762 imageDescInfo.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
763 imageDescInfo.sampler = m_textureSampler;
764 imageDescInfo.imageView = textureView;
765
766 for (uint32_t frame = 0; frame < m_r_context.MAX_FRAMES_IN_FLIGHT; ++frame) {
767 VkWriteDescriptorSet write{};
768 write.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
769 write.dstSet = getTextureDescriptorSet(frame, m_r_context.textureIndices[name]);
770 write.dstBinding = 0;
771 write.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
772 write.descriptorCount = 1;
773 write.pImageInfo = &imageDescInfo;
774
775 vkUpdateDescriptorSets(m_r_context.device, 1, &write, 0, nullptr);
776 }
777 log("Updated texture descriptors for '%s'", name.c_str());
778 }
779 void VulkanResources::unloadTexture(const std::string& name) {
780 if (name == "default") return;
781
782 auto it = m_textures.find(name);
783 if (it == m_textures.end()){
784 log("Texture %s not loaded", name.c_str());
785 return;
786 }
787
788 vkDeviceWaitIdle(m_r_context.device);
789
790 if (m_textureViews[name] != VK_NULL_HANDLE) {
791 vkDestroyImageView(m_r_context.device, m_textureViews[name], nullptr);
792 m_textureViews[name] = VK_NULL_HANDLE;
793 }
794
795 if (m_textureImages[name] != VK_NULL_HANDLE && m_textureAllocations[name] != VK_NULL_HANDLE) {
796 vmaDestroyImage(m_r_context.allocator, m_textureImages[name], m_textureAllocations[name]);
797 m_textureImages[name] = VK_NULL_HANDLE;
798 }
799
800 m_textures.erase(name);
801 m_textureImages.erase(name);
802 m_textureAllocations.erase(name);
803 m_textureViews.erase(name);
804 log("Texture %s unloaded", name.c_str());
805 }
