113/* The following is essentially identical
114 to every crc32 anyone uses up to trivial
115 differences in starting conditions.
116 The algorithm appears on
117 https://en.wikipedia.org/wiki/Cyclic_redundancy_check
118 in a simple logic loop (written in no actual
119 language but English) and the following is a translation
120 into C. Just like every other crc calculation.
121
122 There is no error indicated if buf does not point
123 to a readable area of memory at least len bytes long!
124
125 The unsigned int return will not get the same CRC as
126 Linux/Macos if int/unsigned-int is 16 bits (Windows),
127 but there the crc does not matter in practice.
128 The results on big-endian are not the same
129 as little-endian machines.*/
130unsigned int
131dwarf_basic_crc32 (const unsigned char *buf,
132 unsigned long len,
133 unsigned int crc)
134{
135 const unsigned char *end = 0;
136
137 if (!buf) {
138 return 0;
139 }
140 crc = ~crc;
141 for (end = buf + len; buf < end; ++buf)
142 crc = crc32_table[(crc ^ *buf) & 0xff] ^ (crc >> 8);
143 return ~crc;
144}
