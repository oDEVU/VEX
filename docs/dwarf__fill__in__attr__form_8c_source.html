52static void
53build_alloc_ab_error(Dwarf_Debug dbg,
54 Dwarf_Unsigned count,
55 const char *fieldname,
56 Dwarf_Error *error)
57{
58 dwarfstring m;
59
60 dwarfstring_constructor(&m);
61 dwarfstring_append_printf_s(&m,
62 "DW_DLE_ALLOC_FAIL :"
63 " Attempt to malloc space for %s ",
64 (char *)fieldname);
65 dwarfstring_append_printf_u(&m,
66 " with %u entries failed.",
67 count);
68 _dwarf_error_string(dbg,error,DW_DLE_ALLOC_FAIL,
69 dwarfstring_string(&m));
70 dwarfstring_destructor(&m);
71}

73/*
74 This is a pre-scan of the abbrev/form list.
75 We will not handle DW_FORM_indirect here as that
76 accesses data outside of the abbrev section.
77*/
78int
79_dwarf_fill_in_attr_form_abtable(Dwarf_CU_Context context,
80 Dwarf_Byte_Ptr abbrev_ptr,
81 Dwarf_Byte_Ptr abbrev_end,
82 Dwarf_Abbrev_List abbrev_list,
83 Dwarf_Error *error)
84{
85 Dwarf_Debug dbg = 0;
86 Dwarf_Unsigned i = 0;
87
88 dbg = context->cc_dbg;
89 abbrev_list->abl_attr = (Dwarf_Half*)
90 calloc(abbrev_list->abl_abbrev_count,
91 SIZEOFT16);
92 if (!abbrev_list->abl_attr) {
93 build_alloc_ab_error(dbg,abbrev_list->abl_abbrev_count,
94 "abbrev_list->abl_attr",error);
95 return DW_DLV_ERROR;
96 }
97 abbrev_list->abl_form = (Dwarf_Half *)
98 calloc(abbrev_list->abl_abbrev_count,
99 SIZEOFT16);
100 if (!abbrev_list->abl_form) {
101 build_alloc_ab_error(dbg,abbrev_list->abl_abbrev_count,
102 "abbrev_list->abl_form",error);
103 return DW_DLV_ERROR;
104 }
105 if (abbrev_list->abl_implicit_const_count > 0) {
106 abbrev_list->abl_implicit_const = (Dwarf_Signed *)
107 calloc(abbrev_list->abl_abbrev_count,
108 sizeof(Dwarf_Signed));
109 if (!abbrev_list->abl_implicit_const) {
110 build_alloc_ab_error(dbg,abbrev_list->abl_abbrev_count,
111 "abbrev_list->abl_implicit_const",error);
112 return DW_DLV_ERROR;
113 }
114 }
115
116 for (i = 0; i < abbrev_list->abl_abbrev_count; ++i) {
117 Dwarf_Unsigned attr = 0;
118 Dwarf_Unsigned attr_form = 0;
119 Dwarf_Signed implicit_const = 0;
120 int res = 0;
121
122 res = _dwarf_leb128_uword_wrapper(dbg,
123 &abbrev_ptr,abbrev_end,&attr,error);
124 if (res == DW_DLV_ERROR) {
125 return res;
126 }
127 if (attr > 0xffff) {
128 dwarfstring m;
129 dwarfstring_constructor(&m);
130 dwarfstring_append(&m,
131 "DW_DLE_ATTR_FORM_SIZE_BAD :"
132 " reading Attribute number ");
133 dwarfstring_append(&m," for abbrev list entry"
134 " the ULEB number is too large. Corrupt Dwarf.");
135 _dwarf_error_string(dbg,error,DW_DLE_ATTR_FORM_SIZE_BAD,
136 dwarfstring_string(&m));
137 dwarfstring_destructor(&m);
138 return DW_DLV_ERROR;
139 }
140 abbrev_list->abl_attr[i] = (Dwarf_Half)attr;
141 if (attr > DW_AT_hi_user) {
142 _dwarf_error(dbg, error,DW_DLE_ATTR_CORRUPT);
143 return DW_DLV_ERROR;
144 }
145 res = _dwarf_leb128_uword_wrapper(dbg,
146 &abbrev_ptr,abbrev_end,&attr_form,error);
147 if (res == DW_DLV_ERROR) {
148 return res;
149 }
150 if (attr_form > 0xffff) {
151 _dwarf_error_string(dbg, error,
152 DW_DLE_ATTR_FORM_SIZE_BAD,
153 "DW_DLE_ATTR_FORM_SIZE_BAD :"
154 " reading attr_form of"
155 " an abbrev list entry: "
156 "the ULEB form number is too large "
157 "to be valid. Corrupt Dwarf.");
158 return DW_DLV_ERROR;
159 }
160 if (!_dwarf_valid_form_we_know(attr_form,attr)) {
161 dwarfstring m;
162
163 dwarfstring_constructor(&m);
164 dwarfstring_append(&m,"Reading an abbreviation list "
165 " we find the attribute form pair to be "
166 " impossible or unknown.");
167 dwarfstring_append_printf_u(&m," attr 0x%x ",attr);
168 dwarfstring_append_printf_u(&m," attrform 0x%x ",
169 attr_form);
170 _dwarf_error_string(dbg, error, DW_DLE_UNKNOWN_FORM,
171 dwarfstring_string(&m));
172 dwarfstring_destructor(&m);
173 return DW_DLV_ERROR;
174 }
175 abbrev_list->abl_attr[i] = (Dwarf_Half)attr;
176 abbrev_list->abl_form[i] = (Dwarf_Half)attr_form;
177 if (attr_form == DW_FORM_implicit_const) {
178 res = _dwarf_leb128_sword_wrapper(dbg,
179 &abbrev_ptr,abbrev_end,&implicit_const,error);
180 if (res == DW_DLV_ERROR) {
181 return res;
182 }
183 abbrev_list->abl_implicit_const_count++;
184 abbrev_list->abl_implicit_const[i] = implicit_const;
185 }
186#if 0 /* Do nothing special for DW_FORM_indirect here. Ignore. */
187 if (attr_form == DW_FORM_indirect) {
188 /* Do nothing special here. Do not read
189 from the DIE till reading for
190 a specific DIE, which we are not
191 intending here, we do not know
192 where the DIE is. */
193 }
194#endif
195 }
196 return DW_DLV_OK;
197}
