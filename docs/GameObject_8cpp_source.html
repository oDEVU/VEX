20 void GameObject::Destroy() {

22
23 log("[GameObject] Destroying entity ID: %d", (int)m_entity);
24
25 if (m_engine.getRegistry().valid(m_entity)) {
26 auto view = m_engine.getRegistry().view<TransformComponent>();
27 for (auto entity : view) {
28 auto& transform = view.get<TransformComponent>(entity);
29 if (transform.getParent() == m_entity) {
30 transform.setParent(entt::null);
31 }
32 }
33 m_engine.getRegistry().destroy(m_entity);
34 }
