55void Engine::run() {
56 Uint64 lastTime = SDL_GetPerformanceCounter();
57
58 while (m_running) {
59
60 Uint64 now = SDL_GetPerformanceCounter();
61 float deltaTime = (float)((now - lastTime) / (float)SDL_GetPerformanceFrequency());
62 lastTime = now;
63
64 SDL_Event event;
65 while (SDL_PollEvent(&event)) {
66 m_inputSystem->processEvent(event, deltaTime);
67 processEvent(event, deltaTime);
68 m_imgui->processEvent(&event);
69 //m_vexUI->processEvent(event);
70 //
71 auto uiView = m_registry.view<UiComponent>();
72 std::vector<UiComponent> uiObjects;
73 for (auto entity : uiView) {
74 if(uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
75 uiView.get<UiComponent>(entity).m_vexUI->processEvent(event);
76 }
77 }
78 switch (event.type) {
79 case SDL_EVENT_QUIT:
80 m_running = false;
81 break;
82 case SDL_EVENT_DID_ENTER_FOREGROUND:
83 log("Binding window to Vulkan...");
84 m_interface->bindWindow(m_window->GetSDLWindow());
85 break;
86 case SDL_EVENT_WILL_ENTER_BACKGROUND:
87 m_interface->unbindWindow();
88 break;
89 case SDL_EVENT_WINDOW_RESIZED:
90 m_resolutionManager->update();
91 auto renderRes = m_resolutionManager->getRenderResolution();
92 m_interface->setRenderResolution(renderRes);
93 break;
94 }
95 }
96
97 m_inputSystem->update(deltaTime);
98
99 if(m_frame > 0){
100
101 auto uiView = m_registry.view<UiComponent>();
102 std::vector<UiComponent> uiObjects;
103 for (auto entity : uiView) {
104 if(!uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
105 //uiView.get<UiComponent>(entity).m_vexUI = std::make_unique<VexUI>(*m_interface->getContext(), m_vfs.get(), m_interface->getResources());
106 uiView.get<UiComponent>(entity).m_vexUI->init();
107 uiView.get<UiComponent>(entity).m_vexUI->update();
108 }
109 }
110
111 if(!m_paused){
112 update(deltaTime);
113 m_physicsSystem->update(deltaTime, m_registry);
114 }
115 }else{
116
117 auto uiView = m_registry.view<UiComponent>();
118 std::vector<UiComponent> uiObjects;
119 for (auto entity : uiView) {
120 if(!uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
121 //uiView.get<UiComponent>(entity).m_vexUI = std::make_unique<VexUI>(*m_interface->getContext(), m_vfs.get(), m_interface->getResources());
122 uiView.get<UiComponent>(entity).m_vexUI->init();
123 uiView.get<UiComponent>(entity).m_vexUI->update();
124 }
125 }
126
127 beginGame();
128 }
129
130 render();
131 m_frame++;
132 }
133}

161void Engine::render() {
162 //log("Render function called");
163 auto renderRes = m_resolutionManager->getRenderResolution();
164
165 glm::mat4 view = glm::mat4(1.0f);
166 glm::mat4 proj = glm::mat4(1.0f);
167
168 auto cameraEntity = getCamera();
169 auto& transform = m_registry.get<TransformComponent>(cameraEntity);
170 auto& camera = m_registry.get<CameraComponent>(cameraEntity);
171
172 view = glm::lookAt(
173 transform.getWorldPosition(m_registry),
174 transform.getWorldPosition(m_registry) + transform.getForwardVector(m_registry),
175 transform.getUpVector(m_registry)
176 );
177 proj = glm::perspective(
178 glm::radians(camera.fov),
179 renderRes.x / static_cast<float>(renderRes.y),
180 camera.nearPlane,
181 camera.farPlane
182 );
183 proj[1][1] *= -1;
184
185 //log("Calling Renderer::renderFrame()");
186 try{
187 m_interface->getRenderer().renderFrame(view, proj, renderRes, m_registry, *m_imgui, m_frame);
188 } catch (const std::exception& e) {
189 SDL_LogError(SDL_LOG_CATEGORY_ERROR, "Frame not in fact rendered :C");
190 handle_exception(e);
191 }
192 //log("Frame Rendered");
193}
