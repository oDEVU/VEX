71void Engine::run(std::function<void()> onUpdateLoop) {
72 Uint64 lastTime = SDL_GetPerformanceCounter();
73
74 while (m_running) {
75 if (onUpdateLoop) onUpdateLoop();
76
77 Uint64 now = SDL_GetPerformanceCounter();
78 float deltaTime = (float)((now - lastTime) / (float)SDL_GetPerformanceFrequency());
79 lastTime = now;
80
81 SDL_Event event;
82 while (SDL_PollEvent(&event)) {
83 processEvent(event, deltaTime);
84 m_imgui->processEvent(&event);
85 auto uiView = m_registry.view<UiComponent>();
86 std::vector<UiComponent> uiObjects;
87 for (auto entity : uiView) {
88 if(uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
89 uiView.get<UiComponent>(entity).m_vexUI->processEvent(event);
90 }
91 }
92 switch (event.type) {
93 case SDL_EVENT_QUIT:
94 m_running = false;
95 break;
96 case SDL_EVENT_DID_ENTER_FOREGROUND:
97 log("Binding window to Vulkan...");
98 m_interface->bindWindow(m_window->GetSDLWindow());
99 m_internally_paused = false;
100 break;
101 case SDL_EVENT_WILL_ENTER_BACKGROUND:
102 m_interface->unbindWindow();
103 m_internally_paused = true;
104 break;
105 case SDL_EVENT_WINDOW_RESIZED:
106 m_resolutionManager->update();
107 //auto renderRes = m_resolutionManager->getRenderResolution();
108 //m_interface->setRenderResolution(renderRes);
109 break;
110 }
111 }
112
113 auto transformView = m_registry.view<TransformComponent>();
114 for (auto entity : transformView) {
115 if(!transformView.get<TransformComponent>(entity).isReady()){
116 transformView.get<TransformComponent>(entity).setRegistry(m_registry);
117 }
118 }
119
120 update(deltaTime);
121 }
122}

183

197 render();
198 m_frame++;

203void Engine::render() {
204 //log("Render function called");
205 auto renderRes = m_resolutionManager->getRenderResolution();
206 auto cameraEntity = getCamera();
207
208 if (cameraEntity == entt::null) {
209 return;
210 }
211
212 //std::cout << "renderRes x:" << renderRes.x << ", y:" << renderRes.y << std::endl;
213
214 //log("Calling Renderer::renderFrame()");
215 try{
216 vex::SceneRenderData renderData;
217
218 if (!m_interface->getRenderer().beginFrame(renderRes, renderData)) {
219 return;
220 }
221
222 #if DEBUG
223 const std::vector<DebugVertex>* debugLines = nullptr;
224 if(m_renderPhysicsDebug) {
225 auto* dbg = m_interface->getPhysicsDebug();
226 dbg->Clear();
227 m_physicsSystem->setDebugRenderer(dbg);
228 m_physicsSystem->drawDebug();
229 debugLines = &dbg->GetLines();
230 }
231 m_interface->getRenderer().renderScene(renderData, cameraEntity, m_registry, m_frame, debugLines);
232 #else
233 m_interface->getRenderer().renderScene(renderData, cameraEntity, m_registry, m_frame);
234 #endif
235
236 m_interface->getRenderer().composeFrame(renderData, *m_imgui, false);
237 m_interface->getRenderer().endFrame(renderData);
238 } catch (const std::exception& e) {
239 log(LogLevel::ERROR, "Frame render failed");
240 handle_exception(e);
241 }
242 //log("Frame Rendered");
243}
