87void Engine::run(std::function<void()> onUpdateLoop) {
88 Uint64 lastTime = SDL_GetPerformanceCounter();
89
90 while (m_running) {
91 Uint64 frameStart = SDL_GetPerformanceCounter();
92 if (onUpdateLoop) onUpdateLoop();
93
94 SDL_Event event;
95 while (SDL_PollEvent(&event)) {
96 processEvent(event, m_deltaTime);
97 m_imgui->processEvent(&event);
98 auto uiView = m_registry.view<UiComponent>();
99 std::vector<UiComponent> uiObjects;
100 for (auto entity : uiView) {
101 if(uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
102 uiView.get<UiComponent>(entity).m_vexUI->processEvent(event);
103 }
104 }
105 switch (event.type) {
106 case SDL_EVENT_GAMEPAD_ADDED:
107 SDL_OpenGamepad(event.gdevice.which);
108 log("Gamepad connected and opened.");
109 break;
110 case SDL_EVENT_QUIT:
111 m_running = false;
112 break;
113 case SDL_EVENT_DID_ENTER_FOREGROUND:
114 log("Binding window to Vulkan...");
115 m_interface->bindWindow(m_window->GetSDLWindow());
116 m_internally_paused = false;
117 break;
118 case SDL_EVENT_WILL_ENTER_BACKGROUND:
119 m_interface->unbindWindow();
120 m_internally_paused = true;
121 break;
122 case SDL_EVENT_WINDOW_RESIZED:
123 m_resolutionManager->update();
124 //auto renderRes = m_resolutionManager->getRenderResolution();
125 //m_interface->setRenderResolution(renderRes);
126 break;
127 }
128 }
129
130 if (!m_running) break;
131
132 auto transformView = m_registry.view<TransformComponent>();
133 for (auto entity : transformView) {
134 if(!transformView.get<TransformComponent>(entity).isReady()){
135 transformView.get<TransformComponent>(entity).setRegistry(m_registry);
136 }
137 }
138
139 update(m_deltaTime);
140
141 float targetFps = (float)m_targetFps;
142
143 #ifdef __linux__
144 if (m_vsyncEnabled && m_isWayland) {
145 targetFps = m_window->getRefreshRate();
146 }
147 #endif
148
149 if (targetFps > 0) {
150 float targetFrameTime = 1000.0f / targetFps;
151
152 Uint64 frameEnd = SDL_GetPerformanceCounter();
153 float elapsedMS = (frameEnd - frameStart) / (float)SDL_GetPerformanceFrequency() * 1000.0f;
154
155 if (elapsedMS < targetFrameTime) {
156 SDL_Delay(static_cast<Uint32>(targetFrameTime - elapsedMS));
157 }
158 }
159
160 Uint64 now = SDL_GetPerformanceCounter();
161 m_deltaTime = (float)((now - lastTime) / (float)SDL_GetPerformanceFrequency());
162 lastTime = now;
163 }
164
165 if (m_interface) {
166 log("Engine shutdown requested. Waiting for GPU to finish...");
167 m_interface->WaitForGPUToFinish();
168 }
169}

245void Engine::update(float deltaTime) {
246 m_inputSystem->update(deltaTime);
247
248 auto cameraEntity = getCamera();
249 if (cameraEntity != entt::null) {
250 m_audioSystem->Update(cameraEntity);
251 }
252
253 if(m_frame > 0){
254
255 auto uiView = m_registry.view<UiComponent>();
256 std::vector<UiComponent> uiObjects;
257 for (auto entity : uiView) {
258 if(!uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
259 //uiView.get<UiComponent>(entity).m_vexUI = std::make_unique<VexUI>(*m_interface->getContext(), m_vfs.get(), m_interface->getResources());
260 uiView.get<UiComponent>(entity).m_vexUI->init();
261 uiView.get<UiComponent>(entity).m_vexUI->update();
262 }
263 }
264
265 if(!(m_paused || m_internally_paused)){
266 m_sceneManager->scenesUpdate(deltaTime);
267 m_physicsSystem->update(deltaTime);
268 }
269 }else{
270
271 auto uiView = m_registry.view<UiComponent>();
272 std::vector<UiComponent> uiObjects;
273 for (auto entity : uiView) {
274 if(!uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
275 uiView.get<UiComponent>(entity).m_vexUI->init();
276 uiView.get<UiComponent>(entity).m_vexUI->update();
277 }
278 }
279
280 beginGame();
281 }
282
283 if(!m_internally_paused){
284 render();
285 m_frame++;
286 }
287}

290void Engine::render() {
291 //log("Render function called");
292 auto renderRes = m_resolutionManager->getRenderResolution();
293 auto cameraEntity = getCamera();
294
295 if (cameraEntity == entt::null) {
296 return;
297 }
298
299 //std::cout << "renderRes x:" << renderRes.x << ", y:" << renderRes.y << std::endl;
300
301 //log("Calling Renderer::renderFrame()");
302 try{
303 vex::SceneRenderData renderData;
304
305 if (!m_interface->getRenderer().beginFrame(renderRes, renderData)) {
306 return;
307 }
308
309 #if DEBUG
310 const std::vector<DebugVertex>* debugLines = nullptr;
311 if(m_renderPhysicsDebug) {
312 auto* dbg = m_interface->getPhysicsDebug();
313 dbg->Clear();
314 m_physicsSystem->setDebugRenderer(dbg);
315 m_physicsSystem->drawDebug();
316 debugLines = &dbg->GetLines();
317 }
318 m_interface->getRenderer().renderScene(renderData, cameraEntity, m_registry, m_frame, debugLines);
319 #else
320 m_interface->getRenderer().renderScene(renderData, cameraEntity, m_registry, m_frame);
321 #endif
322
323 m_interface->getRenderer().composeFrame(renderData, *m_imgui, false);
324 m_interface->getRenderer().endFrame(renderData);
325 } catch (const std::exception& e) {
326 log(LogLevel::ERROR, "Frame render failed");
327 handle_exception(e);
328 }
329 //log("Frame Rendered");
330}
