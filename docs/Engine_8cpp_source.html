52void Engine::run() {
53 Uint64 lastTime = SDL_GetPerformanceCounter();
54
55 while (m_running) {
56
57 Uint64 now = SDL_GetPerformanceCounter();
58 float deltaTime = (float)((now - lastTime) / (float)SDL_GetPerformanceFrequency());
59 lastTime = now;
60
61 SDL_Event event;
62 while (SDL_PollEvent(&event)) {
63 m_inputSystem->processEvent(event, deltaTime);
64 processEvent(event, deltaTime);
65 m_imgui->processEvent(&event);
66 //m_vexUI->processEvent(event);
67 //
68 auto uiView = m_registry.view<UiComponent>();
69 std::vector<UiComponent> uiObjects;
70 for (auto entity : uiView) {
71 if(uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
72 uiView.get<UiComponent>(entity).m_vexUI->processEvent(event);
73 }
74 }
75 switch (event.type) {
76 case SDL_EVENT_QUIT:
77 m_running = false;
78 break;
79 case SDL_EVENT_DID_ENTER_FOREGROUND:
80 log("Binding window to Vulkan...");
81 m_interface->bindWindow(m_window->GetSDLWindow());
82 break;
83 case SDL_EVENT_WILL_ENTER_BACKGROUND:
84 m_interface->unbindWindow();
85 break;
86 case SDL_EVENT_WINDOW_RESIZED:
87 m_resolutionManager->update();
88 auto renderRes = m_resolutionManager->getRenderResolution();
89 m_interface->setRenderResolution(renderRes);
90 break;
91 }
92 }
93
94 m_inputSystem->update(deltaTime);
95
96 if(m_frame > 0){
97
98 auto uiView = m_registry.view<UiComponent>();
99 std::vector<UiComponent> uiObjects;
100 for (auto entity : uiView) {
101 if(!uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
102 //uiView.get<UiComponent>(entity).m_vexUI = std::make_unique<VexUI>(*m_interface->getContext(), m_vfs.get(), m_interface->getResources());
103 uiView.get<UiComponent>(entity).m_vexUI->init();
104 uiView.get<UiComponent>(entity).m_vexUI->update();
105 }
106 }
107
108 update(deltaTime);
109 }else{
110
111 auto uiView = m_registry.view<UiComponent>();
112 std::vector<UiComponent> uiObjects;
113 for (auto entity : uiView) {
114 if(!uiView.get<UiComponent>(entity).m_vexUI->isInitialized()){
115 //uiView.get<UiComponent>(entity).m_vexUI = std::make_unique<VexUI>(*m_interface->getContext(), m_vfs.get(), m_interface->getResources());
116 uiView.get<UiComponent>(entity).m_vexUI->init();
117 uiView.get<UiComponent>(entity).m_vexUI->update();
118 }
119 }
120
121 beginGame();
122 }
123
124 render();
125 m_frame++;
126 }
127}

155void Engine::render() {
156 //log("Render function called");
157 auto renderRes = m_resolutionManager->getRenderResolution();
158
159 glm::mat4 view = glm::mat4(1.0f);
160 glm::mat4 proj = glm::mat4(1.0f);
161
162 auto cameraEntity = getCamera();
163 auto& transform = m_registry.get<TransformComponent>(cameraEntity);
164 auto& camera = m_registry.get<CameraComponent>(cameraEntity);
165
166 view = glm::lookAt(
167 transform.getWorldPosition(m_registry),
168 transform.getWorldPosition(m_registry) + transform.getForwardVector(m_registry),
169 transform.getUpVector(m_registry)
170 );
171 proj = glm::perspective(
172 glm::radians(camera.fov),
173 renderRes.x / static_cast<float>(renderRes.y),
174 camera.nearPlane,
175 camera.farPlane
176 );
177 proj[1][1] *= -1;
178
179 //log("Calling Renderer::renderFrame()");
180 try{
181 m_interface->getRenderer().renderFrame(view, proj, renderRes, m_registry, *m_imgui, m_frame);
182 } catch (const std::exception& e) {
183 SDL_LogError(SDL_LOG_CATEGORY_ERROR, "Frame not in fact rendered :C");
184 handle_exception(e);
185 }
186 //log("Frame Rendered");
187}
