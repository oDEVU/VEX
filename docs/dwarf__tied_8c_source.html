81/* Tied data Key is Dwarf_Sig8.
82 A hash needed because we are using a hash search
83 here. Would not be needed for the other tree searchs
84 like balanced trees.. */
85DW_TSHASHTYPE
86_dwarf_tied_data_hashfunc(const void *keyp)
87{
88 const struct Dwarf_Tied_Entry_s * enp = keyp;
89 DW_TSHASHTYPE hashv = 0;
90 /* Just take some of the 8 bytes of the signature. */
91 memcpy(&hashv,enp->dt_key.signature,sizeof(hashv));
92 return hashv;
93}

95int
96_dwarf_tied_compare_function(const void *l, const void *r)
97{
98 const struct Dwarf_Tied_Entry_s * lp = l;
99 const struct Dwarf_Tied_Entry_s * rp = r;
100 const char *lcp = (const char *)&lp->dt_key.signature;
101 const char *rcp = (const char *)&rp->dt_key.signature;
102 const char *lcpend = lcp + sizeof(Dwarf_Sig8);
103
104 for ( ;lcp < lcpend; ++lcp,++rcp) {
105 if (*lcp < *rcp) {
106 return -1;
107 }
108 if (*lcp > *rcp) {
109 return 1;
110 }
111 }
112 /* match. */
113 return 0;
114}

124/* This presumes only we are reading the debug_info
125 CUs from tieddbg. That is a reasonable
126 requirement, one hopes.
127 Currently it reads all the tied CUs at once up to
128 the point of finding a match unless there is an error..
129 This the only way we call _dwarf_next_cu_header*( )
130 on the tied file, so safe.
131 */
132static int
133_dwarf_loop_reading_debug_info_for_cu(
134 Dwarf_Debug tieddbg,
135 struct Dwarf_Tied_Entry_s *targsig,
136 Dwarf_Error *error)
137{
138 /* We will not find tied signatures
139 for .debug_addr (or line tables) in .debug_types.
140 it seems. Those signatures point from
141 'normal' to 'dwo/dwp' (DWARF4) */
142 int is_info = TRUE;
143 Dwarf_Unsigned next_cu_offset = 0;
144
145 for (;;) {
146 int sres = DW_DLV_OK;
147 Dwarf_Half cu_type = 0;
148 Dwarf_CU_Context latestcontext = 0;
149 Dwarf_Unsigned cu_header_length = 0;
150 Dwarf_Unsigned abbrev_offset = 0;
151 Dwarf_Half version_stamp = 0;
152 Dwarf_Half address_size = 0;
153 Dwarf_Half extension_size = 0;
154 Dwarf_Half length_size = 0;
155 Dwarf_Sig8 signature;
156 Dwarf_Bool has_signature = FALSE;
157 Dwarf_Unsigned typeoffset = 0;
158
159 memset(&signature,0,sizeof(signature));
160 sres = _dwarf_next_cu_header_internal(tieddbg,
161 is_info,
162 /* no CU die wanted*/ NULL,
163 &cu_header_length, &version_stamp,
164 &abbrev_offset, &address_size,
165 &length_size,&extension_size,
166 &signature, &has_signature,
167 &typeoffset,
168 &next_cu_offset,
169 &cu_type, error);
170 if (sres == DW_DLV_ERROR) {
171 return sres;
172 }
173 if (sres == DW_DLV_NO_ENTRY) {
174 break;
175 }
176
177 latestcontext = tieddbg->de_info_reading.de_cu_context;
178
179 if (has_signature) {
180 void *retval = 0;
181 Dwarf_Sig8 consign;
182 void *entry = 0;
183
184 if (!latestcontext) {
185 /* FAILED might be out of memory.*/
186 return DW_DLV_NO_ENTRY;
187 }
188 consign = latestcontext->cc_signature;
189 entry = _dwarf_tied_make_entry(&consign,latestcontext);
190 if (!entry) {
191 return DW_DLV_NO_ENTRY;
192 }
193 /* Insert this signature and context. */
194 retval = dwarf_tsearch(entry,
195 &tieddbg->de_tied_data.td_tied_search,
196 _dwarf_tied_compare_function);
197 if (!retval) {
198 free(entry);
199 /* FAILED might be out of memory.*/
200 return DW_DLV_NO_ENTRY;
201 } else {
202 struct Dwarf_Tied_Data_s * retent =
203 *(struct Dwarf_Tied_Data_s**) retval;
204 if (retent == entry) {
205 /* we added a record. */
206 int res = _dwarf_tied_compare_function(
207 targsig,entry);
208 if (!res) {
209 /* Found match, stop looping */
210 return DW_DLV_OK;
211 }
212 continue;
213 } else {
214 /* found existing, no add */
215 free(entry);
216 continue;
217 }
218 }
219 }
220 }
221 /* Apparently we never found the sig we are looking for.
222 Pretend ok. Caller will check for success. */
223 return DW_DLV_OK;
224}

226/* If out of memory just return DW_DLV_NO_ENTRY.
227 This ensures all the tied CU contexts have been
228 created though the caller has most likely
229 never tried to read CUs in the tied-file.
230*/
231int
232_dwarf_search_for_signature(Dwarf_Debug tieddbg,
233 Dwarf_Sig8 sig,
234 Dwarf_CU_Context *context_out,
235 Dwarf_Error *error)
236{
237
238 void *entry2 = 0;
239 struct Dwarf_Tied_Entry_s entry;
240 struct Dwarf_Tied_Data_s * tied = &tieddbg->de_tied_data;
241 int res = 0;
242
243 if (!tied->td_tied_search) {
244 dwarf_initialize_search_hash(&tied->td_tied_search,
245 _dwarf_tied_data_hashfunc,0);
246 if (!tied->td_tied_search) {
247 return DW_DLV_NO_ENTRY;
248 }
249 }
250 entry.dt_key = sig;
251 entry.dt_context = 0;
252 entry2 = dwarf_tfind(&entry,
253 &tied->td_tied_search,
254 _dwarf_tied_compare_function);
255 if (entry2) {
256 struct Dwarf_Tied_Entry_s *e2 =
257 *(struct Dwarf_Tied_Entry_s **)entry2;
258 *context_out = e2->dt_context;
259 return DW_DLV_OK;
260 }
261 /* We now ensure all tieddbg CUs signatures
262 are in the td_tied_search,
263 The caller is NOT doing
264 info section read operations
265 on the tieddbg in this (tied)dbg, so it
266 cannot goof up their _dwarf_next_cu_header*(). */
267 res = _dwarf_loop_reading_debug_info_for_cu(tieddbg,&entry,
268 error);
269 if (res == DW_DLV_ERROR) {
270 return res;
271 }
272 entry2 = dwarf_tfind(&entry,
273 &tied->td_tied_search,
274 _dwarf_tied_compare_function);
275 if (entry2) {
276 struct Dwarf_Tied_Entry_s *e2 =
277 *(struct Dwarf_Tied_Entry_s **)entry2;
278 *context_out = e2->dt_context;
279 return DW_DLV_OK;
280 }
281 return DW_DLV_NO_ENTRY;
282}
