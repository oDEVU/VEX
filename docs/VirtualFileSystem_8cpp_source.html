38bool VirtualFileSystem::load_vpk_file(const std::string& vpk_path) {
39 m_loaded_vpk = std::make_unique<LoadedVPK>();
40 m_loaded_vpk->file_path = vpk_path;
41
42 m_loaded_vpk->file_stream.open(vpk_path, std::ios::binary);
43 if (!m_loaded_vpk->file_stream) {
44 //std::cerr << "Failed to open VPK file: " << vpk_path << std::endl;
45 log(LogLevel::ERROR, "Failed to open VPK file: %s", vpk_path.c_str());
46 throw_error("Failed to open VPK file.");
47 m_loaded_vpk.reset();
48 return false;
49 }
50
51 m_loaded_vpk->file_stream.read(
52 reinterpret_cast<char*>(&m_loaded_vpk->header),
53 sizeof(VPKHeader)
54 );
55
56 if (std::strncmp(m_loaded_vpk->header.magic, "VPAK", 4) != 0) {
57 //std::cerr << "Invalid VPK file: bad magic" << std::endl;
58 throw_error("Invalid VPK file: bad magic");
59 m_loaded_vpk.reset();
60 return false;
61 }
62
63 m_loaded_vpk->entries.resize(m_loaded_vpk->header.file_count);
64 m_loaded_vpk->file_stream.seekg(sizeof(VPKHeader));
65 m_loaded_vpk->file_stream.read(
66 reinterpret_cast<char*>(m_loaded_vpk->entries.data()),
67 m_loaded_vpk->header.file_count * sizeof(VPKFileEntry)
68 );
69
70 m_loaded_vpk->file_stream.seekg(m_loaded_vpk->header.names_offset);
71 for (uint32_t i = 0; i < m_loaded_vpk->header.file_count; ++i) {
72 std::string name;
73 std::getline(m_loaded_vpk->file_stream, name, '\0');
74 m_loaded_vpk->file_names.push_back(name);
75 }
76
77 //std::cout << "Loaded VPK with " << m_loaded_vpk->header.file_count << " files" << std::endl;
78 log("Loaded VPK with %u files", m_loaded_vpk->header.file_count);
79
80 // DEBUG: List all files in VPK
81 log("Files in VPK:");
82 for (const auto& name : m_loaded_vpk->file_names) {
83 log("File: %s", name.c_str());
84 }
85
86 return true;
87}

89const VirtualFileSystem::VPKFileEntry* VirtualFileSystem::find_file_entry(const std::string& virtual_path) {
90 if (!m_loaded_vpk) return nullptr;

92 std::string clean_virtual_path = clean_path(virtual_path);
93 auto it = std::find(
94 m_loaded_vpk->file_names.begin(),
95 m_loaded_vpk->file_names.end(),
96 clean_virtual_path
97 );
98
99 if (it == m_loaded_vpk->file_names.end()) {
100 return nullptr;
101 }
102
103 size_t index = std::distance(m_loaded_vpk->file_names.begin(), it);

107std::unique_ptr<VirtualFileSystem::FileData> VirtualFileSystem::load_file(const std::string& virtual_path) {
108 std::string clean_virtual_path = clean_path(virtual_path);
109
110 if (m_use_packed_assets && m_loaded_vpk) {
111 const auto* entry = find_file_entry(clean_virtual_path);
112 if (!entry) {
113 return nullptr;
114 }
115
116 // Read data from VPK
117 auto file_data = std::make_unique<FileData>();
118 file_data->data.resize(entry->data_size);
119 file_data->size = entry->data_size;
120
121 m_loaded_vpk->file_stream.seekg(m_loaded_vpk->header.data_offset + entry->data_offset);
122 m_loaded_vpk->file_stream.read(
123 reinterpret_cast<char*>(file_data->data.data()),
124 entry->data_size
125 );
126
127 return file_data;
128 } else {
129 std::string full_path;
130
131 #ifdef _WIN32
132 full_path = clean_virtual_path;
133 #else
134 full_path = "/" + clean_virtual_path;
135 #endif
136
137 if (!fs::exists(full_path)) {
138 return nullptr;
139 }
140
141 std::ifstream file(full_path, std::ios::binary);
142 if (!file) {
143 return nullptr;
144 }
145
146 file.seekg(0, std::ios::end);
147 size_t size = file.tellg();
148 file.seekg(0, std::ios::beg);
149
150 auto file_data = std::make_unique<FileData>();
151 file_data->data.resize(size);
152 file_data->size = size;
153
154 file.read(reinterpret_cast<char*>(file_data->data.data()), size);
155 return file_data;
156 }

161
162 if (m_use_packed_assets && m_loaded_vpk) {
163 const auto* entry = find_file_entry(clean_virtual_path);
164 if (!entry) {
165 return nullptr;
166 }
167
168 // Read the file data into memory
169 std::vector<char> buffer(entry->data_size);
170
171 // Use the main stream but protect with seek/read
172 std::lock_guard<std::mutex> lock(stream_mutex); // Add a mutex to your class
173 m_loaded_vpk->file_stream.seekg(m_loaded_vpk->header.data_offset + entry->data_offset);
174 m_loaded_vpk->file_stream.read(buffer.data(), entry->data_size);
175
176 if (m_loaded_vpk->file_stream.gcount() != entry->data_size) {
177 return nullptr;

236 std::vector<std::string> result;
237 std::string clean_dir = clean_path(virtual_dir);
238
239 if (m_use_packed_assets && m_loaded_vpk) {
240 for (const auto& name : m_loaded_vpk->file_names) {
241 if (clean_dir.empty() || name.find(clean_dir) == 0) {
242 result.push_back(name);

244 }
245 } else {
246 std::string full_dir = m_base_path + "/Assets/" + clean_dir;
247 if (fs::exists(full_dir) && fs::is_directory(full_dir)) {
248 for (const auto& entry : fs::recursive_directory_iterator(full_dir)) {
249 if (entry.is_regular_file()) {
250 std::string relative_path = fs::relative(entry.path(), m_base_path + "/Assets").generic_string();
251 result.push_back(relative_path);
252 }
253 }
254 }
255 }
256
257 return result;

260std::string VirtualFileSystem::clean_path(const std::string& path) {
261 std::string result = path;
262
263 std::replace(result.begin(), result.end(), '\\', '/');
264
265 if (result.find("./") == 0) {
266 result = result.substr(2);
267 }
268
269 if (!result.empty() && result[0] == '/') {
270 result = result.substr(1);
271 }
272
273 if (result.find("Assets/") == 0 && m_use_packed_assets) {
274 result = result.substr(7);
275 }
276
277 return result;
278}

280std::string VirtualFileSystem::resolve_relative_path(const std::string& base_path, const std::string& relative_path) {
281 std::filesystem::path base(base_path);
282 std::filesystem::path relative(relative_path);
283
